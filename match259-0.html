    public static Node organizeBooks(Node head, Integer[] popularity) {
        
        //TO DO
        int n = popularity.length;

        for (int i=0;i<n-1; i++) {
            Node current = head;
            for (int j=0; j<n-i-1;j++) {
                if (popularity[j]<popularity[j + 1]) {
                    int tempPopularity = popularity[j];
                    popularity[j] = popularity[j + 1];
                    popularity[j + 1] = tempPopularity;
                        
                    Object temp = current.elem;
                    current.elem = current.next.elem;
                    current.next.elem = temp;
                }
                current = current.next;
            }
        }
         return head;
    }

public class AssignmentTask4{
    
    // MUST SUBMIT this method
    public static Node idGenerator(Node head1, Node head2, Node head3) {
        
        //TO DO
        Node newHead = null;
        Node temp1=head1;

        while(temp1!=null){
            newHead=new Node((int)temp1.elem,newHead);


            temp1=temp1.next;
        }
        
        Node temp2=head2;
        Node temp3=head3;

        while(temp2!=null && temp3!=null){
            int sum=(int)temp2.elem + (int)temp3.elem;
            if(sum>=10){
                sum=sum%10;
            }

            Node temp=newHead;
            while(temp.next!=null){
                temp=temp.next;
            }
            temp.next=new Node(sum,null);
            temp2=temp2.next;
            temp3=temp3.next;
        }
        return newHead;
    }




public class AssignmentTask5 {

    public static void sumOddAppend(Node dh) {
        // TO DO
        Node prev=dh;
        Node curr=dh.next;
        int sum=0;

        while(curr!=dh){
            int num= (int)curr.elem;

            if(num%2!=0){
                sum+=num;


                prev.next=curr.next;
            }
            else prev=curr;

            curr=curr.next;
        }

        Node sumOdd = new Node(sum);
        sumOdd.next=dh;
        prev.next=sumOdd;

    }





public class AssignmentTask7 {

    public static void rangeMove(DNode dh, int start, int end) {
        if (dh == null || dh.next == dh) {
            return;
        }

        DNode curr = dh.next;
        int size = 0;
    
        while (curr!=dh) {
            size++;
            curr = curr.next;
        }

        curr = dh.next;
        for (int i = 0; i < size; i++) {
            DNode nextNode = curr.next;
            int val = (int)curr.element;

            if (val >= start && val <= end) {


                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;

                curr.next = dh;
                curr.prev = dh.prev;
            
                dh.prev.next = curr;
                dh.prev = curr;
            }

            curr = nextNode;
       }
    }
