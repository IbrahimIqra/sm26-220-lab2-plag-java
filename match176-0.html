    public static String checkSimilar( Node building1, Node building2 ){
        if(building1==null || building2==null) return null;
        Node temp1 = building1;
        Node temp2 = building2;

        while(temp1!=null && temp2!=null){
            if(!temp1.elem.equals(temp2.elem)){
                return "Not Similar";
            }
            temp1 = temp1.next;
            temp2 = temp2.next;
        }



        if(temp1!=null || temp2!=null){
            return "Not Similar";
        }
        return "Similar";
    }

    public static Node organizeBooks(Node head, Integer[] popularity) {
        if(head==null || head.next==null) return head;
        int length = popularity.length;

        for(int i=0; i<length-1; i++){
            Node temp = head;
            int idx = 0;
            while(temp.next!=null){


                if(popularity[idx]<popularity[idx+1]){
                    int tempScore = popularity[idx];
                    popularity[idx] = popularity[idx+1];
                    popularity[idx+1] = tempScore;

                    Object tempElem = temp.elem;
                    temp.elem = temp.next.elem;
                    temp.next.elem = tempElem;
                }
                temp = temp.next;
                idx++;
            }
        }


        return head;
    }

    public static Node alternateMerge( Node head1, Node head2 ){
        Node temp1 = head1;
        Node temp2 = head2;

        while(temp1!=null && temp2!=null){
            Node next1 = temp1.next;
            Node next2 = temp2.next;
            temp1.next = temp2;
            if(temp1.next!=null){


                temp1.next.next = next1;
            }
            temp1 = next1;
            temp2 = next2;
        }
        return head1;
    }

        Node temp = newHead;

            Node newNode = new Node(cur.elem);
            newNode.next = newHead;
            newHead = newNode;
            cur = cur.next;

            int val = (int)temp2.elem + (int)temp3.elem;

        }

    public static void sumOddAppend(Node dh) {
        Node prev = dh;
        Node cur = dh.next;
        int sum = 0;

        while(cur!=dh){
            if((int)cur.elem%2==0){
                prev.next = cur;
                prev = cur;
            }
            else{
                sum += (int)cur.elem;
                prev.next = cur.next;
            }


            cur = cur.next;
        }
        Node newNode = new Node(sum);
        prev.next = newNode;
        newNode.next = dh;
    }

    public static void pairJoin(DNode dh1, DNode dh2) {
        DNode cur1 = dh1.next;
        DNode cur2 = dh2.next;
        DNode lastNode = null;

        while(cur1!=null && cur2!=null){
            DNode next1 = cur1.next;


            DNode next2 = cur2.next;
            cur1.next = cur2;
            cur2.prev = cur1;
            cur2.next = next1;
            if(next1!=null) next1.prev = cur2;
            lastNode = cur2;
            cur1 = next1;


            cur2 = next2;
        }

        lastNode.next = dh1;
        dh1.prev = lastNode;
    }

    public static void rangeMove(DNode dh, int start, int end) {
        DNode cur = dh.next;
        DNode tail = dh.prev;
        DNode tempTail = tail;

        while(cur!=dh){
            DNode nextNode = cur.next;



            if((int)cur.elem>=start && (int)cur.elem<=end){
                cur.next.prev = cur.prev;
                cur.prev.next = cur.next;
                tempTail.next = cur;
                cur.prev = tempTail;
                cur.next = dh;
                dh.prev = cur;
                tempTail = cur;
            }

            if(cur==tail) break;
            cur = nextNode;
        }
    }
