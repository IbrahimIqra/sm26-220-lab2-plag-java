public static String checkBuilding(Node head1, Node head2) {

    Node temp1 = head1;
    Node temp2 = head2;

    while (temp1 != null && temp2 != null) {

        if (temp1.element.equals(temp2.element)) {
            temp1 = temp1.next;
            temp2 = temp2.next;
        } 
        else {
            return "Not Similar";
        }

    }

    if (temp1 == null && temp2 == null) {
        return "Similar";
    }

    return "Not Similar";
}

public static Node organizeBooks(Node head, int[] popularity) {



    int n = popularity.length;

    for (int i = 0; i < n - 1; i++) {

        Node temp = head;

        for (int j = 0; j < n - 1 - i; j++) {

            if (popularity[j] < popularity[j + 1]) {

                int t = popularity[j];
                popularity[j] = popularity[j + 1];
                popularity[j + 1] = t;

                String s = (String) temp.element;
                temp.element = temp.next.element;
                temp.next.element = s;
            }

            temp = temp.next;
        }
    }

    return head;
}

public static Node organizeBooks(Node head, Integer[] popularity) {

    Node newHead = null;
    Node newTail = null;

    while (head != null) {

        Node maxNode = head;
        Node maxPrev = null;

        Node prev = head;
        Node temp = head.next;

        int maxIndex = 0;
        int index = 1;

        while (temp != null) {

            if (popularity[index] > popularity[maxIndex]) {
                maxNode = temp;
                maxPrev = prev;
                maxIndex = index;
            }

            prev = temp;
            temp = temp.next;
            index++;
        }

        if (maxPrev == null) {
            head = head.next;
        } else {
            maxPrev.next = maxNode.next;
        }

        maxNode.next = null;

        if (newHead == null) {
            newHead = maxNode;
            newTail = maxNode;
        } else {
            newTail.next = maxNode;
            newTail = maxNode;
        }

        popularity[maxIndex] = Integer.MIN_VALUE;
    }

    return newHead;
}

public static Node alternateMerge(Node head1, Node head2) {

    Node temp1 = head1;
    Node temp2 = head2;

    while (temp1 != null && temp2 != null) {

        Node next1 = temp1.next;


        Node next2 = temp2.next;

        temp1.next = temp2;

        if (next1 == null) {
            break;
        }

        temp2.next = next1;

        temp1 = next1;
        temp2 = next2;
    }

    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {

    Node result = null;
    Node tail = null;

    int[] first = new int[4];
    int i = 0;

    Node temp = head1;
    while (temp != null) {
        first[i] = (int) temp.elem;
        temp = temp.next;
        i++;
    }

    for (i = 3; i >= 0; i--) {
        Node newNode = new Node(first[i], null);

        if (result == null) {
            result = newNode;
            tail = newNode;
        } else {
            tail.next = newNode;


            tail = newNode;
        }
    }

    Node t2 = head2;
    Node t3 = head3;

    while (t2 != null && t3 != null) {

        int sum = (int) t2.elem + (int) t3.elem;

        if (sum >= 10) {
            sum = sum % 10;
        }

        Node newNode = new Node(sum, null);
        tail.next = newNode;
        tail = newNode;



        t2 = t2.next;
        t3 = t3.next;
    }

    return result;
}

public static void sumOddAppend(Node dh) {

    int sum = 0;

    Node prev = dh;
    Node temp = dh.next;

    while (temp != dh) {

        if ((int) temp.element % 2 != 0) {
            sum += (int) temp.element;
            prev.next = temp.next;
            temp = temp.next;
        }
        else {
            prev = temp;
            temp = temp.next;
        }
    }

    Node newNode = new Node(sum);

    temp = dh;
    while (temp.next != dh) {


        temp = temp.next;
    }

    temp.next = newNode;
    newNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {

    DNode temp1 = dh1.next;
    DNode temp2 = dh2.next;

    while (temp1 != null && temp2 != null) {

        DNode next1 = temp1.next;


        DNode next2 = temp2.next;

        temp1.next = temp2;
        temp2.prev = temp1;

        temp2.next = next1;

        if (next1 != null) {
            next1.prev = temp2;
        }

        temp1 = next1;
        temp2 = next2;
    }

    temp1 = dh1;
    while (temp1.next != null) {
        temp1 = temp1.next;
    }



    temp1.next = dh1;
    dh1.prev = temp1;
}

public static void rangeMove(DNode dh, int start, int end) {

    DNode last = dh.prev;
    DNode temp = dh.next;

    while (temp != dh) {

        DNode next = temp.next;

        if ((int) temp.element >= start && (int) temp.element <= end) {

            if (temp == last) {
                temp = next;
                continue;
            }

            temp.prev.next = temp.next;
            temp.next.prev = temp.prev;

            last.next = temp;
            temp.prev = last;
            temp.next = dh;
            dh.prev = temp;

            last = temp;
        }

        temp = next;
    }
}
