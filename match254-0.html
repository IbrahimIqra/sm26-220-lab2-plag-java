        this.color = color;
        this.next = null;

public class BuildingSimilarity {




    public static String checkSimilarity(Node head1, Node head2) {


        Node temp1 = head1;
        Node temp2 = head2;


        while (temp1 != null && temp2 != null) {


            if (!temp1.color.equals(temp2.color)) {
                return "Not Similar";
            }


            temp1 = temp1.next;
            temp2 = temp2.next;
        }


        if (temp1 == null && temp2 == null) {
            return "Similar";
        }


        return "Not Similar";
    }
}

        this.book = book;

class LibraryOrganizer {


    public Node organizeBooks(Node head, int[] popularity) {


        if (head == null || head.next == null) {
            return head;
        }


        for (int i = 0; i < popularity.length - 1; i++) {


            Node current = head;




            for (int j = 0; j < popularity.length - i - 1; j++) {


                if (popularity[j] < popularity[j + 1]) {


                    int tempScore = popularity[j];
                    popularity[j] = popularity[j + 1];
                    popularity[j + 1] = tempScore;




                    String tempBook = current.book;
                    current.book = current.next.book;
                    current.next.book = tempBook;
                }


                current = current.next;
            }
        }


        return head;
    }
}

class AlternateMerge {


    public Node alternate_merge(Node head1, Node head2) {


        Node p1 = head1;
        Node p2 = head2;




        while (p1 != null && p2 != null) {


            Node next1 = p1.next;
            Node next2 = p2.next;


            p1.next = p2;


            if (next1 == null) {
                break;
            }


            p2.next = next1;


            p1 = next1;
            p2 = next2;
        }


        return head1;
    }
}

class IDGenerator {


    public Node idGenerator(Node head1, Node head2, Node head3) {




        Node result = null;
        Node temp = head1;


        while (temp != null) {
            Node newNode = new Node(temp.data);
            newNode.next = result;
            result = newNode;
            temp = temp.next;
        }


        Node tail = result;
        while (tail.next != null) {
            tail = tail.next;
        }
        Node p1 = head2;
        Node p2 = head3;


        while (p1 != null && p2 != null) {


            int digit = (p1.data + p2.data) % 10;




            tail.next = new Node(digit);
            tail = tail.next;


            p1 = p1.next;
            p2 = p2.next;
        }


        return result;
    }
}

    public static void sumOddAppend(Node dh) {
         int sum = 0;


        Node prev = dh;
        Node curr = dh.next;


        while (curr != dh) {


            if (curr.elem % 2 != 0) {
                sum += curr.elem;


                prev.next = curr.next;
                curr = prev.next;
            } else {
                prev = curr;
                curr = curr.next;
            }
        }


        Node newNode = new Node(sum, dh);




        Node tail = dh;
        while (tail.next != dh) {
            tail = tail.next;
        }
        tail.next = newNode;
    }

            DNode next1 = p1.next;
            DNode next2 = p2.next;


            p1.next = p2;

            p2.next = next1;


            if (next1 != null) {
                next1.prev = p2;
            }


            p1 = next1;
            p2 = next2;
