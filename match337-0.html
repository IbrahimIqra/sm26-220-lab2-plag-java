// AssignmentTask5: Sum odd append 
public static void sumOddAppend(Node dh) {
        // TO DO
        int sum=0;
        Node prev =dh;
        Node curr= dh.next;
        while(curr!=dh){
            if((int)curr.elem %2==0){
                prev=curr;
                curr=curr.next;
            }
            else{
                sum+= (int)curr.elem;
                prev.next=curr.next;


                curr=curr.next;
            }
        }Node nn= new Node(sum);
        prev.next=nn;
        nn.next=dh;


    }

public static void pairJoin(DNode dh1, DNode dh2) {

        DNode curr2=dh2.next;
        DNode tail=null;
        while(curr1!=null && curr2!=null){
            DNode next1=curr1.next;
            DNode next2=curr2.next;
            curr1.next=curr2;


            curr2.prev=curr1;
            curr2.next=next1;
            if(next1!=null){
                next1.prev=curr2;
             }
            tail=curr2;
            curr1=next1;
            curr2=next2;


        }
        if (tail!=null){
            tail.next=dh1;
            dh1.prev=tail;
        }




    }
