public static String checkSimilar(Node building1, Node building2){
        
        Node temp1 = building1;
        Node temp2 = building2;

        while(temp1 != null && temp2 != null){
          if(temp1.elem == null){
            if(temp2.elem != null){ 
              return "Not Similar";
            }
          }


          else if(!temp1.elem.equals(temp2.elem)){
                return "Not Similar";
            }

            temp1 = temp1.next;
            temp2 = temp2.next;
        }

        if(temp1 == null && temp2 == null){
          return "Similar";
        } 
        else{
          return "Not Similar";
        }
    }

public static Node organizeBooks(Node head, Integer[] popularity){
      if(head == null || head.next == null){
        return head;
      }

      boolean swapped;
      do{
        swapped = false;
        Node current = head;
        int i = 0;

        while(current != null && current.next != null){
          if(popularity[i] < popularity[i + 1]){
            int tempPop = popularity[i];


            popularity[i] = popularity[i + 1];
            popularity[i + 1] = tempPop;

            Object tempElem = current.elem;
            current.elem = current.next.elem;
            current.next.elem = tempElem;

            swapped = true;
          }
          current = current.next;
          i++;
        }
      }
      while(swapped);

      return head;
    }

public static Node alternateMerge(Node head1, Node head2){
    
    if(head1 == null) 
      return head2;
    if(head2 == null) 
      return head1;
    
    Node curr1 = head1;
    Node curr2 = head2;
    
    while(curr1 != null && curr2 != null){
      Node next1 = curr1.next;
      Node next2 = curr2.next;
      
      curr1.next = curr2;
      
      if(next1 == null){
        break;
      }
      
      curr2.next = next1;
      
      curr1 = next1;
      curr2 = next2;
    }
    
    return head1;
  }

 public static Node idGenerator(Node head1, Node head2, Node head3){
        
      Node prev = null;
      Node curr = head1;
      while(curr != null){
        Node next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
      }
      Node newHead = prev;

      Node tail = newHead;
      while(tail.next != null){


        tail = tail.next;
      }

      Node t2 = head2;
      Node t3 = head3;
      while(t2 != null && t3 != null){
        int val2 = (int)t2.elem;
        int val3 = (int)t3.elem;
        int sum = val2 + val3;
        if(sum >= 10){
          sum = sum % 10;
        }

        Node newNode = new Node(sum);
        tail.next = newNode;
        tail = newNode;

        t2 = t2.next;
        t3 = t3.next;
      }

      return newHead;
    }

public static void sumOddAppend(Node dh){
      if(dh == null) 
        return;

      Node prev = dh;
      Node curr = dh.next;
      int sum = 0;

      while(curr != dh){


        int val = (Int)curr.elem;
        if(val % 2 != 0){
          sum += val;
          prev.next = curr.next;
        } 
        else{
          prev = curr;
        }
        curr = curr.next;
      }

      Node sumNode = new Node(sum, dh);
      prev.next = sumNode;
    }

public static void pairJoin(DNode dh1, DNode dh2){


      if(dh1 == null || dh2 == null || dh1.next == dh1 || dh2.next == dh2){
        return;
      }

      DNode curr1 = dh1.next;
      DNode curr2 = dh2.next;
      DNode tail = dh1;

      while(curr1 != dh1 && curr2 != dh2){
        DNode next1 = curr1.next;
        DNode next2 = curr2.next;

        tail.next = curr1;
        curr1.prev = tail;


        tail = curr1;

        tail.next = curr2;
        curr2.prev = tail;
        tail = curr2;

        curr1 = next1;
        curr2 = next2;
      }

      while(curr1 != dh1){
        DNode next1 = curr1.next;
        tail.next = curr1;
        curr1.prev = tail;
        tail = curr1;
        curr1 = next1;
      }

      while(curr2 != dh2){
        DNode next2 = curr2.next;
        tail.next = curr2;
        curr2.prev = tail;
        tail = curr2;
        curr2 = next2;
      }

      tail.next = dh1;
      dh1.prev = tail;
    }

public static void rangeMove(DNode dh, int start, int end){
      if(dh == null || dh.next == dh){
        return;
      }

      DNode first = null;
      DNode last = null;
      DNode curr = dh.next;

      while(curr != dh){
        int val = (Int)curr.elem;
        if(val >= start && val <= end){
          if(first == null){
            first = curr;
          }
          last = curr;
        }
        curr = curr.next;
      }

      if(first == null){
        return;
      }



      first.prev.next = last.next;
      last.next.prev = first.prev;

      DNode originalTail = dh.prev;

      originalTail.next = first;
      first.prev = originalTail;
      last.next = dh;
      dh.prev = last;
    }
