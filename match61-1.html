public static Node organizeBooks(Node head, Integer[] popularity) {
        int n = popularity.length;

        for(int i=0; i<n-1; i++){
            Node current = head;
            Node nextNode = current.next;
            for(int j=0; j<n-i-1; j++){
                if(popularity[j] < popularity[j+1]){
                    int currentPopularity = popularity[j];
                    popularity[j] = popularity[j+1];
                    popularity[j+1] = currentPopularity;

                    Object temp = current.elem;
                    current.elem = nextNode.elem;
                    nextNode.elem = temp;
                }
                current = current.next;
                nextNode = current.next;
            }
        }
        return head;
    }

public static void sumOddAppend(Node dh){
        int oddSum = 0;
        Node prev = dh;
        Node curr = dh.next;

        while (curr != dh) {
            int value = (int) curr.elem;

            if (value % 2 != 0) {
                oddSum += value;


                prev.next = curr.next;
            } else {
                prev = curr;
            }
            curr = curr.next;
        }

        Node sumNode = new Node(oddSum);
        sumNode.next = dh;
        prev.next = sumNode;
    }

public static void pairJoin(DNode dh1, DNode dh2){
        DNode c1 = dh1.next;
        DNode c2 = dh2.next;

        while (c1 != dh1){
            DNode n1 = c1.next;
            DNode n2 = c2.next;

            c1.next = c2;


            c2.prev = c1;

            c2.next = n1;
            n1.prev = c2;
            
            c1 = n1;
            c2 = n2;
        }
    }

public static void rangeMove(DNode dh, int start, int end){
        DNode lastNode = dh.prev;
        DNode oriEnd = lastNode;
        DNode curr = dh.next;

        while (curr != oriEnd.next) {
            DNode nextNode = curr.next;
            int value = (int) curr.elem;

            if (value >= start && value <= end) {
                if (curr == oriEnd) {
                    oriEnd = origEnd.prev;
                }

                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;

                lastNode.next = curr;
                curr.prev = lastNode;
                curr.next = dh;
                dh.prev = curr;

                lastNode = curr;
            }

            curr = nextNode;
        }
    }
