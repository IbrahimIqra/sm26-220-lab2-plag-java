 public static String checkSimilar( Node building1, Node building2 ){
       Node cur1 = building1;
       Node cur2 = building2;
       while(cur1!=null && cur2!=null){
        if(!cur1.elem.equals(cur2.elem)){
        return "Not Similar";
       }
    cur1 = cur1.next;
    cur2 = cur2.next;
    
    }
    if(cur1==null && cur2==null){
    return "Similar";
    }    
    else{}
    return "Not Similar";
    }





// AssignmentTask2: Organize Books

public static Node organizeBooks(Node head, Integer[] popularity) {
       if(head==null || head.next==null){
        return head;
       } 
       int l = popularity.length;
       for(int i=0; i<l-1; i+=1){
        Node cur = head;
        for(int j=0; j<l-1-i; j+=1){


         if(popularity[j]<popularity[j+1]){
          int temP = popularity[j];
          popularity[j] = popularity[j+1];
          popularity[j+1] = temP;

          Object temB = cur.elem;
          cur.elem = cur.next.elem;
          cur.next.elem = temB;
         }
         cur = cur.next;
        }
       }
        return head; 
    }





// AssignmentTask3: Alternate Merge

public static Node alternateMerge( Node head1, Node head2 ){
     if(head1==null){
       return head2;   
      }
     else if(head2==null){
       return head1;
     } 
     else{
      Node cur1 = head1;
      Node cur2 = head2;
      while(cur1!=null && cur2!=null){
       Node nex1 = cur1.next;
       Node nex2 = cur2.next;

       cur1.next = cur2;


       if(cur1.next==null){
        break;
       }
       cur2.next = nex1;
       cur1 = nex1;
       cur2 = nex2;
      }
     }
   return head1;
    }

public static void sumOddAppend(Node dh) {
        int os = 0;
        Node prev = dh;
        Node cur = dh.next;
      while(cur!=dh){
       int val = (Integer)cur.elem;
       if(val%2!=0){
       os+=val;
       prev.next = cur.next;


       cur = cur.next;
      }
     else{
      prev = cur;
      cur = cur.next;
     }
   }
     Node sn = new Node(os);
     prev.next = sn;
     sn.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
       DNode cur1 = dh1.next;
       DNode cur2 = dh2.next;
       DNode t = dh1;
       while(cur1!=null && cur2!=null){
       DNode n1 = cur1.next;
       DNode n2 = cur2.next;
        t.next = cur1;
        cur1.prev = t;
        t = cur1;
        t.next = cur2;
        cur2.prev = t;
        t = cur2;
        cur1 = n1;


        cur2 = n2;
       }
       t.next = dh1;
       dh1.prev = t;
    }

 public static void rangeMove(DNode dh, int start, int end) {
      DNode mh = null;
      DNode mt = null;
      DNode cur = dh.next;
      while(cur!=dh){
       DNode n = cur.next;
       int val = (Integer)cur.elem;

        cur.prev.next = cur.next;
        cur.next.prev = cur.prev;
       if(mh==null){
        mh = cur;
        mt = cur;
       }
       else{
        mt.next = cur;
        cur.prev = mt;
        mt = cur;
       }   
    mt.next = null;


    }
    cur = n;      
    }
    if(mh!=null){
     DNode t = dh.prev;
     t.next = mh;
     mh.prev = t;
     mt.next = dh;
     dh.prev = mt;
    }
