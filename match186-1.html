            public static Node organizeBooks(Node head, Integer[] popularity) {
                if (head == null) return null;
                int n = popularity.length;
                for (int i = 0; i < n - 1; i++){
                    Node current = head;
                    for (int j = 0; j < n - 1 - i; j++){
                        if (popularity[j] < popularity[j + 1]){
                            Object tempElem = current.elem;
                            current.elem = current.next.elem; current.next.elem = tempElem;
                            int tempPop = popularity[j];
                            popularity[j] = popularity[j + 1]; popularity[j + 1] = tempPop;
                        }
                current = current.next;
                    }
                }
                return head;
            }

            public static Node alternateMerge(Node head1, Node head2) {
                if (head1 == null) return head2;
                if (head2 == null) return head1;
                Node result = head1;


                while (head1 != null && head2 != null){
                    Node next1 = head1.next; Node next2 = head2.next;
                    head1.next = head2;
                    if (next1 == null) break;
                    head2.next = next1; head1 = next1; head2 = next2;
                }
                return result;
            }

            public static void sumOddAppend(Node dh) {
                int sum = 0;
                Node prev = dh; Node curr = dh.next;
                while (curr != dh){
                    Node nextNode = curr.next;


                    int val = (Integer) curr.elem;
                    if (val % 2 != 0) {sum += val; prev.next = curr.next;}
                    else prev = curr;
                    curr = nextNode;
                }


                Node newNode = new Node(sum);
                prev.next = newNode; newNode.next = dh;
            }

            public static void pairJoin(DNode dh1, DNode dh2) {
                DNode p1 = dh1.next; DNode p2 = dh2.next; DNode last = dh1;
                if (p1 == null) {dh1.next = dh1; dh1.prev = dh1; return;}
                while (p1 != null && p2 != null){
                    DNode next1 = p1.next; DNode next2 = p2.next;
                    p1.next = p2; p2.prev = p1; p2.next = next1;


                    if (next1 != null) next1.prev = p2;
                    last = p2;
                    p1 = next1; p2 = next2;
                }
                last.next = dh1; dh1.prev = last;
            }

            public static void rangeMove(DNode dh, int start, int end) {
                if (dh.next == dh) return;
                DNode oldTail = dh.prev; DNode tail = dh.prev; DNode curr = dh.next;
                while (true){


                    DNode nextNode = curr.next;
                    boolean isOldTail = (curr == oldTail);
                    int val = (Integer) curr.elem;
                    if (val >= start && val <= end){
                        if (curr != tail){
                            curr.prev.next = curr.next; curr.next.prev = curr.prev;
                            tail.next = curr; curr.prev = tail;
                            curr.next = dh; dh.prev = curr;
                            tail = curr;
                        }
                    }
                    if (isOldTail) break;
                    curr = nextNode;
                }
            }
