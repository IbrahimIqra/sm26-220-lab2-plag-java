 public static String checkSimilar( Node building1, Node building2 ){
        Node temp1=building1;
        Node temp2=building2;
        while(temp1!=null&&temp2!=null){
            if(!temp1.elem.equals(temp2.elem)){


                return "Not Similar";
            }
            temp1=temp1.next;
            temp2=temp2.next;
        }
        if(temp1==null&&temp2==null){
            return "Similar";
        }else{
            return "Not Similar";
        }
    }

 public static Node organizeBooks(Node head, Integer[] popularity) {


        int n = popularity.length;
        for(int i=0;i<n-1;i++){
            Node current=head;
            for(int j=0;j<n-1-i;j++){

                if(popularity[j]<popularity[j+1]){
                    int temp = popularity[j];
                    popularity[j]=popularity[j+1];
                    popularity[j+1]=temp;
                    Object book=current.elem;
                    current.elem=current.next.elem;
                    current.next.elem=book;
                }
                current=current.next;
            }
            
        }

        return head;
    }

 public static Node alternateMerge( Node head1, Node head2 ){
        Node temp1=head1;
        Node temp2=head2;
        while(temp1!=null&&temp2!=null){
            Node next1=temp1.next;
            Node next2=temp2.next;
            temp1.next=temp2;
            temp2.next=next1;
            temp1=next1;
            temp2=next2;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node resultHead = null;
        Node temp=head1;
        while(temp!=null){
            Node newNode=new Node((Integer)temp.elem);
            newNode.next=resultHead;
            resultHead=newNode;


            temp=temp.next;
        }
        Node tail=resultHead;
        while(tail.next!=null){
            tail=tail.next;
        }
        Node temp2=head2;
        Node temp3=head3;
        while(temp2!=null&&temp3!=null){
            int sum=(Integer)temp2.elem+(Integer)temp3.elem;
            Node newNode=new Node(sum);


            tail.next=newNode;
            tail=newNode;
            temp2=temp2.next;
            temp3=temp3.next;
        }
        return resultHead;
    }

 public static void sumOddAppend(Node dh) {
        Node prev = dh;
        Node curr = dh.next;
        int sum = 0;
        while(curr != dh) {
            if((Integer)curr.elem% 2 != 0) {
                sum += (Integer)curr.elem;


                prev.next = curr.next;
                curr=prev.next;
            }else {
                prev = curr;
                curr = curr.next;
            }
        }
        Node newNode = new Node(sum);
        Node tail=dh;
        while(tail.next != dh) {
            tail = tail.next;
        }
        tail.next = newNode;
        newNode.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode p1=dh1.next;
        DNode p2=dh2.next;
        while(p1!=null && p2!=null){
            DNode temp1=p1.next;
            DNode temp2=p2.next;
            p1.next=p2;
            p2.prev=p1;
            if(temp1!=null){
                p2.next=temp1;
                temp1.prev=p2;
            }else{
                p2.next=dh1;
                dh1.prev=p2;
            }
            p1=temp1;
            p2=temp2;
        }
        DNode tail=dh1.prev;
        tail.next=dh1;
        dh1.prev=tail;
    }
