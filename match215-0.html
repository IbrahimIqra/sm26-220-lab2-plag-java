public static String checkSimilar( Node building1, Node building2 ){
        Node n1=building1;
        Node n2 = building2;
        while(n1!=null && n2!=null){
            if(!n1.elem.equals(n2.elem)){
                return "Not Similar";
            }
            n1=n1.next;
            n2=n2.next;
        }
        if(n1!=null || n2!=null){
            return "Not Similar";
        }
        return "Similar";
    }

public static void sumOddAppend(Node dh) {
       
        if(dh==null||dh.next==dh){
            return;
        }
        Node prev=dh;
        Node temp=dh.next;
        int sum=0;
        while(temp!=dh){
            if((int)temp.elem%2!=0){
                sum+=(int)temp.elem;


                prev.next=temp.next;
                temp=prev.next;
            }else{
                prev=temp;
                temp=temp.next;
            }
        }
        Node newN =new Node(sum);


        newN.next=dh;
        prev.next=newN;
    }

 public static void rangeMove(DNode dh, int start, int end) {
        DNode curr=dh.next;
        DNode tail=dh.prev;
        DNode temp=dh.next;
        int count =0;
        int totalNodes=0;
    while(temp!=dh){
        totalNodes++;
        temp=temp.next;
    }
    while(count<totalNodes){


        DNode nextN = curr.next;
        int num=(int)curr.elem;
        if(num>=start && num<=end){
            curr.prev.next=curr.next;
            curr.next.prev=curr.prev;
            curr.prev=tail;
            curr.next=dh;
            tail.next=curr;
            dh.prev=curr;
            tail=curr;
        }
        curr=nextN;
        count++;

    }
