            Node a = temp1.next;

    public static Node idGenerator(Node head1, Node head2, Node head3) {
        
        Node prev = null;
        Node current = head1;
          while(current != null){


            Node next = current.next;
            current.next = prev;
      
      
            prev = current;
            current = next;
            }



          Node nextHead = prev;
       
       
         Node tail = nextHead;

          while(tail.next != null){
       
       
            tail = tail.next;
             }

          Node x = head2;
       
       
          Node y = head3;

           while(x != null && y != null){


               int total = (int)x.elem + (int)y.elem;
               if(total > 10 || total == 10){




                  total = total%10;
                  }
             Node n = new Node(total);
            tail.next = n;
              tail = n;

              x = x.next;

               y = y.next;
               }
          return nextHead;


        }

    public static void rangeMove(DNode dh, int start, int end) {


          DNode temp = dh.next;


          DNode last = dh.prev;

           while(temp != null){



                DNode next = temp.next;
               
            
             if((int)temp.elem >= start && (int)temp.elem <= end){

                   DNode prevNode = temp.prev;
                    DNode nextNode = temp.next;
                //prevNode.prev = nextNode;

                   prevNode.next = nextNode;
                   nextNode.prev = prevNode;




                DNode lastNode = dh.prev;
 
                  lastNode.next = temp;


                   temp.prev = lastNode;
                 //temp.prev = dh; 

                temp.next = dh;

                  dh.prev = temp;
              }


                 if(temp == last){
                 break;
                 }
                temp = next;
          }
    }
