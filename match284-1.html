public static String checkSimilar(Node building1, Node building2){
        Node x1=building1;
        Node x2=building2;



        while(x2 != null && x1 != null){
            if(!x1.elem.equals(x2.elem)){
                return "Not Similar";
            }
            x1=x1.next;
            x2=x2.next;
        }

        //length check!
        if(x1 != null || x2 != null){


            return "Not Similar";
        }

        return "Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity){
        int n=popularity.length;

        for(int i=0;i<n-1;i++){
            Node dhead=head;
            int j=0;
            while(dhead != null && j<n-1-i &&  dhead.next != null){
                if(popularity[j]<popularity[j+1]){
                    //exchanging book titles
                    Object temporaryelem=dhead.elem;
                    dhead.elem=dhead.next.elem;
                    dhead.next.elem=temporaryelem;

                    //exchanging popularity scores
                    int teampopularity=popularity[j];
                    popularity[j]=popularity[j+1];
                    popularity[j+1]=teampopularity;
                }
                dhead=dhead.next;
                j++;
            }
        }

        return head;
    }

public static void sumOddAppend(Node dh){
        int net=0;
        Node previous=dh;
        Node current=dh.next ;

        while(current != dh){
            int value=(Integer) current.elem;
            if(value%2 != 0){
                net+=value;
                previous.next=current.next;
                current=current.next;
            }
            else{
                previous=current;
                current=current.next;
            }
        }

        Node netnode=new Node(net);


        netnode.next=dh;
        previous.next=netnode ;
    }

public static void pairJoin(DNode dh1, DNode dh2){
        DNode x1=dh1.next;
        DNode  x2=dh2.next;

        while(x1 != null){


            DNode x1next=x1.next;
            DNode x2next=x2.next;

            x1.next=x2;
            x2.prev=x1;

            if(x1next  != null){
                x2.next=x1next;
                x1next.prev=x2;
            }
            else  {
                x2.next=dh1;
                dh1.prev=x2;
            }

            x1=x1next;
            x2=x2next;
        }
    }

public static void rangeMove(DNode dh, int start, int end){
        DNode current=dh.next;
        DNode shiftedtail=null;
        DNode  shiftedhead=null;

        while(current  != dh){


            DNode next=current.next;
            int value=(Integer) current.elem;

            if(value>=start && value<=end){
                current.prev.next=current.next;
                current.next.prev=current.prev;

                if( shiftedhead == null){
                    shiftedhead=current;
                    shiftedtail=current;
                }
                else{
                    shiftedtail.next=current;
                    current.prev=shiftedtail;
                    shiftedtail=current ;
                }
            }

            current=next;
        }

        if (shiftedhead != null){
            DNode tail=dh.prev;
            tail.next=shiftedhead;
            shiftedhead.prev=tail ;
            shiftedtail.next=dh;
            dh.prev=shiftedtail;
        }
    }
