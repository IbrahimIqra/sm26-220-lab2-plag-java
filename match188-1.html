public static String checkSimilar( Node building1, Node building2 ){
        
        while(building1 !=null && building2 !=null){
            if(building1.elem.equals(building2.elem)){
                building1=building1.next;
                building2=building2.next;
            }
            else{
                return "Not Similar";
            }
        }
        if(building1==null && building2==null){
            return "Similar";
        }
        return "Not Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {


        for(int i=0;i<popularity.length-1;i++){
            Node cur=head;
            for(int j=0;j<popularity.length-1-i;j++){
              int tempVal=0;
              if(popularity[j]<popularity[j+1]){
                tempVal=popularity[j];


                popularity[j]=popularity[j+1];
                popularity[j+1]=tempVal;
                Object book=cur.elem;
                cur.elem=cur.next.elem;
                cur.next.elem=book;
              }
              cur=cur.next;
            }
        }
        return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){
      
        Node pt1=head1;
        Node pt2=head2;
        while(pt1!=null && pt2!=null){
            Node pt3=pt1.next;
            Node pt4=pt2.next;
            pt1.next=pt2;
            pt2.next=pt3;
            pt1=pt3;
            pt2=pt4;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        
        Node prv=null;
        Node cur=head1;
        while(cur!=null){
            Node nt=cur.next;  
            cur.next=prv;  
            prv=cur;  


            cur=nt;
        }
        Node t=prv;
        while(t.next!=null){
            t=t.next;
        }
        while(head2!=null && head3!=null){
            int add=(int)head2.elem + (int)head3.elem;
            int res=0;
            if(add>=10){
                res=add%10;
            }
            else{
                res+=add;
            }


            t.next=new Node(res);
            t=t.next;
            head2=head2.next;
            head3=head3.next;
        }
        return prv;
    }

public static void sumOddAppend(Node dh) {
        Node prv=dh;
        Node cur=dh.next;
        int summation=0;
        while(cur!=dh){
            if((int)cur.elem%2==0){
                prv=cur;
                cur=cur.next;
            }
            else{
                summation+=(int)cur.elem;
                prv.next=cur.next;
                cur=cur.next;
            }
        }
        Node summ=new Node(summation);
        prv.next=summ;


        summ.next=dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode curOne=dh1.next;
        DNode curTwo=dh2.next;
        while(curOne!=null){


            DNode n1=curOne.next;
            DNode n2=curTwo.next;
            curOne.next=curTwo;
            curTwo.prev=curOne;
            if(n1==null){
                curTwo.next=dh1;
                dh1.prev=curTwo;
            }
            else{


                curTwo.next=n1;
                n1.prev=curTwo;
            }
            curOne=n1;
            curTwo=n2;
        }
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode cur=dh.next;
        DNode last=dh.prev;
        while(cur!=dh){
            DNode n=cur.next;
            if((int)cur.elem>=start && (int)cur.elem<=end){
                cur.prev.next=cur.next;
                cur.next.prev=cur.prev;
                DNode t=dh.prev;
                t.next=cur;
                cur.prev=t;
                cur.next=dh;
                dh.prev=cur;
            }
            if(cur==last){
                break;
            }
            cur=n;
        }
    }
