 public static String checkSimilar(Node building1, Node building2) {
        Node n1 = building1;
        Node n2 = building2;
        
        while (n1 != null && n2 != null) {
            if (!n1.elem.equals(n2.elem)) {
                return "Not Similar";
            }
            n1 = n1.next;
            
            n2 = n2.next;
        }
        
        if (n1 == null && n2 == null) {
            return "Similar";
        }
        
        return "Not Similar";
    }

public static Node alternateMerge(Node head1, Node head2) {
        if (head1 == null) return head2;
        if (head2 == null) return head1;
        
        Node n1 = head1;
        Node n2 = head2;
        
        while (n1 != null && n2 != null) {
            Node next1 = n1.next;
            Node next2 = n2.next;
            
            n1.next = n2;
            if (next1 != null) {
                n2.next = next1;
            }
            
            n1 = next1;
            n2 = next2;
        }
        
        return head1;
    }
