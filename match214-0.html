 public static String checkSimilar( Node building1, Node building2 ){
      Node p=building1;
      Node q=building2;
      while(p!=null && q!=null){
        if(p.elem!=q.elem){
            return "Not Similar";
        }
        p=p.next;
        q=q.next;


      }
      if(p!=null||q!=null){
        return "Not Similar";
      }

    }

public static Node organizeBooks(Node head, Integer[] popularity) {
       for(int i=0;i<popularity.length-1;i++){
        Node temp=head;
        Node newNode =temp.next;


        for(int j=0;j<popularity.length-1-i;j++){
            if(popularity[j]<popularity[j+1]){
                int nowp=popularity[j];
                popularity[j]=popularity[j+1];
                popularity[j+1]=nowp;
                Object p=temp.elem;
                temp.elem=newNode.elem;
                newNode.elem=p;
            }
            temp=temp.next;
            newNode=temp.next;


            }
        }


        return head;
    }

                last=tnode;
            }
            p=p.next;
        }

    }

public static void sumOddAppend(Node dh) {
        Node prev= dh;
        Node current =dh.next;
        int total=0;
        while(current!=dh){


            if((Integer)current.elem%2==1){
                 total += (Integer)current.elem;
                 prev.next=current.next;
                 current=prev.next;


            }
            else{
                prev=current;
                current=current.next;
            }


        }
        Node temp = new Node(total);
        temp.next=dh;
        prev.next=temp;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
     DNode p=dh1.next;
     DNode q=dh2.next;
     while(p!=null && q!=null){
        DNode n1=p.next;
        DNode n2=q.next;
        p.next=q;
        q.prev=p;
        q.next=n1;
        if(n1!=null){
            n1.prev=q;
        }
        p=n1;
        q=n2;
     }
     DNode last=dh1;
     while(last.next!=null){
        last=last.next;
     }
     last.next=dh1;
     dh1.prev=last;
    }

 public static void rangeMove(DNode dh, int start, int end) {
        int count=0;
        DNode temp=dh.next;
        while(temp!=dh){
            count++;
            temp=temp.next;
        }
        DNode c=dh.next;
        for(int i=0;i<count;i++){
            DNode newt=c.next;
            int n=(Integer)c.elem;


            if(n>=start && n<=end){
                c.prev.next=c.next;
                c.next.prev=c.prev;
                DNode tail= dh.prev;
                tail.next=c;
                c.prev=tail;
                c.next=dh;
                dh.prev=c;
            }
            c=newt;
        }
    }
