public static String checkSimilar( Node building1, Node building2 ){
    
    Node temp1 = building1;
    Node temp2 = building2;
    
    boolean flag = false;
    
    while(temp1!=null || temp2!=null){
      if(temp1 == null || temp2 == null || temp1.elem!=temp2.elem){
        return "Not Similar";
      }
      else{
        temp1 = temp1.next;
        temp2 = temp2.next;
      }
    }
    return "Similar";
  }

public static Node organizeBooks(Node head, Integer[] popularity) {
    int n = popularity.length;
    for(int i = 0; i<n-1; i++){
      Node cur = head;
      
      for(int j = 0; j<n-i-1; j++){
        if(popularity[j]<popularity[j+1]){
          
          Integer temp = popularity[j];


          popularity[j] = popularity[j+1];
          popularity[j+1] = temp;
          
          String swap = (String )cur.elem;
          cur.elem = cur.next.elem;
          cur.next.elem = swap;
        }
        cur = cur.next;


      }
    }
    return head;
  }

public static Node alternateMerge( Node head1, Node head2 ){
      Node cur1 = head1;
      Node cur2 = head2;
      
      while(cur1!=null && cur2!=null){
        Node temp1 = cur1.next;
        Node temp2 = cur2.next;
        
        cur1.next = cur2;
        if(temp1 == null){
          break;
        }
        
        cur2.next = temp1;
        cur1 = temp1;
        cur2 = temp2;
      }
      return head1;       
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node id = null;
        Node cur1 = head1;
        while (cur1 != null) {
            Node temp = new Node(cur1.elem);
            temp.next = id;
            id = temp;
            cur1 = cur1.next;
        }

        Node Temp = id;
        while (Temp.next != null) {


            Temp = Temp.next;
        }

        Node cur2 = head2;
        Node cur3 = head3;
        while (cur2 != null && cur3 != null) {
            int sum = (int)cur2.elem + (int)cur3.elem;
            if (sum >= 10) {
                sum %= 10;
            }
            Node s = new Node(sum);


            Temp.next = s;
            Temp = s;
            cur2 = cur2.next;
            cur3 = cur3.next;
        }
        return id;
    }

public static void sumOddAppend(Node dh) {
      Node temp = dh;
      int sum = 0;
      
      while(temp.next!=dh){
        
        if((int)temp.next.elem%2!=0){
          sum+=(int)temp.next.elem;
          temp.next = temp.next.next;
        }


        temp = temp.next;
      }
      Node s = new Node(sum);
      temp.next = s;
      s.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
      DNode cur1 = dh1.next;
      DNode cur2 = dh2.next;
      
      while(cur1!=null){
        DNode temp1 = cur1.next;


        DNode temp2 = cur2.next;
        
        cur1.next = cur2;
        cur2.prev = cur1;
        
        cur2.next = temp1;
        if(temp1 != null){
          temp1.prev = cur2;
        }
        cur1 = temp1;
        cur2 = temp2;
      }
    }

public static void rangeMove(DNode dh, int start, int end) {
      int len = 0;
      DNode temp = dh.next;
      while(temp!=dh){
        len++;
        temp = temp.next;
      }
      
      DNode cur = dh.next;
      for(int i = 0; i<len; i++){
        DNode Temp = cur.next;
        
        int val = (int)cur.elem;
        if(val>=start && val<=end){
          cur.prev.next = cur.next;
          cur.next.prev = cur.prev;
          
          DNode tail = dh.prev;
          
          tail.next = cur;
          cur.prev = tail;
          
          cur.next = dh;
          dh.prev = cur;
        }
        cur = Temp;
      }
    }
