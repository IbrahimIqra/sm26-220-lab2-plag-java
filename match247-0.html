public static String checkSimilar( Node building1, Node building2 ){
        Node current1 = building1;
        Node current2 = building2;

        while (current1 != null && current2 != null) {
            if (!current1.elem.equals(current2.elem)) {
                return "Not Similar";
            }
            current1 = current1.next;
            current2 = current2.next;
        }

        if (current1 != null || current2 != null) {
            return "Not Similar";
        }

        return "Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {
        
        if (head == null || head.next == null) {
            return head;
        }

        Node current = head;
        for (int i = 0; current != null; i++) {
            Node maxNode = current;
            Node nextNode = current.next;
            int maxIndex = i;

            Node temp = current;
            int j = i + 1;

            while (nextNode != null) {
                if (popularity[j] > popularity[maxIndex]) {
                    maxNode = nextNode;
                    maxIndex = j;
                }
                nextNode = nextNode.next;
                j++;
            }
            
            if (maxNode != current) {
                Object tempBook = current.elem;
                current.elem = maxNode.elem;
                maxNode.elem = tempBook;

                int tempScore = popularity[i];
                popularity[i] = popularity[maxIndex];
                popularity[maxIndex] = tempScore;
            }

            current = current.next;
        }

        return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){
        
        if (head1 == null) return head2;
        if (head2 == null) return head1;

        Node current1 = head1;
        Node current2 = head2;
        while (current1 != null && current2 != null) {

            Node next1 = current1.next;


            Node next2 = current2.next;

            current1.next = current2;

            if (next1 == null) {
                break;
            }

            current2.next = next1;
            current1 = next1;
            current2 = next2;
        }

        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        
        Node prev = null;
        Node current = head1;
        while (current != null) {
            Node next = current.next;
            current.next = prev;
            prev = current;


            current = next;
        }

        Node newHead = prev;
        Node tail = newHead;
        while (tail.next != null) {
            tail = tail.next;
        }

        Node temp2 = head2;
        Node temp3 = head3;

        while (temp2 != null && temp3 != null) {
            int sum = (Integer) temp2.elem + (Integer) temp3.elem;

            if (sum >= 10) {
                sum = sum % 10;
            }

            Node newNode = new Node(sum, null);
            tail.next = newNode;
            tail = tail.next;

            temp2 = temp2.next;
            temp3 = temp3.next;
        }

        return newHead;
    }

public static void sumOddAppend(Node dh) {
        if (dh == null || dh.next == dh) return;

    Node prev = dh;
    Node current = dh.next;
    int sum = 0;

    while (current != dh) {
        int value = (Integer) current.elem;
        if (value % 2 != 0) {
            sum += value;
            prev.next = current.next;
        } else {
            prev = current;
        }
        current = current.next;
    }

    prev = dh;
    while (prev.next != dh) {
        prev = prev.next;
    }
    
    Node newNode = new Node(sum);
    prev.next = newNode;
    newNode.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode p1 = dh1.next;
        DNode p2 = dh2.next;

        while (p1 != null && p2 != null && p1 != dh1 && p2 != dh2) {


            DNode next1 = p1.next;
            DNode next2 = p2.next;

            p1.next = p2;
            p2.prev = p1;

            if (next1 != dh1 && next1 != null) {
                p2.next = next1;
                next1.prev = p2;
            }
            else {
                p2.next = dh1;
                dh1.prev = p2;
            }

            p1 = next1;
            p2 = next2;
        }



        DNode tail = dh1;
        while (tail.next != dh1 && tail.next != null) {
            tail = tail.next;
        }

        tail.next = dh1;
        dh1.prev = tail;
    }

public static void rangeMove(DNode dh, int start, int end) {
        if (dh == null || dh.next == dh) return;

        DNode cur = dh.next;
        DNode stop = dh;
        DNode originalTail = dh.prev;

        while (cur != stop) {
            DNode next = cur.next;
            int val = (int) cur.elem;

            if (val >= start && val <= end) {
                if (cur == originalTail || cur == dh) {
                    cur = next;
                    continue;
                }



                cur.prev.next = cur.next;
                cur.next.prev = cur.prev;

                DNode last = dh.prev;
                last.next = cur;
                cur.prev = last;
                cur.next = dh;
                dh.prev = cur;
            }

            cur = next;
        }
    }
