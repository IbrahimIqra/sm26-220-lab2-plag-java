public static String checkSimilar(Node building1, Node building2) {
    int y=count_length(building1);
    int z=count_length(building2);
    if (y!=z) {
        return "Not Similar";
    }
    for (int i=0; i<y; i++) {
        if (!building1.elem.equals(building2.elem)) {
            return "Not Similar";
        }
        building1 = building1.next;
        building2 = building2.next;
    }
    return "Similar";
}
public static int count_length(Node x) {
    int n = 0;
    Node y = x;
    while (y!= null) {
        n++;
        y = y.next;
    }
    return n;
}

public static Node organizeBooks(Node head, Integer[] popularity) {
   int n = popular.length;
   for (int i = 0; i < n - 1; i++) {
     Node current = head;


     Node next = head.next;
     for (int j = 0; j < n - 1 - i; j++) {
       if (popular[j] < popular[j + 1]) {
           int temp = popular[j];
           popular[j] = popular[j + 1];
           popular[j + 1] = temp;
           Object tempBook = current.elem;
           current.elem = next.elem;
           next.elem = tempBook;
          }
          current = current.next;
          next = next.next;
        }
    }
    return head;
}

public static Node alternateMerge(Node head1, Node head2) {
  Node t1 = head1;
  Node t2 = head2;


  while (t1 != null && t2 != null) {
     Node next1 = t1.next;
     Node next2 = t2.next;
     t1.next = t2;
     if (next1 == null) {
            break;
     }
     t2.next = next1;
     t1 = next1;
     t2 = next2;
    }
    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) 
    Node curr = head1;
    Node prev = null;
    while (curr != null) {
        Node next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }
    Node newH1 = prev;
    Node tail = newH1;
    while (tail.next != null) {


        tail = tail.next;
    }
    while (head2 != null && head3 != null) {
      int sum = (Integer) head2.elem + (Integer) head3.elem;
      if (sum >= 10) {
            sum = sum % 10;
        }
        tail.next = new Node(sum, null);
        tail = tail.next;
        head2 = head2.next;
        head3 = head3.next;
    }

    return newH1;
}

public static void sumOddAppend(Node dh) {
  Node prev = dh;
  Node curr = dh.next;
  int sum = 0;
  while (curr != dh) {
    if ((Integer) curr.elem % 2 != 0) {


       sum = sum + (Integer) curr.elem;
       prev.next = curr.next;
       curr=curr.next;
     }
    else {
      prev=curr;
      curr=curr.next;
        }
    }
    Node newNode = new Node(sum, dh);
    prev.next = newNode;
}

public static void pairJoin(DNode dh1, DNode dh2) {
  DNode t1 = dh1.next;
  DNode t2 = dh2.next;
  while (t1 != null && t2 != null) {
     DNode next1 = t1.next;
     DNode next2 = t2.next;
     t1.next = t2;
     t2.prev = t1;
     t2.next = next1;
     if (next1 != null) {
       next1.prev = t2;
    }
     t1 = next1;
     t2 = next2;
    }
}

public static void rangeMove(DNode dh, int start, int end) {
  DNode curr = dh.next;
  DNode tail = dh.prev;
  while (curr != dh) {
   DNode next = curr.next;
   int value = (Integer) curr.elem;
   if (value >= start && value <= end) {
        if (curr != tail) {
           curr.prev.next = curr.next;
           curr.next.prev = curr.prev;
           tail.next = curr;
           curr.prev = tail;
           curr.next = dh;
           dh.prev = curr;
           tail = curr;
         }
      }
    curr = next;
    }
