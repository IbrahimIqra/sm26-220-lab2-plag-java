public class AssignmentTask1 {
  


    public static String checkSimilar( Node building1, Node building2 ){
        
        Node tem1 = building1;
        Node tem2 = building2;

        while(tem1 != null && tem2 != null){
            if(!tem1.elem.equals(tem2.elem)){
                return "Not Similar";
            }
            
            tem1 = tem1.next;
            tem2 = tem2.next;
        }
        
        if(tem1 == null && tem2 == null){
          return "Similar";
        }
        else{
          return "Not Similar";
        }                
    }

public class AssignmentTask3 {
    
    // MUST SUBMIT this method
    public static Node alternateMerge( Node head1, Node head2 ){
        Node temp1 = head1;
        Node temp2 = head2;

        while(temp1 != null && temp2 != null){


            Node n1 = temp1.next;
            Node n2 = temp2.next;

            temp1.next = temp2;
            temp2.next = n1;

            temp1 = n1;
            temp2 = n2;
        }
        
      return head1;
    }

public class AssignmentTask4{
    
    // MUST SUBMIT this method
    public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node temp1 = head1;
        Node temp2 = head2;
        Node temp3 = head3;

        Node revHead = null;
        
        while(temp1 != null){
            Node N = new Node(temp1.elem);
            N.next = revHead;
            revHead = N;
            temp1 = temp1.next;
        }
        Node temporary = revHead;
        while(temporary.next != null){


          temporary = temporary.next;
        }
        
        while(temp2 != null && temp3 != null){

            int sum = (Integer)temp2.elem + (Integer)temp3.elem;
            if(sum >= 10){
                sum = sum % 10;
            }
            Node N = new Node(sum);
            temporary.next = N;
            temporary = N;
           
            temp2 = temp2.next;
            temp3 = temp3.next;
        }            
        return revHead; // Remove this when you're ready to return the new head
    }

public class AssignmentTask5 {

    //SUBMIT ONLY THIS METHOD


    public static void sumOddAppend(Node dh) {
        int sum = 0;
        Node p = dh;
        Node q = dh.next;

        while(q != dh){
            if((Integer)q.elem % 2 != 0){
                sum += (Integer)q.elem;
                p.next = q.next;
                q = q.next;
            }
            else {
                p = q;
                q = q.next;
            
            }
        }
        Node N = new Node(sum);
        Node temp = dh;
        while(temp.next != dh){
            temp = temp.next;
        }
        temp.next = N;
        N.next = dh;       
        }

public class AssignmentTask6 {

    public static void pairJoin(DNode dh1, DNode dh2) {
      DNode p1 = dh1.next;
      DNode p2 = dh2.next;
      
      while(p1 != null && p2 != null){
        DNode q1 = p1.next;
        DNode q2 = p2.next;
        
        p1.next = p2;
        p2.prev = p1;
        p2.next = q1;
        
        if(q1 != null){
          q1.prev = p2;
        }
        
        p1 = q1;
        p2 = q2;
      }
      DNode temp = dh1;
      while(temp.next != null){
        temp = temp.next;
      }
      temp.next = dh1;
      dh1.prev = temp;
    }
