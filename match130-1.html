// AssignmentTask1: Building Blocks



public class AssignmentTask1 {
  public static String checkSimilar(Node building1, Node building2) {
  Node n1 = building1;
  Node n2 = building2;

  while (n1!=null&& n2!=null){
    if (!n1.elem.equals(n2.elem)){
      return "Not Similar";
    }
    n1 = n1.next;
    n2 = n2.next;
  }
  if (n1==null&&n2==null){
    return "Similar";
  }else{
    return "Not Similar";
  }
}

import java.util.Arrays;

public class AssignmentTask2 {
  public static Node organizeBooks(Node head, Integer[] popularity) {
  if (head==null){


    return null;
  }
  Node n1=head;
  int i=0;
  while (n1!=null){
    Node n2=n1.next;
    int j=i + 1;
    while (n2!=null){
      if (popularity[j]>popularity[i]){
        Object tempE=n1.elem;
        n1.elem=n2.elem;
        n2.elem=tempE;
        Integer tempP=popularity[i];
        popularity[i]=popularity[j];
        popularity[j]=tempP;
      }
      n2=n2.next;
      j++;
    }
    n1=n1.next;
    i++;
  }
  return head;
}
// AssignmentTask3: Alternate Merge



public class AssignmentTask3 {
  public static Node alternateMerge(Node head1, Node head2) {
    if (head1==null) 
      return head2;
    if (head2==null) 
      return head1;

    Node n1=head1;
    Node n2=head2;
    while (n1!=null&&n2!= null) {
      Node temp1=n1.next;
      Node temp2=n2.next;
      n1.next=n2;
      if (temp1!=null){
        n2.next=temp1;
      }
      n1=temp1;
      n2=temp2;
    }
    return head1;
  }
}

public class AssignmentTask4 {
  public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node h=null;
    Node n1=head1;

    while (n1!=null){
      Node newNode=new Node(n1.elem);
      newNode.next=h;
      h=newNode;
      n1=n1.next;
    }
    Node t=h;
    while (t.next!=null){
      t=t.next;
    }
    Node n2=head2;
    Node n3=head3;

    while (n2!=null&& n3!=null){
      int v2=(int) n2.elem;
      int v3=(int) n3.elem;
      int s=(v2 + v3)%10;

      t.next=new Node(s);
      t=t.next;

      n2=n2.next;
      n3=n3.next;
    }
    return h;
  }

public class AssignmentTask5 {
  public static void sumOddAppend(Node dh){
  Node p=dh;
  Node c=dh.next;
  int s=0;

  while (c!=dh){
    int val=(Integer) c.elem;

    if (val%2!=0){
      s=s+val;
      p.next=c.next;
      c=c.next;
    }else{
      p=c;


      c=c.next;
    }
  }
  Node n=new Node(s);
  p.next=n;
  n.next=dh;
}

public class AssignmentTask6 {
  public static void pairJoin(DNode dh1, DNode dh2) {
  DNode val1=dh1.next;
  DNode val2=dh2.next;
  DNode tail=dh1;

  while (val1!=null){
    DNode temp1=val1.next;
    DNode temp2=val2.next;
    tail.next=val1;
    val1.prev=tail;
    val1.next=val2;
    val2.prev=val1;
    tail=val2;
    val1=temp1;
    val2=temp2;
  }
  tail.next=dh1;
  dh1.prev=tail;
}

public class AssignmentTask7 {
  public static void rangeMove(DNode dh, int start, int end) {
  DNode val1=dh.next;

  while (val1!=dh){
    DNode val2=val1.next;
    int mVal=(int) val1.elem;

    if (mVal>=start&& mVal<=end){
      DNode before=val1.prev;
      DNode after=val1.next;
      DNode tail=dh.prev;
      if (val1!=tail){
        before.next=after;
        after.prev=before;
        tail.next=val1;
        val1.prev=tail;
        val1.next=dh;
        dh.prev=val1;
      }
    }
    val1=val2;
  }
}
