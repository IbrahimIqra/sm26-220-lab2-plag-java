class Node {
    String color;
    Node next;

    Node(String color) {
        this.color = color;
    }
}

class Building {
    Node head;

    void append(String color) {
        Node newNode = new Node(color);
        if (head == null) {
            head = newNode;
            return;
        }
        Node current = head;
        while (current.next != null) {
            current = current.next;
        }
        current.next = newNode;
    }

    void print() {


        Node current = head;
        while (current != null) {
            System.out.print(current.color);
            if (current.next != null) System.out.print(" -> ");
            current = current.next;
        }
        System.out.println();
    }
}

public class labTask1{

    static boolean areSimilar(Building b1, Building b2) {
        Node n1 = b1.head;
        Node n2 = b2.head;

        while (n1 != null && n2 != null) {
            boolean sameColor = n1.color.equals(n2.color);
            if (!sameColor) {


                return false;
            }
            n1 = n1.next;
            n2 = n2.next;
        }
        return n1 == null && n2 == null;
    }

    public static void main(String[] args) {


        Building building1 = new Building();
        building1.append("Red");
        building1.append("Green");
        building1.append("Yellow");
        building1.append("Red");
        building1.append("Blue");
        building1.append("Green");

        Building building2 = new Building();
        building2.append("Red");
        building2.append("Green");
        building2.append("Yellow");
        building2.append("Red");
        building2.append("Blue");
        building2.append("Red");

        System.out.print("building_1 = ");
        building1.print();
        System.out.print("building_2 = ");
        building2.print();

        if (areSimilar(building1, building2)) {
            System.out.println("Similar");
        } else {
            System.out.println("Not Similar");
        }
    }
}

class Node {
    String title;
    int score;
    Node next;

    Node(String title) {
        this.title = title;
    }
}

public class labTask2 {

    // bubble sort: swap data of adjacent nodes if out of order (stable, in-place)
    static Node organizeBooks(Node head, int[] popularity) {

        // assign each node its popularity score
        Node current = head;
        int i = 0;
        while (current != null) {
            current.score = popularity[i++];
            current = current.next;
        }

        boolean swapped = true;
        while (swapped) {
            swapped = false;
            Node node = head;
            while (node != null && node.next != null) {
                if (node.score < node.next.score) {
                    String t = node.title; int s = node.score;
                    node.title = node.next.title; node.score = node.next.score;
                    node.next.title = t; node.next.score = s;
                    swapped = true;
                }
                node = node.next;
            }
        }
        return head;
    }

    static Node buildList(String[] titles) {


        Node head = new Node(titles[0]);
        Node tail = head;
        for (int i = 1; i < titles.length; i++) {
            tail.next = new Node(titles[i]);
            tail = tail.next;
        }
        return head;
    }

    static void print(Node head) {
        while (head != null) {
            System.out.print(head.title);
            if (head.next != null) {
                System.out.print(" -> ");
            } else {
                System.out.print("\n");
            }


        head = head.next;
    }
}

    public static void main(String[] args) {
        String[] titles = {"Hamlet", "Persuasion", "IT", "Dracula", "Beloved"};
        int[] popularity = {7, 9, 9, 6, 7};

        Node head = buildList(titles);
        print(organizeBooks(head, popularity));
    }
}

class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
    }
}

public class labTask3 {

    static Node alternateMerge(Node head1, Node head2) {
        Node p1 = head1, p2 = head2;

        while (p1 != null && p2 != null) {
            Node p1Next = p1.next;
            Node p2Next = p2.next;

            p2.next = p1Next;
            p1.next = p2;

            p1 = p1Next;
            p2 = p2Next;
        }
        return head1;
    }

    static Node buildList(int[] values) {


        Node head = new Node(values[0]);
        Node tail = head;
        for (int i = 1; i < values.length; i++) {
            tail.next = new Node(values[i]);
            tail = tail.next;
        }
        return head;
    }

    static void print(Node head) {
        while (head != null) {


            System.out.print(head.data);
            if (head.next != null) {
                System.out.print(" -> ");
            }
            head = head.next;
        }
        System.out.println();
    }

    public static void main(String[] args) {
        Node list1 = buildList(new int[]{1, 2, 6, 8, 11});
        Node list2 = buildList(new int[]{5, 7, 3, 9, 4});
        print(alternateMerge(list1, list2));

        Node list3 = buildList(new int[]{5, 3, 2, -4});
        Node list4 = buildList(new int[]{-4, -6, 1});
        print(alternateMerge(list3, list4));

        Node list5 = buildList(new int[]{4, 2, -2, -4});
        Node list6 = buildList(new int[]{8, 6, 5, -3});
        print(alternateMerge(list5, list6));
    }
}

class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
    }
}

public class labTask4 {

    // reverse the list and returns new head
    static Node reverse(Node head) {
        Node prev = null;
        while (head != null) {
            Node next = head.next;
            head.next = prev;
            prev = head;
            head = next;
        }
        return prev;
    }

    // builds the 8 digit student id
    static Node idGenerator(Node head1, Node head2, Node head3) {
        Node result = reverse(head1);
        Node tail = result;
        while (tail.next != null) {
            tail = tail.next;
        }



        Node cur2 = head2, cur3 = head3;
        while (cur2 != null && cur3 != null) {
            int sum = cur2.data + cur3.data;
            if (sum >= 10) {
                sum = sum % 10;
            }
            Node next2 = cur2.next;
            cur2.data = sum;
            tail.next = cur2;
            tail = cur2;
            cur2 = next2;
            cur3 = cur3.next;
        }
        tail.next = null;
        return result;
    }

    static Node buildList(int[] values) {


        Node head = new Node(values[0]);
        Node tail = head;
        for (int i = 1; i < values.length; i++) {
            tail.next = new Node(values[i]);
            tail = tail.next;
        }
        return head;
    }

    static void print(Node head) {
        while (head != null) {
            System.out.print(head.data);
            if (head.next != null) {
                System.out.print(" -> ");
            }
            head = head.next;
        }


        System.out.println();
    }

    public static void main(String[] args) {
        Node list1 = buildList(new int[] { 0, 3, 2, 2 });
        Node list2 = buildList(new int[] { 5, 2, 2, 1 });
        Node list3 = buildList(new int[] { 4, 3, 2, 1 });
        print(idGenerator(list1, list2, list3));

        Node list4 = buildList(new int[] { 0, 3, 9, 1 });
        Node list5 = buildList(new int[] { 3, 6, 5, 7 });
        Node list6 = buildList(new int[] { 6, 2, 5, 6 });
        print(idGenerator(list4, list5, list6));
    }
}

class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
    }
}

public class labTask5 {

    static void sumOddAppend(Node dummy) {
        int sum = 0;
        Node prev = dummy;
        Node current = dummy.next;

        while (current != dummy) {
            if (current.data % 2 != 0) {
                sum = sum + current.data;
                prev.next = current.next;
                current = prev.next;
            } else {
                prev = current;
                current = current.next;
            }
        }

        Node newNode = new Node(sum); //the only extra node we are allowed to create
        newNode.next = dummy;
        prev.next = newNode;
    }


    static Node buildCircularList(int[] values) {
        Node dummy = new Node(0);
        Node tail = dummy;
        for (int v : values) {
            tail.next = new Node(v);
            tail = tail.next;
        }
        tail.next = dummy;
        return dummy;
    }



    static void print(Node dummy) {
        System.out.print("X");
        Node current = dummy.next;
        while (current != dummy) {
            System.out.print(" -> " + current.data);
            current = current.next;
        }
        System.out.println(" -> X");
    }




    public static void main(String[] args) {
        Node dummy = buildCircularList(new int[]{11, 22, 33, 44, 55, 66});
        sumOddAppend(dummy);
        print(dummy);
    }
}

class Node {
    char data;
    Node prev;
    Node next;

    Node(char data) {
        this.data = data;
    }
}

public class labTask6 {

    static void pairJoin(Node dummy1, Node dummy2) {
        Node p1 = dummy1.next;
        Node p2 = dummy2.next;
        Node prev = dummy1;

        while (p1 != null) {
            Node next1 = p1.next;
            Node next2 = p2.next;

            prev.next = p1;
            p1.prev = prev;
            prev = p1;

            prev.next = p2;
            p2.prev = prev;
            prev = p2;

            p1 = next1;
            p2 = next2;
        }
        prev.next = null;
    }


    static Node buildList(char[] values) {
        Node dummy = new Node('X');
        Node tail = dummy;
        for (char v : values) {
            Node node = new Node(v);
            node.prev = tail;
            tail.next = node;
            tail = node;
        }
        return dummy;
    }



    static void print(Node dummy) {
        System.out.print(dummy.data);
        Node current = dummy.next;
        while (current != null) {
            System.out.print(" <-> " + current.data);
            current = current.next;
        }
        System.out.println();
    }






    public static void main(String[] args) {
        Node list1 = buildList(new char[]{'A', 'B', 'C', 'D'});
        Node list2 = buildList(new char[]{'E', 'F', '8', 'H'});

        pairJoin(list1, list2);
        print(list1);
    }
}

class Node {
    int data;
    Node prev;
    Node next;

    Node(int data) {
        this.data = data;
    }
}

public class labTask7 {

    static void rangeMove(Node dummy, int start, int end) {
        Node current = dummy.next;
        Node movedHead = null;
        Node movedTail = null;

        while (current != dummy) {


            Node next = current.next;

            if (current.data >= start && current.data <= end) {
                current.prev.next = current.next;
                current.next.prev = current.prev;
                if (movedHead == null) {
                    movedHead = current;
                } else {
                    movedTail.next = current;
                    current.prev = movedTail;
                }
                movedTail = current;
                current.next = null;
            }
            current = next;
        }

        if (movedHead != null) {
            Node tail = dummy.prev;
            tail.next = movedHead;
            movedHead.prev = tail;
            movedTail.next = dummy;
            dummy.prev = movedTail;
        }
    }

    
    static Node buildList(int[] values) {
        Node dummy = new Node(0);


        Node tail = dummy;
        for (int v : values) {
            Node node = new Node(v);
            node.prev = tail;
            tail.next = node;
            tail = node;
        }
        tail.next = dummy;
        dummy.prev = tail;
        return dummy;
    }



    static void print(Node dummy) {
        System.out.print("X");
        Node current = dummy.next;
        while (current != dummy) {
            System.out.print(" <-> " + current.data);
            current = current.next;
        }
        System.out.println(" <-> X");
    }



    public static void main(String[] args) {
        Node dummy = buildList(new int[]{5, 3, 7, 1, 9, 6, 2, 4});
        rangeMove(dummy, 5, 7);
        print(dummy);
    }
}
