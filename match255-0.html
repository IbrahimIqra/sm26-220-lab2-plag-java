public static String checkSimilar( Node building1, Node building2 ){
    Node temp1 = building1;
    Node temp2 = building2;
    
    while(temp1!=null && temp2!=null){     
      if(!temp1.elem.equals(temp2.elem)){
        return "Not Similar";
      }      
      temp1 = temp1.next;
      temp2 = temp2.next;
    }
    if (temp1!=null || temp2!=null){
      return "Not Similar";
    }    
    else{
      return "Similar";
    }    
  }

public static Node organizeBooks(Node head, Integer[] popularity){    
    int n = popularity.length;
    
    for(int i=0; i<n-1; i++){      
      Node current = head;      
      for(int j=0; j<n-i-1; j++){        
        if(popularity[j] < popularity[j+1]){
          int temp = popularity[j];
          popularity[j] = popularity[j+1];
          popularity[j+1] = temp;
          Object tempBook = current.elem;
          current.elem = current.next.elem;
          current.next.elem = tempBook;
        }        
        current = current.next;
      }
    }    
    return head;
  }

public static Node alternateMerge( Node head1, Node head2 ){    
    Node temp1 = head1;
    Node temp2 = head2;
    


    while (temp1!=null && temp2!=null){      
      Node next1 = temp1.next;
      Node next2 = temp2.next;    
      temp1.next = temp2;          
      if(next1==null){
        break;
      }    
      temp2.next = next1;          
      temp1 = next1;
      temp2 = next2;
    }    
    return head1;    
  }

public static void sumOddAppend(Node dh){    
    Node prev = dh;
    Node current = dh.next;
    int sum = 0;      
    while(current!=dh){      
      if(((Integer)current.elem)%2 != 0){
        sum += (Integer)current.elem;


        prev.next = current.next;
        current = prev.next;
      } 
      else{
        prev = current;
        current = current.next;
      }
    }
    Node newNode = new Node(sum);


    Node temp = dh;
    while(temp.next != dh){
      temp = temp.next;
    }
    temp.next = newNode;
    newNode.next = dh;
  }

public static void pairJoin(DNode dh1, DNode dh2){    
    DNode prevL1 = dh1.next;
    DNode prevL2 = dh2.next;
    
    while(prevL1!=null && prevL2!=null){      
      DNode next1 = prevL1.next;
      DNode next2 = prevL2.next;            


      prevL1.next = prevL2;
      prevL2.prev = prevL1;      
      prevL2.next = next1;     
      if(next1!=null){
        next1.prev = prevL2;
      }    
      prevL1 = next1;
      prevL2 = next2;
    }
  }
