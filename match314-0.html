public static Node organizeBooks(Node head, Integer[] popularity) {

    int n = popularity.length; 
    for (int i=0;i<n-1;i++) {
        Node prev=null;
        Node curr=head;
        for (int j=0;j<n-i-1;j++) {
            Node next = curr.next;
            if (popularity[j]<popularity[j + 1]) {
                // swap popularity values
                int temp = popularity[j];
                popularity[j] = popularity[j + 1];
                popularity[j + 1] = temp;
                // swap nodes
                curr.next = next.next;
                next.next = curr;
                if (prev == null)
                    head = next;
                else
                    prev.next = next;

                prev = next;
            }
            else {


                prev = curr;
                curr = curr.next;
            }
        }
    }

    return head;
}

public static Node alternateMerge( Node head1, Node head2 ){
    
    Node temp1 = head1;
    Node temp2 = head2;
    while(temp1!=null && temp2!=null){
        Node next1 = temp1.next;
        Node next2 = temp2.next;
        temp1.next = temp2;
        if(next1 == null){
            break;
        }
        temp2.next = next1;
        temp1 = next1;
        temp2 = next2;
    }
    return head1;
    }

public static void sumOddAppend(Node dh) {

    int sum = 0;
    Node prev = dh;


    Node curr = dh.next;
    while (curr != dh) {
        if ((int)curr.elem%2!=0) {
            sum += (int) curr.elem;
            prev.next = curr.next;
            curr = prev.next;
        } 
        else {
            prev = curr;
            curr = curr.next;
        }
    }
    Node nNode = new Node(sum);
    Node temp = dh;
    while (temp.next != dh) {


        temp = temp.next;
    }
    temp.next = nNode;
    nNode.next = dh;
}

 public static void pairJoin(DNode dh1, DNode dh2) {

    DNode a = dh1.next;
    DNode b = dh2.next;
    while (a!=null && b!=null) {
        DNode temp1 = a.next;
        DNode temp2 = b.next;


        a.next = b;
        b.prev = a;
        b.next = temp1;
        if (temp1 != null) {
            temp1.prev = b;
        }

        a=temp1;
        b=temp2;
        }
    DNode c = dh1;
    while (c.next != null) {
        c = c.next;
     }
    c.next = dh1;
    dh1.prev = c;
}

public static void rangeMove(DNode dh, int start, int end) {

    DNode a=dh.prev;
    DNode curr= dh.next;
    while (curr!=dh) {
        DNode next=curr.next;
        int val=(int) curr.elem;
        if (val>=start && val<=end) {
           if (curr != a) {
               curr.prev.next = curr.next;
               curr.next.prev = curr.prev;
               a.next = curr;
               curr.prev = a;
               curr.next = dh;
               dh.prev = curr;

                a = curr;
            }
        }
          curr = next;
    }
