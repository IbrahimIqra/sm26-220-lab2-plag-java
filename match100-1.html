public static Node organizeBooks(Node head, Integer[] popularity) {

        int n=popularity.length;

    for(int i=0;i<n-1;i++){
        
        Node current=head;

        for(int j=0;j<n-1-i;j++){
                
                if (popularity[j+1]>popularity[j]){
                    int  a=popularity[j];


                    popularity[j]=popularity[j+1];
                    popularity[j+1]=a;

                    String s = (String) current.elem;
                    current.elem=current.next.elem;
                    current.next.elem=s;
                }
            
            current=current.next;
        }
    }
        return head;
    }

 public static Node alternateMerge( Node head1, Node head2 ){
        
        Node temp1=head1;
        Node temp2=head2;

        while(temp1!=null && temp2!=null){
            Node next1=temp1.next;
            Node next2=temp2.next;

            temp1.next=temp2;


            temp2.next=next1;

            temp1=next1;
            temp2=next2;
        }
        return head1;
    }

public static void sumOddAppend(Node dh) {
        Node prev=dh;
        Node current=dh.next;
        int sum=0;

        while(current!=dh){
            if((Integer)current.elem%2!=0){
                sum+=(Integer)current.elem;


                prev.next=current.next;
                current=prev.next;
            }
           else{
            prev=current;
            current=current.next;
           }
        }
       
        Node newNode=new Node(sum);
        newNode.next=dh;
        prev.next=newNode;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode a1=dh1.next;
        DNode a2=dh2.next;
        dh2.next=null;

        while(a1!=null && a2!=null){
            DNode next1=a1.next;


            DNode next2=a2.next;

            a1.next=a2;
            a2.prev=a1;
            a2.next=next1;

            if(next1!=null){
                next1.prev=a2;
            }

            a1=next1;
            a2=next2;
        }
        DNode tail = dh1;
       while (tail.next != null) {
         tail=tail.next;
       }

       tail.next = dh1;
       dh1.prev = tail;
    }

  public static void rangeMove(DNode dh, int start, int end) {
        DNode tail=dh.prev;
        DNode last=tail;

        DNode current=dh.next;
        while(current!=last.next){
            DNode next=current.next;
            int value=(Integer)current.elem;

            if(value>=start && value<=end){
                if(current!=tail){
                    current.prev.next=current.next;
                    current.next.prev=current.prev;

                    tail.next=current;
                    current.prev=tail;

                    current.next=dh;
                    dh.prev=current;

                    tail=current;
                }
            }
            current=next;
        }
    }
