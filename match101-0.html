public static String checkSimilar( Node building1, Node building2 ){
        Node b1 = building1;
        Node b2 = building2;
        while (b1!=null && b2!=null){
            if(!b1.elem.equals(b2.elem)){


                return "Not Similar";
            }
            b1 = b1.next;
            b2 = b2.next;
        }
        if(b1==null && b2==null){
            return "Similar";
        }


        return "Not Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {
        int n = popularity.length;
        for (int i=0; i< n-1; i++){
            Node current = head;
            for (int j=0; j< n-1-i; j++){
                if (popularity[j] < popularity[j+1]){
                    int tempPopularity = popularity[j];
                    popularity[j] = popularity[j+1];
                    popularity[j+1] = tempPopularity;
                    Object tempElem = current.elem;
                    current.elem = current.next.elem;
                    current.next.elem = tempElem;
                }
                current = current.next;
            }
        }
        return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){
        Node h1 = head1;
        Node h2 = head2;
        while (h1!=null && h2!=null){
            Node next1 = h1.next;
            Node next2 = h2.next;
            h1.next = h2;
            if(next1==null){
                break;
            }


            h2.next = next1;
            h1 = next1;
            h2 = next2;
        }
        return head1;
    }

public static void sumOddAppend(Node dh) {
        int sum = 0;
        Node prev = dh;
        Node cur = dh.next;
        while (cur!=dh){
            if((int)cur.elem % 2 != 0){
                sum+= (int)cur.elem;


                prev.next = cur.next;
                cur = prev.next;
            }
            else{
                prev = cur;
                cur = cur.next;
            }
        }
        Node newNode = new Node(sum);
        prev.next = newNode;
        newNode.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode last = dh1;
        DNode p1 = dh1.next;
        DNode p2 = dh2.next;
        while (p1!=null && p2!=null){
            DNode next1 = p1.next;
            DNode next2 = p2.next;
            p1.next = p2;
            p2.prev = p1;
            if(next1!=null){
                p2.next = next1;
                next1.prev = p2;
            }
            else{
                p2.next = null;
            }


            last = p2;
            p1 = next1;
            p2 = next2;
        }
        last.next = dh1;
        dh1.prev = last;
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode t1 = dh.prev;
        DNode t2 = t1;
        DNode cur = dh.next;
        while(true){
            DNode next = cur.next;
            int value = (int)cur.elem;


            if(value>=start && value<=end){
                cur.prev.next = cur.next;
                cur.next.prev = cur.prev;
                t1.next = cur;
                cur.prev = t1;
                cur.next = dh;
                dh.prev = cur;
                t1 = cur;
            }
            if(cur==t2){
                break;
            }
            cur = next;
        }
