            if (t1Next == null)
                break;

            t1 = t1Next;

                t2Next = t2.next;
            }

        }

public static Node idGenerator(Node head1, Node head2, Node head3) {

        // TO DO
        // Hint: the Node elements are actually Object, you can type cast them
        // into int or Integer like the following:
        // (int)n.elem or (Integer)n.elem

        // *** approach: first LL reverse korsi, then reverse 1st LL er last node ke
        // tail dhorsi, erpor sum naame new node create kore tail er sathe add korsi,
        // then tail ke update korsi, and finally return prev (new head of the reversed
        // first list) ***
        Node t1 = head1;
        Node t2 = head2;
        Node t3 = head3;

        Node prev = null;
        Node curr = t1;
        Node next = null;

        while (curr != null) {
            next = curr.next;


            curr.next = prev;
            prev = curr;
            curr = next;
        }

        Node tail = prev;

        while (tail.next != null) {
            tail = tail.next;
        }

        while (t2 != null && t3 != null) {



            int sum = (((Integer) t2.elem + (Integer) t3.elem) % 10);

            Node n = new Node(sum);

            tail.next = n;
            tail = n;

            t2 = t2.next;
            t3 = t3.next;
        }

        return prev;
    }

 public static void sumOddAppend(Node dh) {
        // TO DO
        int sum = 0;
        Node prev = dh;
        Node curr = dh.next;
        while (curr != dh) {
            if ((Integer) curr.elem % 2 != 0) {
                sum += (Integer) curr.elem;


                prev.next = curr.next;
                curr = curr.next;
            } else {
                prev = curr;
                curr = curr.next;
            }
        }

        Node lastNode = new Node(sum);
        prev.next = lastNode;
        lastNode.next = dh;

    }

public static void pairJoin(DNode dh1, DNode dh2) {

        DNode temp = dh1.next;
        DNode curr = dh2.next;

        while (temp != null && curr != null) {

            DNode next1 = temp.next;
            DNode next2 = curr.next;

            temp.next = curr;
            curr.prev = temp;

            if (next1 != null) {
                curr.next = next1;
                next1.prev = curr;

            }
            temp = next1;


            curr = next2;

        }
        // circular korar jonno last node e jawar jonno loop and tail diye save kortesi

        DNode tail = dh1;
        while (tail.next != null) {
            tail = tail.next;
        }
        tail.next = dh1;
        dh1.prev = tail;

    }

 public static void rangeMove(DNode dh, int start, int end) {
        // TO DO
        if (dh == null || dh.next == dh) {
            return; // Empty list, kichu kora lagbe na
        }

        // infinite loop e na porar karone count kortesi
        int count = 0;
        DNode temp = dh.next;
        while (temp != dh) {
            count++;
            temp = temp.next;

        }

        DNode curr = dh.next;
        for (int i = 0; i < count; i++) {
            DNode nextNode = curr.next; // 3 jeno haraye na jay



            int val = (Integer) curr.elem;
            if (val >= start && val <= end) {
                // 5 ke ekdom last e niye jacchi
                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;

                DNode lastNode = dh.prev;

                lastNode.next = curr;
                curr.prev = lastNode;
                curr.next = dh;
                dh.prev = curr;
            }
            curr = nextNode; // 5 theke 3 te nilam
        }

    }
