 public static String checkSimilar(Node building1, Node building2) {

        
        Node current1=building1;
        Node current2=building2;
        while(current1 !=null && current2 !=null){
            if(!current1.elem.equals(current2.elem)){
                return "Not Similar";
            }
            current1 = current1.next;
            current2= current2.next;
        }
        if(current1 == null && current2 == null){
            return "Similar";
        }
        return "Not Similar";


       
        
    }

public static void sumOddAppend(Node dh) {

            if((int) current.elem % 2 !=0){
                sum += (int) current.elem;

                current = current.next;

            else{
                prev = current;
                current = current.next;
            }
            }

  public static void rangeMove(DNode dh, int start, int end) {
        DNode tail = dh.prev;
        DNode current = dh.next;


        while(current !=dh){
            DNode next = current.next;
            int value = (int)current.elem;
           
            if(value >= start && value <= end){
                if(current==tail){
                    tail=current;
                }
                else{


                    current.prev.next= current.next;
                    current.next.prev=current.prev;


                    tail.next = current;
                    current.prev=tail;


                    current.next=dh;
                    dh.prev = current;
                    tail= current;
                }
             
            }
            current = next;
        }
