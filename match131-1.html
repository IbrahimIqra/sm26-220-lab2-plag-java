 public static Node organizeBooks(Node head, Integer[] popularity) {



     int s = popularity.length;
     for(int i = 0; i < s - 1; i++) {
      Node n = head;
       for(int j = 0; j < s - 1 - i; j++) {
        if(popularity[j] < popularity[j + 1]) {         
          int t = popularity[j];
          popularity[j] = popularity[j + 1];
          popularity[j + 1] = t;


          Object x = n.elem;
          n.elem = n.next.elem;
          n.next.elem = x;
            } 
            n = n.next;
        }
    }
    return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){
      
    Node p = head1;
    Node q = head2;
    while(p != null && q != null){
     Node p1 = p.next;
     Node q1 = q.next;
     p.next = q;
      if(p1 == null){
       break;
        }
        q.next = p1;
        p = p1;
        q = q1;
    }
         return head1;
    }

 public static void sumOddAppend(Node dh) {
       
     int sum = 0;
     Node p = dh;
     Node c = dh.next;
      while(c != dh){
       int v = (Integer)c.elem;
        if(v % 2 != 0){
         sum = sum + v;


         p.next = c.next;
         c = c.next;
        }
        else{
          p = c;
          c = c.next;
        }
    }
    Node n = new Node(sum);


    p.next = n;
    n.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
       
    DNode p = dh1.next;
    DNode q = dh2.next;
     while (p != null && q != null) {
      DNode p1 = p.next;
      DNode q1 = q.next;
      p.next = q;
      q.prev = p;
       if (p1 != null) {


        q.next = p1;
        p1.prev = q;
        }
         p = p1;
         q = q1;
    }
    DNode t = dh1;
     while (t.next != null) {
      t = t.next;
    }
       t.next = dh1;
       dh1.prev = t;
    }
