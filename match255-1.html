        public static String checkSimilar( Node building1, Node building2 ){
      
        Node n1= building1;
        Node n2= building2;


        while(n1!= null && n2!=null){
            if(!(n1.elem.equals(n2.elem))){
                return "Not Similar";


            }
            
            n1=n1.next;
            n2=n2.next;
        }
        if(n1!=null|| n2!=null){
            return "Not Similar";
        }


        return "Similar";
   }

public static void rangeMove(DNode dh, int start, int end) {
        DNode current=dh.next;
        DNode last=dh.prev;
        while(current!=dh){
            DNode next=current.next;
            if((int)current.elem>=start && (int)current.elem<=end){
                current.prev.next=current.next;
                current.next.prev=current.prev;
                current.prev=dh.prev;
                current.next=dh;
                dh.prev.next=current;
                dh.prev=current;
            }
            if((current==last)){
                break;
            }
            current=next;
        }
   }
