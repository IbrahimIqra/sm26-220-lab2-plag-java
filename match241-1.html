public static String checkSimilar( Node building1, Node building2 ){
        Node temp1= building1;
        Node temp2= building2;
        String flag= "Similar";
        while(temp1!=null && temp2!= null){
            if(temp1.elem!=temp2.elem){
                flag= "Not Similar";
                return flag;
            }
            temp1=temp1.next;
            temp2=temp2.next;
        }
        if(temp1!= null || temp2!= null){
                flag= "Not Similar";
                return flag;
            }
        return flag;
    }

                    int y= popularity[j];
                    popularity[j]=popularity[j+1];
                    popularity[j+1]=y;


                    Object z=temp.elem;
                    temp.elem=temp.next.elem;

                    }

            }
        }

public static void sumOddAppend(Node dh) {

            }
            else{
                tempprev=tempprev.next;

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode temp1=dh1.next;
        DNode temp2=dh2.next;
        while(temp1!=null && temp2!=null){
            DNode temp3=temp1.next;
            DNode temp4=temp2.next;
            temp1.next=temp2;
            temp2.prev=temp1;
            temp2.next=temp3;
            if(temp3!=null){
                temp3.prev=temp2;
            }
            temp1=temp3;
            temp2=temp4;
        }
        DNode l=dh1;


        while(l.next!=null){
            l=l.next;
        }
        l.next=dh1;
        dh1.prev=l;
    }
