public static int count(Node head){
        Node temp = head;
        int count = 0;
        while(temp != null){
            count++;
            temp = temp.next;
        }
        return count;
    }
    public static String checkSimilar( Node building1, Node building2 ){
        boolean flag = true;
        int count1 = count(building1);
        int count2 = count(building2);

        while(building1 != null && building2 != null){
            if(count1 != count2){
                flag = false;
                break;
            }
            else{
            if(((String)building1.elem) != ((String)building2.elem)){
                flag = false;
                break;
            }
            building1 = building1.next;
            building2 = building2.next;
        }
    }

        if(flag){
            return "Similar";
        }
        else{
            return "Not Similar";
        }
    }

public static void sumOddAppend(Node dh){
        int odd_count = 0;
        Node helper = dh;
        Node curr = dh.next;

        while(curr != dh){
            if((int)curr.elem%2 != 0){
                odd_count += (int)curr.elem;


                helper.next = curr.next;
                curr = curr.next;
            }
            else{
               helper = curr;
               curr = curr.next; 
            }
        }

        Node n = new Node(odd_count);
        helper.next = n;
        n.next = dh;  }

public static void pairJoin(DNode dh1, DNode dh2){
        DNode temp1 = dh1.next;
        DNode temp2 = dh2.next;

        while(temp1 != null){
            DNode immediate_next_1 = temp1.next;


            DNode immediate_next_2 = temp2.next;

            temp1.next = temp2;
            temp2.prev = temp1;

            temp2.next = immediate_next_1;
            if(immediate_next_1 != null){
            immediate_next_1.prev = temp2;
            }

            temp1 = immediate_next_1;
            temp2 = immediate_next_2;
        }

        DNode last = dh1;
        while(last.next != null){
            last = last.next;
        }

        if(temp2 != null){
            last.next = temp2;
            temp2.prev = last;
        }
        
        last.next = dh1;
        dh1.prev = last;
    }

public static int count_Nodes(DNode n){
        int counter = 0;
        DNode temp = n.next;


        while(temp != n){
            counter++;
            temp = temp.next;
        }

        return counter;
    }

    public static void rangeMove(DNode dh, int start, int end){
        DNode temp = dh;
        int range = count_Nodes(temp);
        DNode curr = dh.next;

        int i=0;
        while(i<range){


            DNode next_to_work_with = curr.next;
            if((Integer)curr.elem>=start && (Integer)curr.elem<=end){
                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;

                DNode last = dh.prev;
                last.next = curr;
                curr.prev = last;
                curr.next = dh;
                dh.prev = curr;
            }
            curr = next_to_work_with;
            i++;
        }


    }
