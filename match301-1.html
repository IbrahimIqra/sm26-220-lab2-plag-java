public static String checkSimilar( Node building1, Node building2 ){
    Node current_1=building1;
    Node current_2=building2;
    int count_1=0;
    int count_2=0;
    while(current_1!=null){
      count_1++;
      current_1=current_1.next;
    }
    current_1=building1;
    while(current_2!=null){
      count_2++;
      current_2=current_2.next;
    }
    current_2=building2;
    if(count_1<count_2 || count_1>count_2) return "Not Similar";


    while(current_1!=null && current_2!=null){
      if(!current_1.elem.equals(current_2.elem)){
        return "Not Similar";
      }
      current_1=current_1.next;
      current_2=current_2.next;
    }
    return "Similar";
  }

    if(head==null){

    }


    int n=popularity.length;
    for(int i=0;i<n-1;i++){
      Node temp=head;
      for(int j=0;j<n-i-1;j++){
        if(popularity[j]<popularity[j+1]){
          Integer num=popularity[j];


          popularity[j]=popularity[j+1];
          popularity[j+1]=num;
          
          Object book=temp.elem;
          temp.elem=temp.next.elem;
          temp.next.elem=book;
        }
        temp=temp.next;
      }
    }
    return head;
  }

 public static Node alternateMerge( Node head1, Node head2 ){
    Node current_1=head1;
    Node current_2=head2;
    while(current_1!=null && current_2!=null){
      Node next_1=current_1.next;
      Node next_2=current_2.next;
      current_1.next=current_2;
      current_2.next=next_1;
      current_1=next_1;
      current_2=next_2;
      if(next_1==null){
        break;
      }
    }
    return head1;
  }

public static void pairJoin(DNode dh1, DNode dh2){
    DNode a = dh1.next;
    DNode b = dh2.next;
    DNode h = dh1;
    
    while(a!=null && b!=null)
    {
        DNode n1 = a.next;
        DNode n2 = b.next;
        
        a.next = b;
        b.prev = a;
        
        b.next = n1;
        if(n1!=null)
        {
            n1.prev=b;
        }
       a = n1;
        b = n2;
    }
    
    dh2.next = null;
    dh2.prev = null;
    
    DNode last = h;
    while(last.next != null)
    {


        last = last.next;
    }
    last.next = h;
    h.prev = last;
}

public static void rangeMove(DNode dh, int start, int end){
    DNode lastNode=dh.prev;
    DNode stopNode=dh.prev;
    DNode current=dh.next;
    while(true){
      DNode nextNode=current.next;
      if((Integer)current.elem>=start && (Integer)current.elem<=end){
        if(current!=lastNode){


          current.prev.next=current.next;
          current.next.prev=current.prev;
          lastNode.next=current;
          current.prev=lastNode;
          current.next=dh;
          dh.prev=current;
          lastNode=current;
        }
      }
      if(current==stopNode){
        break;
      }
      current=nextNode;
    }
  }
