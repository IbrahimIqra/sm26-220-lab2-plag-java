public static String checkSimilar(Node building1, Node building2) {
    int c1=0;
    Node t1=building1;
    while(t1!=null){
        c1++;
        t1=t1.next;
    }

    int c2=0;
    Node t2=building2;
    while(t2!=null){
        c2++;
        t2=t2.next;
    }

    if (c1!=c2) {
        return "Not Similar";
    }

    Node n1=building1;
    Node n2=building2;
    boolean same=true;

    while (n1!=null) {
        if (!n1.elem.equals(n2.elem)){
            same=false;
        }
        n1=n1.next;
        n2=n2.next;
    }
    if(same==true){
        return "Similar";
    }
    else {
        return "Not Similar";
    }
}

public static Node organizeBooks(Node head, Integer[] popularity) {
    boolean sw=true;
    while (sw) {
        sw=false;
        Node c=head;
        int i=0;



        while (c!=null && c.next!=null) {
            if (popularity[i]<popularity[i+1]) {
                Object te=c.elem;
                c.elem=c.next.elem;
                c.next.elem=te;

                int tp=popularity[i];
                popularity[i]=popularity[i+1];
                popularity[i+1]=tp;


                sw=true;
            }
            c=c.next;
            i++;
        }
    }
    return head;
}

public static Node alternateMerge(Node head1, Node head2) {
    if(head1==null){
        return head2;
    }
    if(head2==null){
        return head1;
    }

    Node nh=head1;
    Node p1=head1;
    Node p2=head2;

    while(p1!=null && p2!=null){
        Node p1n=p1.next;
        Node p2n=p2.next;
        p1.next=p2;

        if(p1n!=null){


            p2.next=p1n;
        }

        p1=p1n;
        p2=p2n;

    }
    return nh;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node prv=null;
    Node cur=head1;
    while(cur!=null){
        Node nxt=cur.next;
        cur.next=prv;
        prv=cur;
        cur=nxt;
    }
    Node rev=prv;
    Node tl=rev;
    while(tl!=null && tl.next!=null){
        tl=tl.next;
    }
    Node p2=head2;
    Node p3=head3;
    while(p2!=null && p3!=null){
        int d2=(Integer)p2.elem;
        int d3=(Integer)p3.elem;
        int sum=d2+d3;


        p2.elem=sum%10;
        p2=p2.next;
        p3=p3.next;
    }
    tl.next=head2;
    return rev;
}

public static void sumOddAppend(Node dh) {
    int sum=0;
    Node prv=dh;
    Node cur=dh.next;
    while(cur!=dh){
        int val=(Integer)cur.elem;
        if(val%2!=0){
            sum+=val;
            prv.next=cur.next;
            cur=prv.next;
        }else{
            prv=cur;
            cur=cur.next;
        }
    }
    Node sn=new Node(sum);
    prv.next=sn;
    sn.next=dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode p1=dh1.next;
    DNode p2=dh2.next;
    DNode lst=dh1;

    while(p1!=null && p2!=null){
        DNode p1n=p1.next;
        DNode p2n=p2.next;
        p1.next=p2;
        p2.prev=p1;
        p2.next=p1n;
        if(p1n!=null){
            p1n.prev=p2;
        }

        lst=p2;
        p1=p1n;
        p2=p2n;
    }

    lst.next=dh1;


    dh1.prev=lst;
}

public static void rangeMove(DNode dh, int start, int end) {
    DNode cur=dh.next;
    DNode rh=null;
    DNode rt=null;
    while(cur!=dh){
        DNode nxt=cur.next;
        int val=(Integer)cur.elem;
        if(val>=start && val<=end){
            cur.prev.next=cur.next;
            cur.next.prev=cur.prev;
            if(rh==null){
                rh=cur;
                rt=cur;
            }else{
                rt.next=cur;
                cur.prev=rt;
                rt=cur;
            }
        }
        cur=nxt;
    }
    DNode tl=dh.prev;
    tl.next=rh;
    rh.prev=tl;
    rt.next=dh;
    dh.prev=rt;
}
