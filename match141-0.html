public class AssignmentTask2{
    
   
    public static Node organizeBooks(Node head, Integer[] popularity) {
        
   if (head == null || head.next == null)
        return head;

    boolean swapped = true;

    while (swapped==true) {
        swapped = false;
        Node current = head;
        int i = 0;



        while (current.next != null) {
            if (popularity[i] < popularity[i + 1]) {
                
                Object tempElem = current.elem;
                 current.elem = current.next.elem;
                 current.next.elem = tempElem;

                
                int tempPop = popularity[i];
                 popularity[i] = popularity[i + 1];
                 popularity[i + 1] = tempPop;

                swapped = true; 
            }

            current = current.next;
             i++;
        }
    }

   return head;
}

    while (n1 != null && n2 != null) {

       K.next = n1;

    public static void rangeMove(DNode dh, int start, int end) {

        DNode temp = dh;
        DNode tail = dh.prev;

       while (temp.next!=originalTail) {
        DNode cur = temp.next;
       
            int val = (int)cur.elem;
     
        if ( val >= start && val  <= end) {
    
            temp.next = temp.next.next;
            cur.next.prev = temp;

            tail.next = cur;
            cur.prev = tail;
            cur.next = dh;

            tail = cur;

            temp = temp.next;
