public static String checkSimilar( Node building1, Node building2 ){
        Node B1 = building1;
        Node B2 = building2;
        
        while (B1 != null && B2 != null){
            if (!B1.elem.equals(B2.elem)){
                return " Not Similar";
            }B1 = B1.next;
            B2 = B2.next;

        }

public static Node alternateMerge( Node head1, Node head2 ){
        

        if (head1 == null) return head2;
        if (head2 == null) return head1;
 
        Node p1 = head1;
        Node p2 = head2;
 
        while (p1 != null && p2 != null) {


            Node p1Next = p1.next;
            Node p2Next = p2.next;
 
            p1.next = p2;
            p2.next = p1Next;
 
            p1 = p1Next;
            p2 = p2Next;
        }
 
        return head1;
    }
