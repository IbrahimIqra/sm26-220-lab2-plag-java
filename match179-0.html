 public static void sumOddAppend(Node dh) {

        Node crnt = dh.next;

        while(crnt!= dh){
            int val = (int) crnt.elem;

                pre = crnt;
            }
            else{
                sum+=val;
                pre.next = crnt.next;
            }


            crnt = crnt.next;
