public static String checkSimilar( Node building1, Node building2 ){
        
        Node temp1 = building1;
        Node temp2 = building2;
        while(temp1 != null && temp2 != null){  
            if(!temp1.data.equals(temp2.data)){


                return "Not Similar";
            }
            temp1 = temp1.next;
            temp2 = temp2.next;
        }
        if(temp1 == null && temp2 == null){
            return "Similar";
        }
        else{
            return "Not Similar";
        } 
    }

    public static Node organizeBooks(Node head, Integer[] popularity) {
        
        int count = popularity.length;


        for (int i=0; i<count; i++){
            Node current = head;
            Node nextNode = current.next;
            for (int j=0; j<count-i-1; j++){
                if (popularity[j]< popularity[j+1]){
                    int currentPopularity = popularity[j];
                    popularity[j]=popularity[j+1];
                    popularity[j+1]=currentPopularity;
                    Object currentData = current.elem;
                    current.elem = nextNode.elem;
                    nextNode.elem = currentData;
                }
                current =current.next;
                nextNode = nextNode.next;
        }
    }
        return head; 
    }

 public static Node alternateMerge( Node head1, Node head2 ){

            Node temp1 = nextNode1.next;

public static void sumOddAppend(Node dh) {
        
        int sum = 0;
        Node dhLL = dh;
        Node current = dh.next;

        while(current!=dh){
            
            if(current.data %2==1){
                sum+=current.data;


                dhLL.next = current.next;
                current = current.next;
            }
            else{
                dhLL = current;
                current = current.next;
            }
        }
        Node sumNode = new Node(sum);
        dhLL.next = sumNode;
        sumNode.next = dh;
        return dh;

    }

 public static void pairJoin(DNode dh1, DNode dh2) {
         
        Node newDHLL1 = dh1.next;
        Node newDHLL2 = dh2.next;
       
        while (newDHLL1 != null) {

            Node newNode1 = newDHLL1.next;
            Node newNode2 = newDHLL2.next;
          
            newDHLL1.next = newDHLL2;


            newDHLL2.prev = newDHLL1;
          
            newDHLL2.next = newNode1;
            newNode1.prev = newDHLL2;
          
           newDHLL1 = newNode1;
           newDHLL2 = newNode2;
        }
        newDHLL2.next = dh1;
        dh1.prev = newDHLL2;
    }

ublic static void rangeMove(DNode dh, int start, int end) {
         
        int size = 0;
        Node pointer = dh.next;
        while (pointer != dh) {
            size++;
            pointer = pointer.next;
        }
        Node before = dh.next;
        
        for (int i = 0; i < size; i++) {
            Node nextNode = before.next;
            
            if (before.data >= start && before.data <= end) {


                before.prev.next =before.next;
                before.next.prev = before.prev;

                Node after = dh.prev;
                after.next = before;
                before.prev = after;
                before.next = dh;
                dh.prev = before;
            }
            before = nextNode;
        }
}
