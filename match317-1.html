        public static void sumOddAppend(Node dh) {
            if (dh == null || dh.next == dh) {
                return;
            }
            Node before = dh;
            Node cur = dh.next;
            int sum = 0;
 
            while (cur != dh) {
                int value = (Integer) cur.elem;
                if (value % 2 != 0) {
                    sum += value;


                    before.next = cur.next;
                } else {
                    before = cur;
                }
                cur = cur.next;
            }
            Node Newnode = new Node(sum);
            Newnode.next = dh;
            before.next = Newnode;
        }

        public static void pairJoin(DNode dh1, DNode dh2) {


            if (dh1 == null || dh2 == null || dh1.next == null || dh2.next == null) {
                return;
            }
            DNode cur1 = dh1.next;
            DNode cur2 = dh2.next;
            DNode end = dh1;
            while (cur1 != null && cur2 != null) {
                DNode nxt1 = cur1.next;
                DNode nxt2 = cur2.next;
                end.next = cur1;


                cur1.prev = end;
                cur1.next = cur2;
                cur2.prev = cur1;
                end = cur2;
                cur1 = nxt1;
                cur2 = nxt2;
            }
 
            end.next = dh1;
            dh1.prev = end;
        }

        public static void rangeMove(DNode dh, int start, int end) {
            if (dh == null || dh.next == dh) {
                return;
            }
            DNode cur = dh.next;
            DNode last = dh.prev;
            DNode stp = dh.prev;
            while (cur != stp.next) {
                DNode nxt = cur.next;
                int val = (Integer) cur.elem;
 
                if (val >= start && val <= end) {


                    cur.prev.next = cur.next;
                    cur.next.prev = cur.prev;
                    last.next = cur;
                    cur.prev = last;
                    cur.next = dh;
                    dh.prev = cur;
                    last = cur;
                }
                cur = nxt;
            }
        }
