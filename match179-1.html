public static String checkSimilar( Node building1, Node building2 ){

    Node t1 = building1;
    Node t2 = building2;

    while(t1 != null && t2 != null){
        if(!t1.elem.equals(t2.elem)){


            return "Not Similar";
        }
        t1 = t1.next;
        t2 = t2.next;
    }

    if(t1== null && t2 == null){
        return "Similar";
    }

    return "Not Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {

    int n = popularity.length;

    for(int i=0;i<n-1;i++){

        Node curr = head;


        Node prev = null;

        for(int j=0;j<n-i-1;j++){

            if(popularity[j] < popularity[j+1]){

                int temp = popularity[j];
                popularity[j] = popularity[j+1];
                popularity[j+1] = temp;

                Node nextNode = curr.next;

                if(prev == null){
                    curr.next = nextNode.next;
                    nextNode.next = curr;
                    head = nextNode;
                    prev = nextNode;
                }
                else{
                    prev.next = nextNode;
                    curr.next = nextNode.next;
                    nextNode.next = curr;
                    prev = nextNode;
                }
            }
            else{
                prev = curr;
                curr = curr.next;
            }
        }
            }

    return head;
    }

 public static Node alternateMerge(Node head1, Node head2){

    Node a1 = head1;
    Node a2 = head2;

    while(a1 != null && a2 != null){

        Node next1 = a1.next;
        Node next2 = a2.next;

        a1.next = a2;

        if(next1 == null){
            break;
        }



        a2.next = next1;

        a1 = next1;
        a2 = next2;
    }

    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {

    Node newHead = null;
    Node tail = null;

    Node temp = head1;

    while(temp!= null){
        Node newNode = new Node((Integer)temp.elem);



        if(newHead== null){
            newHead = newNode;
            tail = newNode;
        }
        else{
            newNode.next= newHead;
            newHead = newNode;
        }

        temp = temp.next;
    }

    temp = head2;
    Node temp3 = head3;

    while(temp!= null && temp3!= null){

        int sum = (Integer)temp.elem + (Integer)temp3.elem;

        if(sum>= 10){
            sum %= 10;
        }

        tail.next = new Node(sum);


        tail = tail.next;

        temp = temp.next;
        temp3 = temp3.next;
    }

    return newHead;
}

public static void sumOddAppend(Node dh) {
       Node prev = dh;
Node curr = dh.next;
int sum = 0;

while (curr!= dh) {
    int val = (Integer) curr.elem;

        sum += val;
        prev.next = curr.next;
        curr = prev.next;
    } else{
        prev = curr;


        curr = curr.next;
    }
}

Node newNode = new Node(sum);
newNode.next = dh;
prev.next = newNode;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
      DNode a1 = dh1.next;
DNode a2 = dh2.next;

while (a1 != null && a2 != null) {
    DNode n1 = a1.next;
    DNode n2 = a2.next;

    a1.next = a2;
    a2.prev = a1;

    if (n1!= null) {
        a2.next = n1;
        n1.prev = a2;
    } else {
        a2.next = dh1;
        dh1.prev = a2;
        
    }

    a1 = n1;
    a2 = n2;
}
    }

public static void rangeMove(DNode dh, int start, int end) {
   DNode last = dh.prev;
DNode curr = dh.next;

while (curr!= last.next) {
    DNode next = curr.next;
    int val = (Integer)curr.elem;

    if (val >= start && val<= end) {
        curr.prev.next = curr.next;
        curr.next.prev = curr.prev;

        dh.prev.next = curr;
        curr.prev = dh.prev;
        curr.next = dh;
        dh.prev = curr;
    }

    curr = next;
}
