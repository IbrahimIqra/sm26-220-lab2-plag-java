public static String checkSimilar( Node building1, Node building2 ){

    Node t1 = building1;
    Node t2 = building2;

    while (t1 != null && t2 != null) {
        if (!t1.elem.equals(t2.elem)) {
            return "Not Similar";
        }
        t1 = t1.next;
        t2 = t2.next;
    }
    if (t1 != null || t2 != null) {
        return "Not Similar";
    }
    return "Similar";
}

public static Node organizeBooks(Node head, Integer[] popularity) {

    int n = popularity.length;
    boolean swapped = true;

    while (swapped) {

        swapped = false;
        Node cur = head;
        int i = 0;

        while (cur != null && cur.next != null && i < n - 1) {



            if (popularity[i] < popularity[i + 1]) {

                Object tempElem = cur.elem;
                cur.elem = cur.next.elem;
                cur.next.elem = tempElem;

                int tempPop = popularity[i];
                popularity[i] = popularity[i + 1];
                popularity[i + 1] = tempPop;

                swapped = true;
            }
            cur = cur.next;
            i++;
        }
    }

    return head;
}

public static Node alternateMerge( Node head1, Node head2 ){

    Node p1 = head1;
    Node p2 = head2;

    while (p1 != null && p2 != null) {

        Node p1Next = p1.next;
        Node p2Next = p2.next;

        p1.next = p2;
        if (p1Next != null && p2Next != null) {
            p2.next = p1Next;
        }

        p1 = p1Next;
        p2 = p2Next;
    }

    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {

    Node resultHead = null;
    Node resultTail = null;

    
    Node cur = head1;
    while (cur != null) {

        Node newNode = new Node(cur.elem);
        newNode.next = resultHead;
        resultHead = newNode;
        cur = cur.next;
    }

    if (resultHead != null) {

        resultTail = resultHead;
        while (resultTail.next != null) {
            resultTail = resultTail.next;
        }
    }

    Node p2 = head2;
    Node p3 = head3;
    while (p2 != null && p3 != null) {

        int sum = (Integer) p2.elem + (Integer) p3.elem;
        if (sum >= 10) {
            sum = sum % 10;
        }

        Node newNode = new Node(sum);
        if (resultHead == null) {
            resultHead = newNode;
            resultTail = newNode;
        } else {
            resultTail.next = newNode;
            resultTail = newNode;
        }

        p2 = p2.next;
        p3 = p3.next;
    }

    return resultHead;
}

public static void sumOddAppend(Node dh) {

    int sum = 0;
    Node prev = dh;
    Node cur = dh.next;

    while (cur != dh) {

        int val = (Integer) cur.elem;
        if (val % 2 != 0) {
            sum += val;
            prev.next = cur.next;
            cur = cur.next;
        } else {
            prev = cur;
            cur = cur.next;
        }
    }

    Node newNode = new Node(sum);
    prev.next = newNode;
    newNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {

    DNode p1 = dh1.next;
    DNode p2 = dh2.next;

    while (p1 != null && p2 != null) {



        DNode p1Next = p1.next;
        DNode p2Next = p2.next;

        p1.next = p2;
        p2.prev = p1;

        if (p1Next != null) {
            p2.next = p1Next;
            p1Next.prev = p2;
        }

        p1 = p1Next;
        p2 = p2Next;
    }

    DNode tail = dh1;
    while (tail.next != null) {

        tail = tail.next;
    }
    tail.next = dh1;
    dh1.prev = tail;
}

public static void rangeMove(DNode dh, int start, int end) {

    DNode cur = dh.next;
    DNode movedHead = null;
    DNode movedTail = null;

    while (cur != dh) {

        DNode next = cur.next;
        int val = (Integer) cur.elem;

        if (val >= start && val <= end) {

            cur.prev.next = cur.next;
            cur.next.prev = cur.prev;

            if (movedHead == null) {
                movedHead = cur;
                movedTail = cur;
            } 
            else {
                movedTail.next = cur;
                cur.prev = movedTail;
                movedTail = cur;
            }
        }

        cur = next;
    }

    if (movedHead != null) {

        DNode tail = dh.prev;
        tail.next = movedHead;
        movedHead.prev = tail;
        movedTail.next = dh;
        dh.prev = movedTail;
    }
}
