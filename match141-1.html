public static String checkSimilar( Node building1, Node building2 ){
       
        Node temp1=building1;
        Node temp2=building2;


        while(temp1!=null && temp2!=null){
            if(!temp1.elem.equals(temp2.elem)){
                return "Not Similar";
            }
            temp1=temp1.next;
            temp2=temp2.next;
        }


        if(temp1==null && temp2==null) return "Similar";      
       
        return "Not Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {
       
        boolean swapped=true;


        while(swapped){
            swapped=false;
            int i=0;
            Node currNode=head;




            while(currNode!=null && currNode.next!=null){
                if(popularity[i]<popularity[i+1]){
                    Object tempElem=currNode.elem;
                    currNode.elem=currNode.next.elem;
                    currNode.next.elem=tempElem;


                    int tempPop=popularity[i];
                    popularity[i]=popularity[i+1];
                    popularity[i+1]=tempPop;


                    swapped=true;
                }
                currNode=currNode.next;
                i++;
            }
        }


        return head; // Remove this when you're ready to return the new head
    }

public static Node alternateMerge( Node head1, Node head2 ){
       
        if(head1==null) return head2;
        if(head2==null) return head1;


        Node curr1=head1;
        Node curr2=head2;


        while(curr1!=null && curr2!=null){
            Node temp1=curr1.next;


            Node temp2=curr2.next;


            curr1.next=curr2;
            if(temp1==null) break;
            curr2.next=temp1;


            curr1=temp1;
            curr2=temp2;
        }


        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
       
        Node result=null;
        Node curr1=head1;


        while(curr1!=null){
            Node newNode=new Node(curr1.elem);
            newNode.next=result;
            result=newNode;
            curr1=curr1.next;
        }




        Node tail=result;
        while(tail.next!=null){
            tail=tail.next;
        }


        Node curr2=head2;
        Node curr3=head3;
        while(curr2!=null && curr3!=null){
            Integer sum=(Integer)curr2.elem+(Integer)curr3.elem;


            sum=sum%10;


            Node newNode=new Node(sum);
            tail.next=newNode;
            tail=tail.next;


            curr2=curr2.next;
            curr3=curr3.next;
        }


        return result;
    }

public static void sumOddAppend(Node dh) {
        Node pred=dh;
        Node curr=dh.next;
        int sum=0;


        while(curr!=dh){
            int value=(Integer)curr.elem;
            if(value%2!=0){
                sum+=value;
                pred.next=curr.next;
            }else{
                pred=curr;
            }
            curr=curr.next;
        }


        Node newNode=new Node(sum);
        pred.next=newNode;
        newNode.next=dh;
       
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode curr1=dh1.next;
        DNode curr2=dh2.next;
        DNode tail=dh1;


        while(curr1!=null){
            DNode temp1=curr1.next;
            DNode temp2=curr2.next;


            curr1.next=curr2;
            curr2.prev=curr1;
            curr2.next=temp1;


            if(temp1!=null){
                temp1.prev=curr2;
            }else{
                tail=curr2;
            }


            curr1=temp1;
            curr2=temp2;
        }


        tail.next=dh1;
        dh1.prev=tail;


    }

public static void rangeMove(DNode dh, int start, int end) {
       
        DNode pred=dh;
        DNode curr=dh.next;
        DNode nHead=null;
        DNode nTail=null;
       
        while(curr!=dh){
            DNode temp=curr.next;
            int val=(Integer) curr.elem;


            if(val>=start && val<=end){
                pred.next=temp;
                temp.prev=pred;


                if(nHead==null){
                    nHead=curr;
                    nTail=curr;
                }else{
                    nTail.next=curr;
                    curr.prev=nTail;
                    nTail=curr;
                }


                curr=temp;
            }else{
                pred=curr;
                curr=temp;
            }
        }


        if(nHead!=null){
            pred.next=nHead;
            nHead.prev=pred;
            nTail.next=dh;
            dh.prev=nTail;
        }


    }
