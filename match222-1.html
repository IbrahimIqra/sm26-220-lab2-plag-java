    public static String checkSimilar( Node building1, Node building2 ){
            Node tempNode1 = building1;
            Node tempNode2 = building2;



            while ((tempNode1 != null) && (tempNode2 != null)) { 
                if (!tempNode1.elem.equals(tempNode2.elem)) {
                    return "Not Similar";
                }

                tempNode1 = tempNode1.next;
                tempNode2 = tempNode2.next;
            }

            if ((tempNode1 == null) && (tempNode2 == null)) {
                return "Similar";
            }

        return "Not Similar";
    }

    public static Node organizeBooks(Node head, Integer[] popularity) {
        if (head.next == null) {
            return head;
        }

        int length = popularity.length - 1;

        for (int i = 0; i < length; i++) {
            Node previousNode = null;
            Node currentNode = head;
            Node nextNode = head.next;



            for (int j = 0; j < length - i; j++) {
                if (popularity[j] < popularity[j + 1]) {
                    int temp = popularity[j];
                    popularity[j] = popularity[j + 1];
                    popularity[j + 1] = temp;

                    currentNode.next = nextNode.next;


                    nextNode.next = currentNode;

                    if (previousNode == null) {
                        head = nextNode;
                    }

                    else {
                        previousNode.next = nextNode;
                    }

                    previousNode = nextNode;
                    nextNode = currentNode.next;
                }
                
                else {
                    previousNode = currentNode;
                    currentNode = nextNode;
                    nextNode = nextNode.next;
                }
            }
        }

        return head;
    }

    public static Node alternateMerge( Node head1, Node head2 ){
        if (head1 == null) {
            return head2;
        }

        if (head2 == null) {
            return head1;
        }

        Node node1 = head1;
        Node node2 = head2;



        while ((node1 != null) && (node2 != null)) { 
            Node nextNode1 = node1.next;
            Node nextNode2 = node2.next;
            
            node1.next = node2;
            
            if (nextNode1 == null) {
                break;
            }

            node2.next = nextNode1;

            node1 = nextNode1;
            node2 = nextNode2;
        }

        return head1;
    }

    public static void sumOddAppend(Node dh) {
        Node previousNode = dh;
        Node currentNode = dh.next;

        int sum = 0;

        while (currentNode != dh) {
            int data = (int) currentNode.elem;

            if (data % 2 != 0) {
                sum += data;


                previousNode.next = currentNode.next;
                currentNode = currentNode.next;
            }

            else {
                previousNode = currentNode;
                currentNode = currentNode.next;
            }
        }

        Node newNode = new  Node(sum);
        Node tail = dh;



        while (tail.next != dh) {
            tail = tail.next;
        }

        tail.next = newNode;
        newNode.next = dh;
    }

    public static void pairJoin(DNode dh1, DNode dh2) {
        DNode node1 = dh1.next;
        DNode node2 = dh2.next;

        while ((node1 != dh1) && (node2 != dh2)) { 
            DNode nextNode1 = node1.next;
            DNode nextNode2 = node2.next;

            node1.next = node2;
            node2.prev = node1;

            if (nextNode1 == null) {
                node2.next = dh1;
                dh1.prev = node2;

                dh2.next = dh2;
                dh2.prev = dh2;


                break;
            }

            node2.next = nextNode1;
            nextNode1.prev = node2;

            node1 = nextNode1;
            node2 = nextNode2;
        }
    }

    public static void rangeMove(DNode dh, int start, int end) {
        DNode tail = dh.prev;
        DNode newtail = dh.prev;
        DNode currentNode = dh.next;
        
        while (true) {
            DNode nextNode = currentNode.next;
            int data = (int) currentNode.elem;

            if ((data >= start) && (data <= end)) {


                currentNode.prev.next = currentNode.next;
                currentNode.next.prev = currentNode.prev;

                newtail.next = currentNode;
                currentNode.prev = newtail;
                currentNode.next = dh;
                dh.prev = currentNode;
                newtail = currentNode;
            }

            if (currentNode == tail) {
                break;
            }

            currentNode = nextNode;
        }
    }




    public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node previousNode = null;
        Node currentNode = head1;

        while (currentNode != null) {
            Node nextNode = currentNode.next;
            currentNode.next = previousNode;
            previousNode = currentNode;
            currentNode = nextNode;
        }

        Node head = previousNode;



        Node tail = head;
        while (tail.next != null) {
            tail = tail.next;
        }
        
        Node node2 = head2;
        Node node3 = head3;

        while ((node2 != null) && (node3 != null)) {
            int sum = (int) node2.elem + (int) node3.elem;

            if (sum >= 10) {
                sum %= 10;
            }

            tail.next = new Node(sum);
            tail = tail.next;

            node2 = node2.next;
            node3 = node3.next;
        }

        return head;
    }
