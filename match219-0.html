//AssignmentTask1: Building Blocks



public static String checkSimilar( Node building1, Node building2 ) {
  
  Node pointer1=building1;
  Node pointer2=building2;
  
  while(pointer1!=null && pointer2!=null){ //duita ekshathe traverse
    
    
    if(!pointer1.elem.equals(pointer2.elem)){  //compare kora
      return "Not Similar";
    }
    
    
    pointer1=pointer1.next;
    pointer2=pointer2.next;
    
    
  }
  
  
  
  if (pointer1!=null || pointer2!=null){ //different length er ll thakle


    return "Not Similar";
  }
  
  
  return "Similar";
}


//AssignmentTask2: Organize Books

public static Node organizeBooks(Node head, Integer[] popularity) {
  
  if(head==null || head.next==null){
    return head;
  }
  
  int totBooks=popularity.length;
  
  for(int i=0;i<totBooks-1;i++){ //bubble sorting
    
    Node temp=head;
    
    for(int j=0;j<totBooks-1-i;j++){
      
      if(popularity[j]<popularity[j+1]){
        
        
        
        int popTemp=popularity[j]; //popularity swap
        popularity[j]=popularity[j+1];
        popularity[j+1]=popTemp;
        
        
        String nameTemp=(String)temp.elem; //boi er naam swap
        temp.elem=temp.next.elem;


        temp.next.elem=nameTemp;
        
        
      }
      
      
      temp = temp.next;
    }
  }
  
  return head;
}



//AssignmentTask3: Alternate Merge

public static Node alternateMerge( Node head1, Node head2 ){ 
  
  Node temp1=head1;
  Node temp2=head2;
  
  while(temp1!=null && temp2!=null){  //duita ekshathe traverse
    
    Node next1=temp1.next;  //node save na korle harai jabe
    Node next2=temp2.next;
    
    temp1.next=temp2;  //1 er por 2
    
    if(next1==null){ //stop korar jonno
      break;
    }
    


    temp2.next=next1; //2 er por abr 1
    
    temp1=next1;
    temp2=next2;
  }
  
  return head1;
}

//AssignmentTask4: ID Generator


public static Node idGenerator(Node head1, Node head2, Node head3) {
  
  Node result=reverse(head1);
  result=addingLL(result, head2, head3); //ulta 1 er shathe notun ta link korbe
  
  return result;
}

public static Node reverse(Node head){ //1 nmbr reverse
  Node newHead=null;
  Node temp=head;
  
  while(temp!=null){
    
    Node newNode=new Node(temp.elem);
    newNode.next=newHead;
    newHead=newNode;
    
    temp=temp.next;
  }
  
  return newHead;
}

public static Node addingLL(Node head,Node head2,Node head3){ //2 ar 3 ekshathe traverse kore add


  Node tail=head;
  
  while(tail.next!=null){
    tail=tail.next;
  }
  
  
  Node temp2=head2;
  Node temp3=head3;
  
  while(temp2!=null && temp3!=null){
    
    int sum=((int)temp2.elem+(int)temp3.elem)%10; 


    
    tail.next=new Node(sum);
    tail=tail.next;
    
    temp2=temp2.next;
    temp3=temp3.next;
  }
  
  return head;
}

public static void sumOddAppend(Node dh){
  
  
  Node pre=dh;
  Node cur=dh.next;
  
  
  int sum=0;
  
  while(cur!=dh){
    
    int val=(int)cur.elem;
    
    if(val%2!=0){
      sum+=val;


      pre.next=cur.next; //node remove
      cur=pre.next;
    }
    
    
    else{
      pre=cur; //skip 
      cur=cur.next;
    }
    
  }
  
  Node newNode=new Node(sum);
  
  Node temp=dh;

  while(temp.next!=dh){


    temp=temp.next;
  }
  
  temp.next=newNode; //last node e add
  newNode.next=dh; //circular korar jonno
}

public static void pairJoin(DNode dh1, DNode dh2) {
  
  DNode temp1=dh1.next; //node save
  DNode temp2=dh2.next;
  
  
  
  while(temp1!=null && temp2!=null){
    
    DNode next1=temp1.next;  //1 er por 2
    DNode next2=temp2.next;
    
    
    
    temp1.next=temp2;
    temp2.prev=temp1;
    
    
    
    if(next1==null){ //stopping
      break;
    }
    
    
    
    temp2.next=next1; //2 er por 1


    next1.prev=temp2;

    temp1=next1;
    temp2=next2;
    
    
  }
  
  DNode tail=dh1; //circular korar jonno
  
  
  while(tail.next!=null){
    tail=tail.next;
  }
  
  
  tail.next=dh1;
  dh1.prev=tail;
  
}

public static void rangeMove(DNode dh, int start, int end) {
  
  
  if(dh.next==dh){ //empty hole return
    return;
  }
  
  
  DNode tail=dh.prev; //last node ber kora
  
  int count=0;
  
  DNode temp=dh.next;
  
  while(temp!=dh){ //node count
    count++;
    temp=temp.next;
  }
  
  temp=dh.next;
  
  while(count>0){
    
    DNode nextNode=temp.next; //next node save korar jonno
    
    int val=(int)temp.elem;
    
    if(val>=start && val<=end){
      
      if(temp!=tail){    //already last node na hole
        
        
        temp.prev.next = temp.next;  //current node remove korar jonno
        temp.next.prev = temp.prev;
        
        
        
        temp.next=dh;   //last e add korar jonno
        temp.prev=tail;
        tail.next=temp;
        dh.prev=temp;
        
        
        
        tail=temp; //notun tail update korar jonno
      }
    }
    
    
    temp=nextNode;
    count--; //loop control
    
    
  }
}
