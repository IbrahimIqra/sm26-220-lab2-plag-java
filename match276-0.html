     public static String checkSimilar(Node building1, Node building2) {


    Node CN1 = building1 ;
    Node CN2 = building2 ;


    while( CN1!= null && CN2 != null ){


        if( !CN1.elem.equals(CN2.elem) ){
          return "Not Similar" ;
        }


       CN1 = CN1.next ;
       CN2 = CN2.next ;
    }


    if( CN1 != null || CN2 != null ){
       return "Not Similar" ;
    }


    return "Similar";
}

    DNode tail2 =  dh2 ;  // tail of 2nd LL
       
while( curr1 != null && curr2 != null ){

    DNode temp2 = curr2.next ;  // saving the next one

    curr1.prev = tail1 ;  // connecting with the 1st LL


    tail1 = curr1 ;  // updating the tail of 1st LL
    tail1.next = curr2 ;  // connecting with the 2nd LL
    curr2.prev = tail1 ;  // connecting with the 2nd LL
    tail1 = curr2 ;  // updating the tail of 1st LL

    curr2 = temp2 ;  // moving to the next node of 2nd LL


    }


  tail1.next = tail2 ;  // connecting with the tail of 2nd LL  
  tail2.prev = tail1 ;  // connecting with the tail of 2nd LL




}
