public static String checkSimilar( Node building1, Node building2 ){


        Node a1 = building1;
        Node a2 = building2;




        while (a1 != null && a2 != null){
            if(a1.elem!=a2.elem){
                return "Not Similar";
            }


            a1 = a1.next;
            a2 = a2.next;
        }
        if(a1 == null && a2 == null){
            return "Similar";
        } 
        else {
            return "Not Similar";
        }


    }

public static Node organizeBooks(Node head, Integer[] popularity){
        
        int n = popularity.length;




        for(int i=0; i<n-1; i++){
            Node current = head;
            Node nextNode = current.next;


            for(int j=0; j<n-i-1; j++){
                if(popularity[j] < popularity[j+1]){
                    int currentPopular = popularity[j];
                    popularity[j] = popularity[j+1];
                    popularity[j+1] = currentPopular;


                    Object temp = current.elem;
                    current.elem = nextNode.elem;
                    nextNode.elem = temp;
                }
                current = current.next;
                nextNode = current.next;
            }
        }
        return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){
        if (head1 == null) 
            return head2;


        if (head2 == null) 
            return head1;
    


        Node a1 = head1;
        Node a2 = head2;


        while(a1!=null && a2!=null){
            Node front1 = a1.next;
            a1.next = a2;
            Node front2 = a2.next;
            a2.next = front1;
            a1 = front1;
            a2 = front2;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3){

    Node prev = null;


    while (curr != null) {
        Node front = curr.next;
        curr.next = prev;

        curr = front;

        temp2 = temp2.next;

    }

public static void sumOddAppend(Node dh){
    int sum = 0;
    Node a1 = dh;
    Node current = dh.next;


    while(current != dh){
        if((int) current.elem % 2 != 0){
            sum += (int) current.elem;


            a1.next = current.next;
            current = current.next;
        }
        else{
            a1 = current;
            current = current.next;
        }
    }
    Node sumnode = new Node(sum);
    a1.next = sumnode;
    sumnode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2){


    DNode current1 = dh1.next;
    DNode current2 = dh2.next;


    DNode last = dh1; 


    while(current1 != null){
        DNode next1 = current1.next;
        DNode next2 = current2.next;


        current1.next = current2;
        current2.prev = current1;
        current2.next = next1;
        
        if(next1 != null){
            next1.prev = current2;
        }
        last = current2;


        current1 = next1;
        current2 = next2;
    }


    last.next = dh1;
    dh1.prev = last;
}

public static void rangeMove(DNode dh, int start, int end){


    int size = 0;
    DNode counter = dh.next;
    while(counter != dh){
        size++;
        counter = counter.next;
    }


    DNode current = dh.next;


    for(int i = 0; i < size; i++){
        DNode a1 = current.next;




        if((int) current.elem >= start && (int) current.elem <= end){
            current.prev.next = current.next;
            current.next.prev = current.prev;


            DNode tail = dh.prev;
            tail.next = current;
            current.prev = tail;
            current.next = dh;
            dh.prev = current;
        }


        current = a1;
    }
}
