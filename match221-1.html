public static String checkSimilar(Node building1, Node building2){


                           Node temp1 = building1;
    Node temp2 = building2;


    while(temp1 !     = null &&        temp2 !         = null){




                 if(!temp1.elem.equals(temp2.elem)){
            return "Not Similar";
        }


        temp1 = temp1.next;
                         temp2 = temp2.next;
    }


    if(temp1 == null && temp2 == null){
                 return "Similar";
    }


              return "Not Similar";
}

            if(popularity[j] < popularity[j + 1]){


                Integer tempPopularity = popularity[j];

                      popularity[j + 1] = tempPopularity;

                                      current.elem = next.elem;

            }




            current = current.next;
                             next = next.next;
        }


    }


    return head;
                                  }

public static Node alternateMerge(Node head1, Node head2){


    Node temp1 = head1;
                      Node temp2 = head2;


    while(temp1 != null && temp2 != null){


                           Node next1 = temp1.next;
        Node next2 = temp2.next;


        temp1.next = temp2;
                         temp2.next = next1;


        temp1 = next1;
        temp2 = next2;
    }


    return head1;


}

public static Node reverseCopy(Node head){


    if(head.next == null){
                                return new Node((Integer)head.elem);
    }


    Node newHead = reverseCopy(head.next);


    Node temp = newHead;
                                         while(temp.next != null){
        temp = temp.next;
   
    }


                            temp.next = new Node((Integer)head.elem);


    return newHead;
  
}

public static void sumOddAppend(Node dh) {


           int sum = 0;


    Node prev = dh;
                               Node curr = dh.next;




    while(curr != dh){


        if((Integer)curr.elem % 2 != 0){
                                 sum += (Integer)curr.elem;
                                                 prev.next = curr.next;
                                                                curr = prev.next;
        }


        else{
            prev = curr;
                               curr = curr.next;
        }


    }


    Node newNode = new Node(sum);


    Node tail = dh;
                              while(tail.next != dh){
        tail = tail.next;


    }




    tail.next = newNode;
                             newNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {


     DNode p1 = dh1.next;
                         DNode p2 = dh2.next;
                                                  DNode last = dh1;


    while (p1 != null && p2 != null) {


        DNode next1 = p1.next;
                                             DNode next2 = p2.next;


        last.next = p1;
                         p1.prev = last;


        p1.next = p2;
                 p2.prev = p1;


        last = p2;


        p1 = next1;
                                          p2 = next2;


    }




    last.next = dh1;
                              dh1.prev = last;


}

    DNode tail = dh.prev;
                    DNode curr = dh.next;


    while (curr != dh) {


                        DNode next = curr.next;


        int value = (Integer) curr.elem;


        if (value >= start && value <= end) {


                                              if (curr == tail) {
                tail = tail.prev;


            }


            


            curr.prev.next = curr.next;
                                          curr.next.prev = curr.prev;


            
            curr.prev = tail;
                            curr.next = dh;
            tail.next = curr;

            tail = curr;


        }


        curr = next;
    }
