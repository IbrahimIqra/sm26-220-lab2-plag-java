public static String checkSimilar(Node building1, Node building2) {
    Node c1 = building1;
    Node c2 = building2;
    while (c1 != null && c2 != null) {    // Comparing the elements


              if (!c1.elem.equals(c2.elem)) {
            return "Not Similar";
        }
        c1 = c1.next;     // Moving to the next nodes
        c2 = c2.next;
    }
    if (c1 == null && c2 == null) {     
        return "Similar";
    } 
    else {
                return "Not Similar";
    }
}

public static Node organizeBooks(Node head, Integer[] popularity) {
    if (head == null || head.next == null) {
        return head;
    }
    int n = 0;
    Node temp = head;
    while (temp != null) {
        n++;


        temp = temp.next;
    }
    for (int i = 0; i < n - 1; i++) {      //Buble sort kortesi
        Node curr = head;
        for (int j = 0; j < n - 1 - i; j++) {
            if (popularity[j] < popularity[j + 1]) {
                int tempPop = popularity[j];
                popularity[j] = popularity[j + 1];
                popularity[j + 1] = tempPop;
                Object tempElem = curr.elem;
                curr.elem = curr.next.elem;
                curr.next.elem = tempElem;
            }
            curr = curr.next;
        }
    }
    return head;
}

public static Node alternateMerge(Node head1, Node head2) {
    if(head1 == null){
        return head2;
}
    if(head2 == null){
        return head1;
}
    Node current1 = head1;
    Node current2 = head2;
    while(current1 != null && current2 != null){
        Node next1 = current1.next;
        Node next2 = current2.next;
        current1.next = current2;
        if(next1 != null){
            current2.next = next1;
        }
        current1 = next1;
        current2 = next2;
    }
    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node dHead = new Node(null);
    Node tail = dHead;
    Node temp = head1;
    while(temp != null){
        Node newNode = new Node(temp.elem);
        newNode.next = dHead.next;


        dHead.next = newNode;
        temp = temp.next;
    }
    tail = dHead;
    while(tail.next != null){
        tail = tail.next;
    }
    Node t2 = head2;
    Node t3 = head3;
    while(t2 != null && t3 != null){
        int sum = (Integer)t2.elem + (Integer)t3.elem;
        if(sum >= 10){
            sum = sum % 10;
        }
        tail.next = new Node(sum);
        tail = tail.next;
        t2 = t2.next;
        t3 = t3.next;
    }
    return dHead.next;
}

public static void sumOddAppend(Node dh) {
    Node prev = dh;
    Node current = dh.next;
    int sum = 0;
    while (current != dh) {
        int value = (Integer) current.elem;
        if (value % 2 != 0) {
            sum += value;


            prev.next = current.next;
            current = prev.next;
        }
       else {
            prev = current;
            current = current.next;
        }
    }
    Node newNode = new Node(sum, dh);
    prev.next = newNode;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode curr1 = dh1.next;
    DNode curr2 = dh2.next;
    while (curr1 != dh1 && curr2 != dh2) {
        DNode next1 = curr1.next;
        DNode next2 = curr2.next;
        curr1.next = curr2;
        curr2.prev = curr1;
        curr2.next = next1;
        next1.prev = curr2;
        curr1 = next1;
        curr2 = next2;
    }
    dh1.prev = dh2.prev;
    dh2.prev.next = dh1;
}

public static void rangeMove(DNode dh, int start, int end) {
    DNode tail = dh.prev;
    DNode current = dh.next;
    while (current != dh) {
        DNode next = current.next;
        int val = (Integer) current.elem;
        if (val >= start && val <= end && current != tail) {
            current.prev.next = current.next;
            current.next.prev = current.prev;
            current.prev = tail;
            current.next = dh;
            tail.next = current;
            dh.prev = current;
            tail = current;
        }
        current = next;
    }
}
