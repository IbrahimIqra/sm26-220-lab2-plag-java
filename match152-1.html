public static String checkSimilar( Node building1, Node building2 ){
        
        Node myTower= building1;
        Node twinTower= building2;

        while(myTower!=null && twinTower!=null){
            if(!myTower.elem.equals(twinTower.elem)){
                return "Not Similat";
            }
            myTower=myTower.next;
            twinTower=twinTower.next;
        }
        if(myTower==null && twinTower==null){
            return "Similar";
        }
        return "Not Similar";
    }

        for(int i=0; i<n-1;i++){

            for (int j=0; j<n-1-i;j++){
                if(popularity[j]<popularity[j+1]){
                    Object book=curr.elem;
                    curr.elem=curr.next.elem;
                    curr.next.elem=book;

                    int temp=popularity[j];
                    popularity[j]=popularity[j+1];

                }
                curr=curr.next;
            }
        }

        return head; 
    }

 public static void sumOddAppend(Node dh) {
        Node prev=dh;
        Node curr= dh.next;
        int sum=0;
        while(curr !=dh){
            if((int)curr.elem % 2!=0){
                sum+=(int)curr.elem;
                prev.next=curr.next;
                curr=curr.next;
            }
            else{
                prev=curr;
                curr=curr.next;
            }
        }
        prev=dh;
        while(prev.next!=dh){
            prev=prev.next;
        }
        Node newNode= new Node (sum);
        newNode.next=dh;
        prev.next=newNode;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode curr1=dh1.next;
        DNode curr2=dh2.next;
        while(curr1 !=null && curr2!= null){
            DNode next1=curr1.next;
            DNode next2=curr2.next;


            curr1.next=curr2;
            curr2.prev=curr1;
            curr2.next=next1;
            if(next1 !=null){
                next1.prev=curr2;
            }
            curr1=next1;
            curr2=next2;
        }
        DNode tail=dh1;
        while(tail.next !=null){
            tail=tail.next;
        }
        tail.next=dh1;
        dh1.prev=tail;
    }
