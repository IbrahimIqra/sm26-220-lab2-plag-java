public static String checkSimilar( Node building1, Node building2 ){

    //You're not suppose to create any new Linked List for this task
    Node n1 =building1;
    Node n2= building2;
    while(n1 != null && n2 !=null)
    {
        if(!n1.elem.equals(n2.elem))
        {
            return "Not Similar";
        }
        n1 =n1.next;
        n2 = n2.next;
    }
    if(n1== null && n2 ==null)
    {


        return "Similar";
    }
    return "Not Similar";
}


// Lab 2 - Question 2
// Organize Books
public static Node organizeBooks(Node head, Integer[] popularity) {

    int n =0;
    Node t =head;
    while(t != null)
    {
        n= n +1;
        t =t.next;
    }

    for(int i=0;i <n -1; i++)
    {
        Node node =head;
        for(int j=0;j <n -1 -i;j++)
        {
            boolean swap =false;
            if(popularity[j] <popularity[j +1])
            {
                swap =true;
            }
            if(swap)
            {
                Object temp =node.elem;
                node.elem= node.next.elem;
                node.next.elem =temp;



                int p =popularity[j];
                popularity[j] =popularity[j +1];
                popularity[j +1]= p;
            }
            node =node.next;
        }
    }

    return head;
}


// Lab 2 - Question 3
// Alternate Merge
public static Node alternateMerge( Node head1, Node head2 ){

    //You're NOT ALLOWED to create a new singly linked list for this task
    if(head1 == null)
    {
        return head2;
    }
    Node n1 =head1;
    Node n2= head2;
    while(n1 != null && n2 !=null)
    {


        Node next1 =n1.next;
        Node next2= n2.next;
        n1.next= n2;
        n2.next =next1;
        n1 =next1;
        n2= next2;
    }
    if(n2 != null)
    {
        Node tail =head1;
        while(tail.next != null)
        {
            tail =tail.next;
        }


        tail.next= n2;
    }

    return head1;
}


// Lab 2 - Question 4
// ID Generator
public static Node idGenerator(Node head1, Node head2, Node head3) {

    //Hint: the Node elements are actually Object, you can type cast them
    //      into int or Integer like the following:
    //        (int)n.elem  or  (Integer)n.elem
    Node newHead =null;
    Node n =head1;
    while(n != null)
    {
        Node newN =new Node(n.elem);
        newN.next= newHead;
        newHead= newN;


        n =n.next;
    }

    Node tail =newHead;
    while(tail.next != null)
    {
        tail= tail.next;
    }

    Node a =head2;
    Node b= head3;
    while(a != null && b !=null)
    {
        int sum =(int)a.elem + (int)b.elem;
        Node newN =new Node(sum %10);
        tail.next =newN;
        tail= newN;


        a= a.next;
        b =b.next;
    }

    return newHead;
}


// Lab 2 - Question 5
// Sum Odd Append
public static void sumOddAppend(Node dh) {
    int sum =0;
    Node prev =dh;
    Node curr= dh.next;
    while(curr != dh)
    {
        if((int)curr.elem %2 !=0)
        {
            sum =sum + (int)curr.elem;


            prev.next =curr.next;
            curr =curr.next;
        }
        else
        {
            prev =curr;
            curr= curr.next;
        }
    }
    Node newN =new Node(sum);
    newN.next =dh;
    prev.next= newN;
}


// Lab 2 - Question 6
// Pair Join
public static void pairJoin(DNode dh1, DNode dh2) {
    DNode tail =dh1;
    DNode c1 =dh1.next;
    DNode c2= dh2.next;
    while(c1 != null || c2 !=null)
    {
        if(c1 != null)
        {
            DNode nxt =c1.next;
            tail.next =c1;
            c1.prev =tail;
            tail =c1;
            c1= nxt;
        }
        if(c2 != null)
        {
            DNode nxt =c2.next;
            tail.next= c2;


            c2.prev= tail;
            tail =c2;
            c2= nxt;
        }
    }
    tail.next =dh1;
    dh1.prev= tail;
}


// Lab 2 - Question 7
// Range Move
public static void rangeMove(DNode dh, int start, int end) {
    int n =0;
    DNode t =dh.next;
    while(t != dh)
    {
        n= n +1;


        t= t.next;
    }

    DNode curr =dh.next;
    for(int i =0;i <n; i++)
    {
        DNode next =curr.next;
        int val =(int)curr.elem;
        if(val >= start && val <= end)
        {
            curr.prev.next =curr.next;
            curr.next.prev =curr.prev;
            DNode tail =dh.prev;
            tail.next= curr;
            curr.prev =tail;
            curr.next =dh;
            dh.prev =curr;
        }
        curr =next;
    }
}
