public static Integer[][] compressMatrix( Integer[][] matrix ){

        int row = matrix.length/2;


        int col = matrix[0].length/2;
        
        Integer[][] compressMatrix=new Integer[row][col];
        for(int i=0;i<row;i++){
            for(int j=0;j<col;j++){
                int r=i*2;
                int c=j*2;
                compressMatrix[i][j]=matrix[r][c]+matrix[r][c+1]+matrix[r+1][c]+matrix[r+1][c+1];
            }
        }


        return compressMatrix;
    }

public static void rotateSecret( Character[][] board ){


        int n = board.length;

        int numLayer = n/ 2;

        
        for (int L= 0; L <numLayer; L++) {
            
            int rot= numLayer - L;

            int last= n -1 -L;

            

            for (int r =0; r <rot; r++) {
                


                char temp= board[L][L];

                
                for (int i= L; i <last; i++) {
                    board[i][L] =board[i + 1][L];
                }

                
                for (int j =L; j <last; j++) {
                    board[last][j] = board[last][j + 1];
                }

                
                for (int i =last; i> L; i--) {
                    board[i][last] =board[i - 1][last];
                }

                
                for (int j= last; j> L + 1; j--) {
                    board[L][j] = board[L][j - 1];
                }

               
                board[L][L + 1] =temp;
            }
        }

        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                System.out.print(board[i][j]);
            }
        }
        System.out.println();
    }

public static void sumOddAppend(Node dh) {
        
    if (dh == null || dh.next == dh) {
        return; 

    }

    Node pre = dh;


     Node cur = dh.next;

      int sum = 0;

    
    while (cur != dh) {

        if ((int) cur.element % 2 != 0) {
            
            sum += (int) cur.element;
            pre.next = cur.next;
            
        } else {
            
            pre = pre.next;

        }
        cur = cur.next;

        return;
    }

    DNode curr1 = dh1.next;

    DNode curr2 = dh2.next;

    DNode tail = dh1; 


    while (curr1 != null && curr2 != null) {
        DNode next1 = curr1.next;

         DNode next2 = curr2.next;

        
        tail.next = curr1;
            curr1.prev = tail;
        tail = curr1;

        
        tail.next = curr2
            curr2.prev = tail;
        tail = curr2;

        
        curr2 = next2;
    }

  
    tail.next = dh1;

    dh1.prev = tail;

    
}

  public static void rangeMove(DNode dh, int start, int end) {

    if (dh ==null || dh.next== dh) {
        
        return;
    }

    DNode rangeStartNode = null;

     DNode rangeEndNode =null;

    DNode curr= dh.next;

    
    while (curr != dh) {
        if ((int) curr.element == start) {
            rangeStartNode = curr;
        }
        if ((int) curr.element == end) {
            rangeEndNode = curr;
        }
        curr = curr.next;
    }

   
    if (rangeStartNode == null || rangeEndNode == null || rangeStartNode == rangeEndNode) {
        return;
    }

    
    DNode beforeRange = rangeStartNode.prev;
    DNode afterRange = rangeEndNode.next;

     beforeRange.next = afterRange;
    afterRange.prev = beforeRange;

  
    DNode tail = dh.prev;

    tail.next = rangeStartNode;
     rangeStartNode.prev = tail;
    
     rangeEndNode.next = dh;
    
    dh.prev = rangeEndNode;
}
