public static void pairJoin(DNode dh1, DNode dh2){
        DNode node1 = dh1.next;
        DNode node2 = dh2.next;
        DNode tail= dh1;

        while (node1 != null && node2 != null) {
            DNode node3 = node2;
            node2 = node2.next;
            node3.next = node1.next;
            node3.prev = node1;
            if (node1.next != null) {
                node1.next.prev = node3;
            }
            node1.next = node3;
            tail = node3;
            node1 = node3.next;
        }
        tail.next = dh1;
        dh1.prev = tail;
    }
