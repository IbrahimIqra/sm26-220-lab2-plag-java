   public static String checkSimilar( Node building1, Node building2 ){
            Node temp1 = building1;
            Node temp2 = building2;
            while(temp1 != null && temp2 != null){
                if( ! temp1.elem.equals(temp2.elem)){
                    return "Not Similar";
                }
                temp1 = temp1.next;
                temp2 = temp2.next;
            }
        if(temp1 != null || temp2 != null){
                return "Not Similar";
            }
            return "Similar";
        }

   public static Node organizeBooks(Node head, Integer[] popularity) {
            int len = popularity.length;
            for(int i = 0; i < len-1; i++){
                Node cur = head;
                for(int j = 0; j < len-i-1; j++){
                    if(popularity[j] < popularity[j+1]){
                        int temp = popularity[j];
                        popularity[j] = popularity[j+1];
                        popularity[j+1] = temp;


                        String tBook = (String)cur.elem;
                        cur.elem = cur.next.elem;
                        cur.next.elem = tBook;
                    }
                    cur = cur.next;
                }
            }
 
            return head;
        }

public static Node alternateMerge( Node head1, Node head2 ){

            return head2;

                return head1;
            }
            Node temp1 = head1;
            Node temp2 = head2;
            while(temp1 != null && temp2 != null){
                Node nx1 = temp1.next;
                Node nx2 = temp2.next;
                temp1.next = temp2;

                temp1 = nx1;

public static Node idGenerator(Node head1, Node head2, Node head3) {
            Node nH = null;
            Node t = null;
            Node temp = head1;
            while(temp != null){
                Node nn = new Node(temp.elem);


                nn.next = nH;
                nH = nn;
                if(t == null){
                    t = nn;
                }
                temp = temp.next;
            }
            Node t2 = head2;
            Node t3 = head3;
            while(t2 != null && t3 != null){
                int sum = (Integer)t2.elem + (Integer)t3.elem;
                if(sum >= 10){
                    sum = sum % 10;
                }
                Node nn1 = new Node(sum);
                t.next = nn1;
                t = nn1;
                t2 = t2.next;
                t3 = t3.next;
            }
            return nH;
        }

  public static void sumOddAppend(Node dh) {
        int sum = 0;
        Node prev = dh;
        Node cur = dh.next;
        while(cur != dh){
            if((int)cur.elem % 2 != 0){
                sum += (int)cur.elem;
                prev.next = cur.next;
                cur = prev.next;
            }
            else{
                prev = cur;
                cur = cur.next;
            }
        }
        Node nn = new Node(sum);
        Node temp = dh;
        while(temp.next != dh){
            temp = temp.next;
        }
        temp.next = nn;
        nn.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {


        DNode t1 = dh1.next;
        DNode t2 = dh2.next;
        while(t1 != dh1 && t2 != dh2){
            DNode nx1 = t1.next;
            DNode nx2 = t2.next;
            t2.prev.next = nx2;
            if(t2.prev != null){
                t2.prev.next = t2.next;
            }
            if(t2.next != null){
                t2.next.prev = t2.prev;
            }
            t1.next = t2;
            t2.prev = t1;  
            t2.next = nx1;
            if(nx1 != null){
                nx1.prev = t2;
            }
                t1 = nx1;
                t2 = nx2;
            DNode last = dh1;
            while(last.next != null){
                last = last.next;
            }          
            last.next = dh1;
            dh1.prev = last;
        }
    }

   public static void rangeMove(DNode dh, int start, int end) {
        DNode last = dh.prev;
        DNode cur = dh.next;
        DNode stop = last;
        while(cur != stop.next){


            DNode nx = cur.next;
            int val = (int)cur.elem;
            if(val >= start && val <= end){
                cur.prev.next = cur.next;
                cur.next.prev = cur.prev;
                last.next = cur;
                cur.prev = last;
                cur.next = dh;
                dh.prev = cur;
                last = cur; 
            }
            cur = nx;
        }
    }
