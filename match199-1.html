public class AssignmentTask2{


    public static Node organizeBooks(Node head, Integer[] popularity) {
        //using bubble sort
        int n = popularity.length;
        for(int i =0; i<n-1; i++){
            Node current = head;
            for(int j=0; j<n-i-1; j++){
                if(popularity[j]<popularity[j+1]){
                    int temp = popularity[j]; // rating order
                    popularity[j]= popularity[j+1];
                    popularity[j+1]= temp;


                    String bookNow = (String) current.elem; // book order
                    current.elem = current.next.elem;
                    current.next.elem= bookNow;
                }
                 current = current.next;
            }
        }
        return head;
    }

    public static Node alternateMerge( Node head1, Node head2 ){
       
        Node current1= head1;
        Node current2= head2;


         if(head1==null){ // if one of head pointers are null
            return head2;
        }
        if(head2==null){
            return head1;
        }


        while(current1 != null && current2 != null){


            Node temp1 = current1.next;
            Node temp2= current2.next;


            current1.next= current2;
            current2.next= temp1;
            current1 = temp1;
            current2 = temp2;


        }
        return head1;
    }

public class AssignmentTask4{




    public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node current1 = head1;
        Node prev= null;


        while(current1!=null){//reverse first linked list
            Node next= current1.next;
            current1.next= prev;
            prev =current1;
            current1= next;
        }
        head1=prev;


        Node tail = head1;// end of reversed list
        while(tail.next!=null){
            tail= tail.next;
        }
        Node current2= head2;
        Node current3= head3;


        while(current2!=null && current3!=null){// adding each elem of list 2+3
            int sum= ((int)current2.elem + (int)current3.elem)%10;
            tail.next= new Node(sum);
            tail=tail.next;


            current2= current2.next;
            current3= current3.next;
        }
        return head1;
    }

public class AssignmentTask5 {


    public static void sumOddAppend(Node dh) {
        Node current= dh.next;
        Node prev =dh;


        int sum =0;
       
        while(current!=dh){
            if((Integer)current.elem % 2!=0){
                sum +=  (Integer) current.elem;


                prev.next= current.next;// delete odd num
                current = current.next;
            }
            else{
                prev=current; //traverse if num = even
                current= current.next;
            }
        }
        Node newNode = new Node(sum);
        prev.next= newNode;
        newNode.next= dh; // circular so connects back to the dh
    }

public class AssignmentTask6 {




    public static void pairJoin(DNode dh1, DNode dh2) {
        DNode c1= dh1.next;
        DNode c2= dh2.next;


        while(c1!= null && c2!=null){
            DNode next1= c1.next;
            DNode next2= c2.next;


            c1.next= c2;
            c2.prev= c1; //inserting node from 2nd list
               
            c2.next= next1;


            if(next1!=null){
                next1.prev= c2;
            }
       
            c1= next1;
            c2= next2;
        }
        DNode last = dh1.next; //connecting last node to dh1
        while(last.next!=null){
            last= last.next;
        }
        last.next=dh1;
        dh1.prev=last;
        dh1.next.prev= dh1;//connecting dh1 to first node again


        dh2.next = null;//making 2nd list empty after converting to 1st list
        dh2.prev = null;
    }

public class AssignmentTask7 {


    public static void rangeMove(DNode dh, int start, int end) {
        DNode c= dh.next;
        int count = 0;//to find the end of list
       
        while(c!=dh){// reaching end of list
            count++;
            c = c.next;
        }
     
         DNode temp= dh.next;
         for(int i =0; i<count; i++){
            DNode newNode= temp.next;// storing next var beside current var
            if((Integer)temp.elem>=start && (Integer)temp.elem<=end){
                    temp.prev.next= temp.next;
                    temp.next.prev= temp.prev;


                    DNode tail = dh.prev;//last index connecting to dh.prev
                    tail.next= temp;
                    temp.prev= tail;
                    temp.next= dh;
                    dh.prev= temp;
            }
            temp = newNode;
         }
    }
