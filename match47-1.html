5. public static void sumOddAppend(Node dh) {

       Node previous = dh;
       Node curr = dh.next;
       while ( curr!= dh){
        int value = (Integer)curr.elem;

            sum+= value;

            previous = curr;

        DNode temp = dh.next;

            length++;

        }

        DNode curr = dh.next;

            DNode nextNode = curr.next;

                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;
                DNode tail = dh.prev ;
                tail.next = curr;
