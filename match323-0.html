public static Node organizeBooks(Node head, Integer[] popularity) { 
    int n = popularity.length;
    
    for (int i = 0; i < n - 1; i++) {
        Node currentN = head;
        for (int j = 0; j < n - 1 - i; j++) {
            Node next = currentN.next;
            
            if (popularity[j] < popularity[j + 1]) {
                
              
                int temporayPop = popularity[j];
                popularity[j] = popularity[j + 1];
                popularity[j + 1] = temporayPop;
                
  
                String temporaryData = (String) currentN.elem;
                currentN.elem = next.elem;
                next.elem = tempData;
            }
            currentN = currentN.next;
        }
    }
    
    return head;
}

public static Node alternateMerge( Node head1, Node head2 ){
        Node nh1 = head1;
    Node nh2 = head2;

    while (nh1 != null && nh2 != null) {

        Node next1 = nh1.next;
        Node next2 = nh2.next;

        nh1.next = nh2;
        nh2.next = next1;

        nh1 = next1;
        nh2 = next2;
    }

    return head1;
}

  public static void sumOddAppend(Node dh) {
        
    if (dh.next == dh) {
        return;
    }
    
    Node current = dh.next;
    Node prev = dh;
    int sum = 0;
    while (current != dh) {
        int val = (Integer) current.elem;
        if (val % 2 != 0) {
            sum += val;
            prev.next = current.next;
            current = current.next;
        } else {
            prev = current;
            current = current.next;
        }
    }
    

    if (sum != 0) {
       
        Node last = dh;
        while (last.next != dh) {
            last = last.next;
        }
        
       
        Node newNode = new Node(sum);
        newNode.next = dh;
        last.next = newNode;
    }
    }
