public static String checkSimilar( Node building1, Node building2 ){
      
       Node temp1 = building1;
       Node temp2 = building2;




       while (temp1 != null) {
           if (temp1 != null && temp2 == null || temp1 == null && temp2 != null || temp1.next == null && temp2.next != null) {
               return "Not similar";
           }
          
           if (temp1.elem != temp2.elem) {
               return "Not similar";
           }
           temp1 = temp1.next;
           temp2 = temp2.next;
       }
       return "similar";
   }

           count2++;
           temp2 = temp2.next;

       for (int idx=0; idx<(count1+count2); idx++) {

public static void sumOddAppend(Node dh) {
       Node prev = dh;
       Node curr = dh.next;
       Node NN= new Node(0);


       while (curr != dh) {
           Node nextNode = curr.next;
           if ((int)curr.elem % 2 == 0) {
               prev.next = curr;
               prev = curr;
               curr = nextNode;
           }
           else {
               int x = (int) NN.elem;
               x += (int) curr.elem;
               NN.elem = x;
               curr = nextNode;
               prev.next = curr;


           }
       }
       prev.next = NN;
       NN.next = dh;
   }

public static void pairJoin(DNode dh1, DNode dh2) {
      
       DNode temp1 = dh1.next;
       DNode temp2 = dh2.next;
       int count = 0;
      
       while (temp1 != null) {
           if (count % 2 == 0) {
               DNode curr = temp1.next;
               temp1.next = temp2;
               temp2.prev = temp1;
               temp1 = curr;
           }
           else {
               DNode curr = temp2.next;
               temp2.next = temp1;
               temp1.prev = temp2;
               temp2 = curr;
           }
           count++;
       }
       temp2.next = dh1;
       dh1.prev = temp2;
      
   }

public static void rangeMove(DNode dh, int start, int end) {
       DNode prev = dh;
       DNode curr = dh.next;
       DNode accPrev = dh.prev;


       while (prev != accPrev) {
           DNode nextNode = curr.next;
           int x = (int) curr.elem;
           if (start <= x && x <= end) {
               DNode temp = curr;
               prev.next = nextNode;
               nextNode.prev = prev;
               curr = nextNode;
               temp.next = dh;
               temp.prev = dh.prev;
               dh.prev.next = temp;
               dh.prev = temp;
           }
           else {
               prev = curr;
               curr = nextNode;
           }


       }


   }
