public static String checkSimilar(Node building1, Node building2)
    {
        
         Node temp1 = building1;
         Node temp2 = building2;

    while (temp1 != null && temp2 != null) 
    {
        if (!temp1.elem.equals(temp2.elem)) 


        {
            return "Not Similar";
        }

        temp1 = temp1.next;
        temp2 = temp2.next;
    }

    if (temp1 == null && temp2 == null) 
    {
        return "Similar";
    }

        return "Not Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) 
    {
        
        int n = popularity.length;

        for (int i = 0; i < n - 1; i++) 
        {
            Node current = head;
        
            for (int j = 0; j < n - i - 1; j++) 
            {
                if (popularity[j] < popularity[j + 1]) 
                {
                    int temp = popularity[j];
                    popularity[j] = popularity[j + 1];
                    popularity[j + 1] = temp;


                    String book = (String) current.elem;
                    current.elem = current.next.elem;
                    current.next.elem = book;
                }
                current = current.next;
            }
        }
            return head;
    }

public static Node alternateMerge(Node head1, Node head2)
    {   
        Node pq1 = head1;
        Node pq2 = head2;
        Node next1;
        Node next2;

        while (pq1 != null && pq2 != null) 
        {
            next1 = pq1.next;
            next2 = pq2.next;
            pq1.next = pq2;

            if (next1 == null)
            {
                break; 
            }
            pq2.next = next1;
            pq1 = next1;
            pq2 = next2;
        }
        return head1; 
    }

public static Node idGenerator(Node head1, Node head2, Node head3) 
    {  
        Node current = head1;
        Node prev = null;
        Node next;

        while (current != null) 
        {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
        }
    
    Node id = prev;
    Node h2 = head2;
    Node h3 = head3;

    while (h2 != null && h3 != null) 
    {
        int sum = (int)h2.elem + (int)h3.elem;

        if (sum >= 10) 
        {
            sum = sum % 10;
        }

        Node newNode = new Node(sum);
        Node temp = id;

        while (temp.next != null) 
        {
            temp = temp.next;
        }
        temp.next = newNode;
        h2 = h2.next;
        h3 = h3.next;
    }
    return id;
    }

public static void sumOddAppend(Node dh) 
    {
        int sum = 0;
        Node prev = dh;
        Node curr = dh.next;

    while (curr != dh) 
    {
        if ((int) curr.elem % 2 != 0) 
        {
            sum += (int) curr.elem;


            prev.next = curr.next;
            curr = prev.next;

        } 
        else 
        {
            prev = curr;
            curr = curr.next;
        }
    }

       Node newNode = new Node(sum);
       newNode.next = dh;
       prev.next = newNode;
}

public static void pairJoin(DNode dh1, DNode dh2) 
    {
        DNode p1 = dh1.next;
        DNode p2 = dh2.next;
        DNode last = dh1;

        while (p1 != null && p2 != null) 
        {
            DNode next1 = p1.next;


            DNode next2 = p2.next;
            p1.next = p2;
            p2.prev = p1;
            p2.next = next1;

        if (next1 != null) 
        {
            next1.prev = p2;
        }

        last = p2;
        p1 = next1;
        p2 = next2;
    }

    if (last.next == null) 
    {
        last.next = dh1;
        dh1.prev = last;
    }
}

public static void rangeMove(DNode dh, int start, int end) 
    {
        DNode originalLast = dh.prev;
        DNode curr = dh.next;

        while (true) 
        {
            DNode next = curr.next;
            int value = (Integer) curr.elem;



            if (value >= start && value <= end) 
            {
                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;
                curr.prev = dh.prev;
                curr.next = dh;
                dh.prev.next = curr;
                dh.prev = curr;
            }

        if (curr == originalLast) 
        {
            break;
        }
        curr = next;
    }
