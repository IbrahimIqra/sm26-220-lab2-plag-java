 public static String checkSimilar( Node building1, Node building2 ){

        Node head1 = building1;
        Node head2 = building2;

        while (head1 != null && head2 != null) {



            if (!head1.elem.equals(head2.elem)) {
                return "Not Similar";
            }

            head1 = head1.next;
            head2 = head2.next;
        }

        
        if (head1 != null || head2 != null) {
            return "Not Similar";
        }

        return "Similar";
    
                
    }

       public static Node organizeBooks(Node head, Integer[] popularity) {

        int n = popularity.length;

        for (int i = 0; i < n - 1; i++) {

            Node present = head;

            for (int j = 0; j < n - 1 - i; j++) {

                
                if (popularity[j] < popularity[j + 1]) {

                    
                    int temp = popularity[j];
                    popularity[j] = popularity[j + 1];
                    popularity[j + 1] = temp;

                    
                    Object book = present.elem;
                    present.elem = present.next.elem;
                    present.next.elem = book;
                }

                present = present.next;
            }
        }

        return head;
    
            }

    public static Node alternateMerge( Node head1, Node head2 ){

            
        Node temp1 = head1;
        Node temp2 = head2;

        while (temp1 != null && temp2 != null) {

            Node next1 = temp1.next;
            Node next2 = temp2.next;

            temp1.next = temp2;

            if (next1 == null) {
                break;
            }



            temp2.next = next1;

            temp1 = next1;
            temp2 = next2;
        }

        return head1;
    
      
        
    }
