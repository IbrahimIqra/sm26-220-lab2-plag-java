    public static String checkSimilar( Node building1, Node building2 ){

        while(building1 != null && building2 != null){

            if(building1.elem.equals(building2.elem)){
                building1 = building1.next;
                building2 = building2.next;
            }


            else return "Not Similar";
        }

        if(building1 == null && building2 == null) return "Similar";
        else return "Not Similar";

    }

    public static Node organizeBooks(Node head, Integer[] popularity) {

        Node temp = head;
        int c = 0;

        while (temp != null) {
            c++;
            temp = temp.next;
        }

        for (int i = 0 ; i < (c - 1) ; i++) {

            Node curr = head;

            for (int j = 0 ; j < (c - 1 - i) ; j++) {

                Node nextNode = curr.next;

                if (popularity[j] < popularity[j + 1]) {

                    int rating = popularity[j];
                    popularity[j] = popularity[j + 1];
                    popularity[j + 1] = rating;

                    Object name = curr.elem;
                    curr.elem = nextNode.elem;
                    nextNode.elem = name;
                }



                curr = curr.next;
            }
        }

        return head;
    }

    public static Node alternateMerge( Node head1, Node head2 ){

        Node curr1 = head1;
        Node curr2 = head2;

        while (curr1 != null && curr2 != null) {

            Node next1 = curr1.next;
            Node next2 = curr2.next;

            curr1.next = curr2;

            if (next1 != null) {
                curr2.next = next1;
            }

            else{
                curr2.next = next2;
                break;
            }



            curr1 = next1;
            curr2 = next2;
        }

        return head1;
    }

    public static Node idGenerator(Node head1, Node head2, Node head3) {


        Node newHead = null;
        Node curr1 = head1;
        Node curr2 = head2;
        Node curr3 = head3;

        while (curr1 != null) {

            int val1 = (int) curr1.elem;

            Node newNode = new Node(val1);
            newNode.next = newHead;


            newHead = newNode;

            curr1 = curr1.next;
        }

        Node tail = newHead;

        while (tail != null && tail.next != null) {
            tail = tail.next;
        }

        while (curr2 != null && curr3 != null) {

            int val2 = (int) curr2.elem;
            int val3 = (int) curr3.elem;
            int sum = val2 + val3;

            int newNum = sum % 10;

            Node newNode = new Node(newNum);

            if (tail == null) {
                newHead = newNode;
                tail = newNode;
            }

            else {


                tail.next = newNode;
                tail = newNode;
            }

            curr2 = curr2.next;
            curr3 = curr3.next;
        }

        return newHead;
    }

    public static void sumOddAppend(Node dh) {

        Node prev = dh;
        Node curr = dh.next;
        int sum = 0;



        while (curr != dh) {
            int val = (int) curr.elem;

            if (val % 2 != 0) {
                sum += val;
                prev.next = curr.next;
                curr = curr.next;
            }

            else {
                prev = curr;
                curr = curr.next;
            }
        }

        Node sum1 = new Node(sum);

        prev.next = sum1;
        sum1.next = dh;
    }

    public static void pairJoin(DNode dh1, DNode dh2) {

        DNode curr1 = dh1.next;
        DNode curr2 = dh2.next;
        DNode tail = dh1;

        while (curr1 != dh1 && curr1 != null && curr2 != dh2 && curr2 != null) {



            DNode next1 = curr1.next;
            DNode next2 = curr2.next;

            tail.next = curr1;
            curr1.prev = tail;
            tail = curr1;

            tail.next = curr2;
            curr2.prev = tail;
            tail = curr2;

            curr1 = next1;
            curr2 = next2;
        }

        tail.next = dh1;
        dh1.prev = tail;
    }

    public static void rangeMove(DNode dh, int start, int end) {

        DNode curr = dh.next;
        DNode tail = dh.prev;

        while (curr != dh) {



            DNode next = curr.next;
            int val = (int) curr.elem;

            if (val >= start && val <= end) {

                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;

                DNode currTail = dh.prev;

                currTail.next = curr;
                curr.prev = currTail;
                curr.next = dh;
                dh.prev = curr;
            }

            if (curr == tail) {
                break;
            }

            curr = next;
        }
    }
