public static String checkSimilar(Node building1, Node building2) {
    Node b1 = building1;
    Node b2 = building2;
    while(b1 != null && b2 != null) {
        if(b1.elem != b2.elem) {
            return "Not Similar";
        }
        b1 = b1.next;
        b2 = b2.next;
    }
    if(b1 == null && b2 == null) {
        return "Similar";
    } else {
        return "Not Similar";
    }
}

public static Node organizeBooks(Node head, Integer[] popularity) {
    int n = popularity.length;
    for(int i = 0; i < n-1; i++) {
        Node current = head;
        Node nextNode = current.next;
        for(int j = 0; j < n-i-1; j++) {


            if(popularity[j] < popularity[j+1]) {
                int currentPopularity = popularity[j];
                popularity[j] = popularity[j+1];
                popularity[j+1] = currentPopularity;
                Object temp = current.elem;
                current.elem = nextNode.elem;
                nextNode.elem = temp;
            }
            current = current.next;
            nextNode = current.next;
        }
    }
    return head;
}

public static Node alternateMerge(Node head1, Node head2) {
    if(head1 == null) 
        return head2;
    if(head2 == null) 
        return head1;
    


    Node temp1 = head1;
    Node temp2 = head2;
    
    while(temp1 != null && temp2 != null) {
        Node front = temp1.next;
        temp1.next = temp2;
        Node front2 = temp2.next;
        temp2.next = front;
        temp1 = front;
        temp2 = front2;
    }
    
    return head1;
  }

public static void sumOddAppend(Node dh) {
    int sum = 0;
    Node store = dh;
    Node curr = dh.next;
    
    while(curr != dh) {
        if((int)curr.elem % 2 == 1) {
            sum += (int)curr.elem;


            store.next = curr.next;
            curr = curr.next;
        } else {
            store = curr;
            curr = curr.next;
        }
    }
    
    Node sumNode = new Node(sum);
    store.next = sumNode;
    sumNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode curr1 = dh1.next;
    DNode curr2 = dh2.next;
    
    while(curr1 != dh1 && curr2 != dh2) {
        DNode next1 = curr1.next;
        DNode next2 = curr2.next;
        
        curr1.next = curr2;


        curr2.prev = curr1;
        
        curr2.next = next1;
        next1.prev = curr2;
        
        curr1 = next1;
        curr2 = next2;
    }
    
    dh2.next = dh2;
    dh2.prev = dh2;
}

public static void rangeMove(DNode dh, int start, int end) {
    int size = 0;
    DNode counter = dh.next;
    while(counter != dh) {
        size++;
        counter = counter.next;
    }
    


    DNode current = dh.next;
    for(int i = 0; i < size; i++) {
        DNode nextNodeToProcess = current.next;
        
        if((int)current.elem >= start && (int)current.elem <= end) {


            current.prev.next = current.next;
            current.next.prev = current.prev;
            
            DNode tail = dh.prev;
            tail.next = current;
            current.prev = tail;
            current.next = dh;
            dh.prev = current;
        }
        
        current = nextNodeToProcess;
    }
}
