public static String checkSimilar( Node building1, Node building2 ){
        
        Node first = building1;
        Node second = building2;


        int firstCount = 0;
        int secondCount = 0;
        while(first!=null){
            firstCount++;
            first = first.next;
        }
        while(second!=null){
            secondCount++;
            second = second.next;
        }
        if(firstCount!=secondCount){
            return "Not Similar";
        }
        first = building1;
        second = building2;


        while(first!=null && second!=null){
            if(!first.elem.equals(second.elem)){
                return "Not Similar";
            }
            first = first.next;
            second = second.next;
        }


        return "Similar";

    }

public static Node organizeBooks(Node head, Integer[] popularity) { 
        
        int leng = popularity.length;
        for(int i = 0;i<leng-1;i++){
            Node currentBook = head;
            Node nextBook = currentBook.next;
            for(int j = 0;j<leng-1-i;j++){
                if(popularity[j]<popularity[j+1]){
                    int tempValue = popularity[j];
                    popularity[j] = popularity[j+1];
                    popularity[j+1]=tempValue;
                    Object substituteBook = currentBook.elem;
                    currentBook.elem = nextBook.elem;


                    currentBook.next.elem = substituteBook;
                }
                currentBook = currentBook.next;
                nextBook = nextBook.next;
            }
        }
        
        return head; 
    }

public static Node alternateMerge( Node head1, Node head2 ){
        
        Node now1 =  head1;
        Node now2 = head2;
        while(now1!=null && now2!=null){
            Node firstN = now1.next;
            Node secondN = now2.next;
            now1.next = now2;
            now2.next = firstN;
            now1 = firstN;
            now2 = secondN;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        
        Node previous = null;
        Node presentPoint = head1;
        while(presentPoint!=null){
            Node frontNext = presentPoint.next;
            presentPoint.next = previous;
            previous = presentPoint;
            presentPoint = frontNext;
        }
        head1 = previous;
        Node presentPoint2 = head2;
        Node presentPoint3 = head3;
        Node last = head1;
        while(last.next!=null){
            last = last.next;
        }
        while(presentPoint2!=null && presentPoint3!=null){
            int add = (int)presentPoint2.elem + (int)presentPoint3.elem;
            if(add>=10){
                add = add%10;
            }
            Node newNode = new Node(add);
            last.next = newNode;
            last = last.next;
            presentPoint2 = presentPoint2.next;
            presentPoint3 = presentPoint3.next;

        }

        return head1;
    }

public static void sumOddAppend(Node dh) {
        
        Node current = dh.next;
        Node earlierVal = dh;
        int summation = 0;
        while(current!=dh){
            if((int)current.elem%2==0){
                earlierVal = current;
            }
            else{
                summation = summation + (int)current.elem;
                earlierVal.next = current.next;
            }


            current = current.next;
        }
        Node newNd = new Node(summation);
        earlierVal.next = newNd;
        newNd.next = dh;
        
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        
        DNode present1 = dh1.next;
        DNode present2 = dh2.next;
        
        while(present1!=null && present2 !=null){
            DNode next_a = present1.next;
            DNode next_b = present2.next;
            present1.next = present2;
            present2.prev = present1;
            
            present2.next = next_a;
            if(next_a==null){
                break;
            }
            next_a.prev = present2;
            present1 = next_a;
            present2 = next_b;
            
        }
        present2.next = dh1;
        dh1.prev = present2;

    }

public static void rangeMove(DNode dh, int start, int end) {
        int length = 0;
        DNode count = dh.next;
        while(count!=dh){
            length++;
            count = count.next;
        }
        DNode thisMoment = dh.next;
        for(int i = 1;i<=length;i++){


            DNode nextMoment = thisMoment.next;
            if((int)thisMoment.elem>=start && (int)thisMoment.elem<=end){
                thisMoment.prev.next = thisMoment.next;
                thisMoment.next.prev = thisMoment.prev;
                DNode endingNode = dh.prev;
                endingNode.next = thisMoment;
                thisMoment.prev = endingNode;
                thisMoment.next = dh;
                dh.prev = thisMoment;
            }
            thisMoment = nextMoment;

        }
    }
