public static String checkSimilar(Node building1, Node building2) {
    Node temp1 = building1;
    Node temp2 = building2;
    while(temp1!=null && temp2!=null){
        if(!temp1.elem.equals(temp2.elem)){
            return "Not Similar";
        }
        temp1 = temp1.next;
        temp2 = temp2.next;
    }
    if(temp1!=null || temp2!=null){
        return "Not Similar";
    }
    return "Similar";
}

public static Node organizeBooks(Node head, Integer[] popularity) {
    if(head==null || head.next==null){
        return head;
    }
    int n = popularity.length;
    Node current = head;
    for(int i=0; i<n-1; i++){
        Node prev = null;
        Node curr = head;
        Node next = curr.next;
        while(next!=null){
            for (int j=0; j<n-1-i; j++){
                int popCurr = popularity[j];
                int popNext = popularity[j+1];
                if (popCurr < popNext){
                    if(prev==null){
                        head = next;
                    }else{
                        prev.next = next;
                    }
                    curr.next = next.next;
                    next.next = curr;
                    popularity[j] = popNext;
                    popularity[j+1] = popCurr;
                    prev = next;
                    next = curr.next;
                }else{
                    prev = curr;
                    curr = next;
                    next = next.next;
                }
            }
        }
        return head;
    }
}

public static Node alternateMerge( Node head1, Node head2 ){
    if(head1==null){
        return head2;
    }
    if(head2==null){
        return head1;
    }
    Node curr1 = head1;
    Node curr2 = head2;
    Node next1, next2;
    while(curr1!=null && curr2!=null){
        next1 = curr1.next;  


        next2 = curr2.next;  
        curr1.next = curr2;
        if(next1==null){
            break;
        }
        curr2.next = next1;
        curr1 = next1;
        curr2 = next2;
    }
    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node prev = null;
    Node curr = head1;
    Node next = null;
    while(curr!=null){
        next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }


    Node reversedHead1 = prev;
    Node idHead = reversedHead1;
    Node tail = idHead;
    while (tail.next!=null){
        tail = tail.next;
    }
    Node p2 = head2;
    Node p3 = head3;
    while (p2!=null && p3!=null){
        int sum = ((Integer)p2.elem+(Integer)p3.elem)%10;  
        Node newNode = new Node(sum); 
        tail.next = newNode; 
        tail = newNode; 
        p2 = p2.next;
        p3 = p3.next;
    }
    return idHead;
}

public static void sumOddAppend(Node dh) {
    if(dh==null || dh.next==dh){
        return; 
    }
    Node prev = dh;       
    Node curr = dh.next;  
    int sum = 0;
    while(curr!=dh){
        int value = (Integer)curr.elem;
        if(value%2!=0){ 
            sum+=value;
            prev.next = curr.next;  
        }else{
            prev = curr;  
        }
        curr = curr.next; 
    }
    Node newNode = new Node(sum);
    prev.next = newNode;
    newNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode n1 = dh1.next;
    DNode n2 = dh2.next;
    while(n1!=null && n2!=null){


        DNode n1Next = n1.next; 
        DNode n2Next = n2.next; 
        n1.next = n2;
        n2.prev = n1;
        if(n1Next!=null){
            n2.next = n1Next;
            n1Next.prev = n2;
        }else{
            n2.next = dh1; 
            dh1.prev = n2;
            break;
        }
        n1 = n1Next;
        n2 = n2Next;
        if(n2==null){
            n1.prev.next = dh1;
            dh1.prev = n1.prev.next;
            break;
        }
    }


    DNode tail = dh1;
    while(tail.next!=null && tail.next!=dh1){
        tail = tail.next;
    }
    tail.next = dh1;
    dh1.prev = tail;
}

public static void rangeMove(DNode dh, int start, int end) {
    DNode current = dh.next;
    while(current!=dh){
        int val = (int)current.elem;
        DNode nextNode = current.next; 
        if(val>=start && val<=end){


            current.prev.next = current.next;
            current.next.prev = current.prev;
            DNode last = dh.prev;
            last.next = current;
            current.prev = last;
            current.next = dh;
            dh.prev = current;
        }
        current = nextNode; 
    }
}
