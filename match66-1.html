public static void sumOddAppend(Node dh) {
        int sum = 0;
        Node prev = dh;
        Node curr = dh.next;

        while (curr!=dh){
            int value = (Integer)curr.elem;
            if(value%2!=0){
                sum+=value;


                prev.next=curr.next;
                curr = prev.next;
            }
            else{
                prev= curr;
                curr = curr.next;
            }
        }
        Node newNode = new Node (sum);
        Node tail = dh;

        while(tail.next!=dh){


            tail = tail.next;
        }
        tail.next = newNode;
        newNode.next = dh;
}

     public static void pairJoin(DNode dh1, DNode dh2) {
        DNode p1 = dh1.next;
        DNode p2 = dh2.next;

        while (p1!=null && p2!=null){
            DNode next1 = p1.next;
            DNode next2 =  p2.next;

            p1.next= p2;
            p2.prev = p1;

            if(next1!=null){
                p2.next = next1;
                next1.prev = p2;
            }
            else{
                p2.next = null;
            }

            p1 = next1;
            p2 = next2;
        }

        DNode tail = dh1;
        while (tail.next!=null){
            tail = tail.next;
        }
        tail.next = dh1;
        dh1.prev = tail;
    }
