public static String checkSimilar(Node building1, Node building2) {
    Node pointer1 = building1;
    Node pointer2 = building2;



    while (pointer1 != null && pointer2 != null) {
        if (!pointer1.elem.equals(pointer2.elem)) {
            return "Not Similar";
        }
        pointer1 = pointer1.next;
        pointer2 = pointer2.next;
    }

    if (pointer1 == null && pointer2 == null) {
        return "Similar";
    }
    return "Not Similar";
}

public static Node organizeBooks(Node head, Integer[] popularity) {
    if (head == null || head.next == null || popularity == null || popularity.length <= 1) {
        return head;
    }

    int n = popularity.length;

    for (int pass = 0; pass < n - 1; pass++) {
        Node previous = null;
        Node current = head;
        Node nextNode = head.next;

        for (int i = 0; i < n - 1 - pass && nextNode != null; i++) {
            if (popularity[i] < popularity[i + 1]) {
                current.next = nextNode.next;
                nextNode.next = current;

                if (previous == null) {
                    head = nextNode;
                } else {
                    previous.next = nextNode;
                }

                int temp = popularity[i];
                popularity[i] = popularity[i + 1];
                popularity[i + 1] = temp;

                previous = nextNode;
                nextNode = current.next;
            } else {
                previous = current;
                current = nextNode;
                nextNode = nextNode.next;
            }
        }
    }

    return head;
}

public static Node alternateMerge(Node head1, Node head2) {
    if (head1 == null) {
        return head2;
    }
    if (head2 == null) {
        return head1;
    }

    Node current1 = head1;
    Node current2 = head2;

    while (current1 != null && current2 != null) {
        Node next1 = current1.next;
        Node next2 = current2.next;

        current1.next = current2;

        if (next1 == null) {
            break;
        }

        current2.next = next1;

        current1 = next1;
        current2 = next2;
    }

    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node resultHead = null;
    Node resultTail = null;

    Node current1 = head1;
    while (current1 != null) {
        resultHead = new Node((Integer) current1.elem, resultHead);
        if (resultTail == null) {


            resultTail = resultHead;
        }
        current1 = current1.next;
    }

    Node current2 = head2;
    Node current3 = head3;

    while (current2 != null && current3 != null) {
        int digit = ((Integer) current2.elem + (Integer) current3.elem) % 10;
        Node newNode = new Node(digit);
        if (resultHead == null) {
            resultHead = newNode;
            resultTail = newNode;
        } else {
            resultTail.next = newNode;
            resultTail = newNode;
        }
        current2 = current2.next;
        current3 = current3.next;
    }

    return resultHead;
}

public static void sumOddAppend(Node dh) {
    int sum = 0;

    Node prev = dh;
    Node current = dh.next;

    while (current != dh) {
        int value = (int) current.elem;

        if (value % 2 != 0) {
            sum += value;
            prev.next = current.next;
            current = prev.next;
        } else {
            prev = current;
            current = current.next;
        }
    }

    Node sumNode = new Node(sum);
    sumNode.next = dh;
    prev.next = sumNode;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode p1 = dh1.next;
    DNode p2 = dh2.next;
    DNode tail = dh1;

    while (p1 != null && p2 != null) {
        DNode next1 = p1.next;
        DNode next2 = p2.next;

        tail.next = p1;
        p1.prev = tail;
        tail = p1;

        tail.next = p2;
        p2.prev = tail;
        tail = p2;

        p1 = next1;
        p2 = next2;
    }

    while (p1 != null) {
        DNode next1 = p1.next;
        tail.next = p1;
        p1.prev = tail;
        tail = p1;
        p1 = next1;
    }

    while (p2 != null) {
        DNode next2 = p2.next;
        tail.next = p2;
        p2.prev = tail;
        tail = p2;
        p2 = next2;
    }

    tail.next = dh1;
    dh1.prev = tail;

    dh2.next = null;
    dh2.prev = null;
}

public static void rangeMove(DNode dh, int start, int end) {
    if (dh == null || dh.next == dh) {
        return;
    }

    DNode originalTail = dh.prev;
    DNode tail = dh.prev;
    DNode current = dh.next;

    while (true) {
        DNode nextNode = current.next;


        int value = (int) current.elem;

        if (value >= start && value <= end) {
            current.prev.next = current.next;
            current.next.prev = current.prev;

            tail.next = current;
            current.prev = tail;
            current.next = dh;
            dh.prev = current;
            tail = current;
        }

        if (current == originalTail) {
            break;
        }

        current = nextNode;
    }
}
