// AssignmentTask1: Building Blocks



    public static String checkSimilar( Node building1, Node building2 ){

        Node a = building1;
        Node b = building2;

        while(a != null && b != null) {
            if(!a.elem.equals(b.elem)) {


                return "Not Similar";
            }

            a = a.next;
            b = b.next;
        }

        if(a == null && b == null) {
            return "Similar";
        } else {
            return "Not Similar";
        }
    }

// AssignmentTask2: Organize Books

    public static Node organizeBooks(Node head, Integer[] popularity) {

        if(head == null) {
            return null;
        }

        boolean kono_kichu_swapped_hoise = true;

        while(kono_kichu_swapped_hoise) {
            kono_kichu_swapped_hoise = false;
            
            Node current = head;
            int index = 0;

            while(current.next != null) {
                if(popularity[index] < popularity[index + 1]) {
                    String title_hold_kore = (String)current.elem;

                    current.elem = current.next.elem;
                    current.next.elem = title_hold_kore;

                    Integer score_hold_kore = popularity[index];

                    popularity[index] = popularity[index + 1];


                    popularity[index + 1] = score_hold_kore;

                    kono_kichu_swapped_hoise = true;
                }

                current = current.next;
                index++;
            }
        }

        return head;
    }

// AssignmentTask3: Alternate Merge

    public static Node alternateMerge( Node head1, Node head2 ){

        Node walk1 = head1;
        Node walk2 = head2;



        while(walk1 != null && walk2 != null) {
            Node rest1 = walk1.next;
            Node rest2 = walk2.next;

            walk1.next = walk2;

            if(rest1 != null) {
                walk2.next = rest1;
            }

            walk1 = rest1;
            walk2 = rest2;
        }

        return head1;
    }

// AssignmentTask5: Sum Odd Append

    public static void sumOddAppend(Node dh) {

        int odd_er_total = 0;

        Node previous = dh;
        Node current = dh.next;

        while(current != dh) {
            int value = (int) current.elem;

            if(value % 2 != 0) {
                odd_er_total += value;
                previous.next = current.next;


                current = current.next;
            } else {
                previous = current;
                current = current.next;
            }
        }

        Node node_er_sum = new Node(odd_er_total);
        node_er_sum.next = dh;

        previous.next = node_er_sum;
    }

// AssignmentTask6: Pair Join

    public static void pairJoin(DNode dh1, DNode dh2) {

        DNode tail = dh1;
        DNode p1 = dh1.next;
        DNode p2 = dh2.next;

        while(p1 != null) {


            DNode next1 = p1.next;
            DNode next2 = p2.next;

            tail.next = p1;
            p1.prev = tail;
            tail = p1;

            tail.next = p2;
            p2.prev = tail;
            tail = p2;

            p1 = next1;
            p2 = next2;
        }

        tail.next = dh1;
        dh1.prev = tail;
    }

// AssignmentTask7: Range Move

    public static void rangeMove(DNode dh, int start, int end) {

        DNode scan = dh.next;
        int count = 0;

        while(scan != dh) {
            count++;
            scan = scan.next;
        }

        DNode current = dh.next;

        for(int i = 0; i < count; i++) {


            DNode next_up = current.next;
            int value = (int) current.elem;

            if(value >= start && value <= end) {
                current.prev.next = current.next;
                current.next.prev = current.prev;

                DNode back = dh.prev;

                back.next = current;
                current.prev = back;
                current.next = dh;
                dh.prev = current;
            }

            current = next_up;
        }
    }
