 public static String checkSimilar( Node building1, Node building2 ){
         Node temp1=building1;
         Node temp2=building2;
        while (temp1!= null && temp2!= null) {
            if (temp1.elem != temp2.elem) {
                return "Not Similar";
            } 
            temp1 = temp1.next;
            temp2 = temp2.next;
        }
        if(temp1!= null || temp2!= null) {


            return "Not Similar";
        }
        return "Similar";
    }

 public static Node organizeBooks(Node head, Integer[] popularity) {
    int n = popularity.length;
    for (int i=0; i < n-1; i++){
        Node temp = head;
        int m= 0;
        while (temp.next != null && m<n-1-i) {


            if (popularity[m] < popularity[m+1]) {
                int temp1 = popularity[m];
                popularity[m] = popularity[m+1];
                popularity[m+1] = temp1;
                Object temp2 = temp.elem;
                temp.elem = temp.next.elem;
                temp.next.elem = temp2;
            }
            temp = temp.next;


            m++;
        }
    }
    return head;
}

public static void sumOddAppend(Node dh) {
     Node newNode=dh.next;
        Node prev=dh;
        int sum=0;
        while(newNode!=dh){
            if((int)newNode.elem%2!=0){
                prev.next=newNode.next;


                sum+=(int)newNode.elem;
                newNode=newNode.next;
            }
            else{
                prev=newNode;
                newNode=newNode.next;
            }
        }
        Node sumNode=new Node(sum);
        sumNode.next=dh;
        prev.next=sumNode;
    }

 public static void rangeMove(DNode dh, int start, int end) {
         DNode newNode = dh.next;
        DNode lastNode = dh.prev;
        DNode endNode = dh.prev;
        while(newNode!=endNode.next) {
            DNode temp = newNode.next;


            if ((int)newNode.elem >= start && (int)newNode.elem <= end) {
                if(newNode!=lastNode){
                newNode.prev.next = newNode.next;
                newNode.next.prev = newNode.prev;
                lastNode.next = newNode;
                newNode.prev = lastNode;
                newNode.next = dh;
                dh.prev = newNode;
                lastNode = newNode;
                }
            }
            newNode=temp;
        }
    }
