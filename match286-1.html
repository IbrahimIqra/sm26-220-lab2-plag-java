public static void pairJoin(DNode dh1, DNode dh2){

    DNode cur= dh1.next;
    DNode cur2=dh2.next;

    while(cur!= null && cur2!= null ){
        DNode next1 =cur.next;
        DNode next2=cur2.next ;

        cur.next =cur2;
        cur2.prev = cur;
        cur2.next= next1 ;

        if (next1!= null){
          next1.prev=cur2;
        }
        cur =next1;
        cur2 =next2;    }

}

    DNode cur= dh.next;
    DNode stop= tail;

    while(true)  {
        DNode next =cur.next;
        int num =(Integer)cur.elem;

        if(num >= start && num <= end && cur != stop){



          cur.prev.next = cur.next;
            cur.next.prev= cur.prev;



          cur.prev= tail;
            cur.next = dh;

           tail.next = cur;
            dh.prev = cur;
