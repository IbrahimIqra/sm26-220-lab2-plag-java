public static String checkSimilar(Node building1, Node building2) {
    Node first = building1;
    Node second = building2;

    // compare the blocks one by one
    while (first != null && second != null) {
        if (!first.elem.equals(second.elem)) {
            return "Not Similar";
        }
        first = first.next;
        second = second.next;
    }

    if (first == null && second == null) {
        return "Similar";
    } else {
        return "Not Similar";
    }
}

public static Node organizeBooks(Node head, Integer[] popularity) {
    if (head == null) {
        return head;
    }

    for (int i = 0; i < popularity.length - 1; i++) {
        Node current = head;

        for (int j = 0; j < popularity.length - i - 1; j++) {
            if (popularity[j] < popularity[j + 1]) {
                Integer tempScore = popularity[j];
                popularity[j] = popularity[j + 1];
                popularity[j + 1] = tempScore;

                Object tempBook = current.elem;
                current.elem = current.next.elem;
                current.next.elem = tempBook;
            }
            current = current.next;
        }
    }
    return head;
}

public static Node alternateMerge(Node head1, Node head2) {
    Node answer = head1;

    while (head1 != null && head2 != null) {
        Node nextFromFirst = head1.next;
        Node nextFromSecond = head2.next;

        head1.next = head2;

        if (nextFromFirst == null) {
            break;
        }



        head2.next = nextFromFirst;
        head1 = nextFromFirst;
        head2 = nextFromSecond;
    }
    return answer;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node idHead = null;
    Node current = head1;

    // inserting at the front reverses the first four digits
    while (current != null) {
        Node digitNode = new Node(current.elem);
        digitNode.next = idHead;
        idHead = digitNode;


        current = current.next;
    }

    Node tail = idHead;
    while (tail.next != null) {
        tail = tail.next;
    }

    Node second = head2;
    Node third = head3;

    while (second != null && third != null) {
        int sum = (Integer) second.elem + (Integer) third.elem;
        Node digitNode = new Node(sum % 10);

        tail.next = digitNode;
        tail = digitNode;


        second = second.next;
        third = third.next;
    }
    return idHead;
}

public static void sumOddAppend(Node dh) {
    int oddSum = 0;
    Node previous = dh;
    Node current = dh.next;

    while (current != dh) {
        int number = (Integer) current.elem;

        if (number % 2 != 0) {
            oddSum += number;
            previous.next = current.next;
            current = previous.next;
        } else {
            previous = current;
            current = current.next;
        }
    }

    // only one new node is made for the sum
    Node sumNode = new Node(oddSum);
    previous.next = sumNode;
    sumNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode list1 = dh1.next;
    DNode list2 = dh2.next;
    DNode last = dh1;

    while (list1 != null && list2 != null) {
        DNode next1 = list1.next;
        DNode next2 = list2.next;

        last.next = list1;


        list1.prev = last;
        list1.next = list2;
        list2.prev = list1;

        last = list2;
        list1 = next1;
        list2 = next2;
    }

    // join the last node with the first dummy head
    last.next = dh1;
    dh1.prev = last;
}

public static void rangeMove(DNode dh, int start, int end) {
    if (dh.next == dh) {
        return;
    }

    DNode oldLast = dh.prev;
    DNode current = dh.next;

    while (current != dh) {
        DNode nextNode = current.next;
        boolean lastOldNode = current == oldLast;


        int number = (Integer) current.elem;

        if (number >= start && number <= end) {
            // remove current from its present place
            current.prev.next = current.next;
            current.next.prev = current.prev;

            // attach it at the end
            DNode last = dh.prev;
            last.next = current;
            current.prev = last;
            current.next = dh;
            dh.prev = current;
        }

        if (lastOldNode) {
            break;
        }
        current = nextNode;
    }
}
