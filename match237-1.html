public static String checkSimilar( Node building1, Node building2 ){
    Node t1 = building1;
    Node t2 = building2;
    while (t1 != null && t2 != null) {
        if (!t1.elem.equals(t2.elem)) {
            return "Not Similar";
        }
        t1 = t1.next;
        t2 = t2.next;
    }
    if (t1 != null || t2 != null) {
        return "Not Similar";
    }
    return "Similar";  
    }

public static Node alternateMerge(Node head1,Node head2 ){
    Node temp1 =head1;

 public static void sumOddAppend(Node dh) {
    int sum =0;
    Node prev =dh;
    Node curr =dh.next;
    while(curr !=dh){
        int val =(Integer)curr.elem;
        if (val%2!=0){    
            sum += val;
            prev.next = curr.next;
            curr = prev.next;
        }
         else 
            {                 
            prev = curr;
            curr = curr.next;
        }
    }


    Node newNode = new Node(sum);
newNode.next = dh;
prev.next = newNode;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode p1 = dh1.next;
    DNode p2 = dh2.next;
    DNode last = dh1;
    while (p1 != null && p2 != null) {
        DNode next1 = p1.next;
        DNode next2 = p2.next;
        p1.next = p2;
        p2.prev = p1;
        if (next1 != null) {
            p2.next = next1;
            next1.prev = p2;
        } else {


            p2.next = null;
        }
        last = p2;
        p1 = next1;
        p2 = next2;
    }
    last.next = dh1;
    dh1.prev = last;
}

public static void rangeMove(DNode dh, int start, int end){
        
    DNode tail = dh.prev;  


    DNode stop = tail;             
    DNode curr = dh.next;
    while (curr!=dh){
        DNode next =curr.next;
        int value =(Integer)curr.elem;
        if (value>= start&&value<= end) {
            if (curr ==stop){
                break;
            }
            curr.prev.next = curr.next;
            curr.next.prev = curr.prev;
            curr.next = dh;
            curr.prev = tail;
            tail.next = curr;
            dh.prev = curr;
            tail = curr;
        }
        if (curr ==stop){
            break;
        }
        curr = next;
    }
}
