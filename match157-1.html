public static String checkSimilar( Node building1, Node building2 ){
        Node temp1 = building1;
        Node temp2 = building2;
        while (temp1 != null && temp2 != null){
            if(!temp1.elem.equals(temp2.elem)){
                return "Not Similar";
            }
                temp1 = temp1.next;
                temp2 = temp2.next;
            }
        if(temp1 != null || temp2 != null){
                    return "Not Similar";
        }
        return "Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {
        if(head == null || head.next == null){
            return head;
        }


        for (int i = 0; i < popularity.length - 1;i ++ ){
        Node temp = head;
            for(int j = 0;j < popularity.length - 1-i;j ++){
                if (popularity[j] < popularity[j+1]){
                    int tempPopularity = popularity[j] ;
                    popularity[j] = popularity[j+1];
                    popularity[j+1] = tempPopularity;


                    Object tempBook = temp.elem;
                    temp.elem = temp.next.elem;
                    temp.next.elem = tempBook;
                }
                temp = temp.next;
            }
        }
        return head; 
    }

 public static Node alternateMerge( Node head1, Node head2 ){
        Node temp1 = head1;
        Node temp2 = head2;
        while(temp1 != null && temp2!= null){
            Node nextNode1 = temp1.next;
            Node nextNode2 = temp2.next;
            temp1.next = temp2;
            temp2.next = nextNode1;
            temp1 = nextNode1;


            temp2 = nextNode2;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node temp1 = head1;
        Node temp2 = head2;
        Node temp3 = head3;
        Node revHead = null;
        while(temp1 != null){
            Node newNode = new Node(temp1.elem);
            newNode.next = revHead;
            revHead = newNode;


            temp1 = temp1.next;
        }
        Node tail = revHead;
        while(tail.next != null){
            tail = tail.next;
        }
        while(temp2 != null && temp3 != null){
            int sum = (int)temp2.elem + (int)temp3.elem;
            if(sum > 9){
                sum %= 10;
            }
            tail.next = new Node(sum);
            tail = tail.next;
            temp2 = temp2.next;
            temp3 = temp3.next;   
        }
         return revHead;     
    }

public static void sumOddAppend(Node dh) {
        Node temp = dh.next;
        if(dh.next == dh){
            return;
        }
        Node list = dh;


        int sum = 0;
        while (temp != dh){
            if((int)temp.elem %2 != 0){
                sum += (int)temp.elem;
                list.next = temp.next;


                temp = list.next;
            }
            else{
                list = temp;
                temp = temp.next;
            }
        }
        Node tail = new Node(sum);
        list.next = tail;;


        tail.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode temp1 = dh1.next;
        DNode temp2 = dh2.next;
        while (temp1 != null && temp2 != null){
            DNode next1 = temp1.next;
            DNode next2 = temp2.next;
            temp1.next = temp2;
            temp2.prev = temp1;
            if(next1 == null){
                temp2.next = dh1;
                dh1.prev = temp2;
                break;
            }
            temp2.next = next1;
            next1.prev = temp2;
            temp1 = next1;


            temp2 = next2;
        }
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode temp = dh.next;
        DNode last = dh.prev;
        while(temp != dh){


             DNode next = temp.next;
            if((int)temp.elem >= start && (int)temp.elem <= end){
                temp.prev.next = temp.next;
                temp.next.prev = temp.prev;
                temp.prev = dh.prev;
                temp.next = dh;
                dh.prev.next = temp;
                dh.prev = temp;
            }
            if (temp == last){
                break;
            }
            temp = next;
        }
    }
