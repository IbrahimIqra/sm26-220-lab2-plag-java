public static String checkSimilar( Node building1, Node building2 ){
        Node n1 = building1;
        Node n2 = building2;
        
        while (n1 != null && n2 != null) {
            if (!n1.elem.equals(n2.elem)) {
                return "Not Similar";
            }
            n1 = n1.next;
            n2 = n2.next;
        }

        if (n1 != null || n2 != null) {
            return "Not Similar";
        }

    return "Similar";
    }

 public static Node organizeBooks(Node head, Integer[] popularity) {
        
    int n = popularity.length;
    Node[] nodes = new Node[n];

    Node current = head;
    for (int i = 0; i < n; i++) {
        nodes[i] = current;
        current = current.next;
    }

    Integer[] order = new Integer[n];
    for (int i = 0; i < n; i++) {
        order[i] = i;
    }

    
    Arrays.sort(order, (a, b) -> popularity[b] - popularity[a]);

    Node newHead = nodes[order[0]];
    Node tail = newHead;
    for (int i = 1; i < n; i++) {
        tail.next = nodes[order[i]];
        tail = tail.next;
    }
    tail.next = null;



    return newHead;
    }

 public static Node alternateMerge( Node head1, Node head2 ){
        
    if (head1 == null) return head2;
    if (head2 == null) return head1;

    Node head = head1;
    Node current1 = head1;
    Node current2 = head2;

    while (current1 != null && current2 != null) {
        Node next1 = current1.next;
        Node next2 = current2.next;

        current1.next = current2;

        if (next1 == null) {
            current2.next = null;


            break;
        }

        current2.next = next1;

        current1 = next1;
        current2 = next2;
    }

    return head;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        
    Node prev = null;
    Node current = head1;
    while (current != null) {
        Node next = current.next;
        current.next = prev;
        prev = current;
        current = next;
    }
    Node reversedHead1 = prev;
    Node tail = reversedHead1;
    while (tail.next != null) {
        tail = tail.next;
    }

    Node n2 = head2;
    Node n3 = head3;
    Node sumHead = null;
    Node sumTail = null;
    while (n2 != null && n3 != null) {
        int val = ((Integer) n2.elem + (Integer) n3.elem) % 10;
        Node newNode = new Node(val);
        if (sumHead == null) {
            sumHead = newNode;
            sumTail = newNode;
        } else {
            sumTail.next = newNode;
            sumTail = newNode;
        }


        n2 = n2.next;
        n3 = n3.next;
    }

    tail.next = sumHead;
    return reversedHead1;
    }

 public static void sumOddAppend(Node dh) {
        int sum = 0;
        Node current = dh;
        while(current.next != dh){          
                if((int)current.next.elem % 2 != 0){
                    sum += (int)current.next.elem;
                    current.next = current.next.next;
                }


            else current = current.next;
        }
        Node sumNode = new Node(sum);
        current.next = sumNode;
        sumNode.next = dh;

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode current1 = dh1.next;
        DNode current2 = dh2.next;
        DNode tail = dh1;

        while(current1 != null && current2 != null){
            DNode temp1 = current1.next;
            DNode temp2 = current2.next;

            tail.next = current1;
            current1.prev = tail;

            current1.next = current2;
            current2.prev = current1;
            tail = current2;

            current1 = temp1;
            current2 = temp2;
        }

        tail.next = dh1;


        dh1.prev = tail;
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode current = dh.next;
        DNode startNode = null;
        DNode endNode = null;

        while(current != dh){
            DNode next = current.next;
            int value = (int) current.elem;
            if(value >= start && value <= end){
                current.prev.next = current.next;
                current.next.prev = current.prev;
                if(startNode == null){
                    startNode = current;
                    endNode = current;
                    current.prev = null;
                    current.next = null;


                } else {
                    endNode.next = current;
                    current.prev = endNode;
                    endNode = current;
                }
            }
            current = next;

        }
        if(startNode != null){
            DNode tail = dh.prev;
            startNode.prev = tail;
            tail.next = startNode;
            endNode.next = dh;
            dh.prev = endNode;
        }
    }
