public static String checkSimilar( Node building1, Node building2 ){
      Node cur1 = building1;
      Node cur2 = building2;
      while(cur1!=null && cur2!=null){
        if(cur1.elem!=cur2.elem){
          return "Not Similar";
        }
        cur1 = cur1.next;
        cur2 = cur2.next;
      }
      if(cur1==null && cur2==null){
        return "Similar";
      }
      else{
        return "Not Similar";
      }
    }

          if(popularity[j]<popularity[j+1]){
            int temp1 = popularity[j];
            popularity[j] = popularity[j+1];
            popularity[j+1] = temp1;
            Object temp2 = current.elem;
            current.elem = newNode.elem;

        }
      }
      return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){


      Node t1 = head1;

public static void sumOddAppend(Node dh) {
      int sum = 0;
      Node current = dh.next;
      Node build = dh;
      while(current!=dh){
        if((Integer)current.elem%2==1){
          sum+=(Integer)current.elem;
          build.next = current.next;
          current = current.next;
        }
        else{
          build = current;
          current = current.next;
        }
      }
      Node newNode = new Node(sum);
      build.next = newNode;
      newNode.next = dh;
      
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode cur1 = dh1.next;
        DNode cur2 = dh2.next;
        while(cur1!=null){
          DNode temp1 = cur1.next;
          DNode temp2 = cur2.next;
          cur1.next = cur2;
          cur2.prev = cur1;
          cur2.next = temp1;
          temp1.prev = cur2;
          cur1 = temp1;
          cur2 = temp2;
        }
        cur2.next = dh1;
        dh1.prev = cur2;
    }

public static void rangeMove(DNode dh, int start, int end) {
        int count = 0;
        DNode current = dh.next;
        while(current!=dh){
          count++;
          current = current.next;
        }
        current = dh.next;
        for(int i=0; i<count; i++){
          DNode temp = current.next;
          if((Integer)current.elem>=start && (Integer)current.elem<=end){
            current.prev.next = current.next;
            current.next.prev = current.prev;
            DNode tail = dh.prev;
            tail.next = current;
            current.prev = tail;
            current.next = dh;
            dh.prev = current;
          }
          current = temp;
        }
    }
