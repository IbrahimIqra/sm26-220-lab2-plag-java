public static void sumOddAppend(Node dh) {

        int sum = 0;


        Node temp_2 = temp1.next;
        while (temp_2 != dh) {
            if((int) temp_2.elem % 2 != 0) {
                sum += (int) temp_2.elem;
                temp1.next = temp_2.next;
                temp_2 = temp_2.next;

            } else {
                temp1 = temp_2;
                temp_2 = temp_2.next;

        temp1.next = newNode;
        newNode.next = dh;
