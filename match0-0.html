 public static void sumOddAppend(Node dh) {
        Node prev= dh;
        Node cur= dh.next;
        Node NN= new Node(0);
        while (cur!=dh) {
            Node nextNode= cur.next;
            if ((int)cur.elem%2== 0) {
                prev.next= cur;
                prev= cur;
                cur= nextNode;}
            else {
                int x= (int)NN.elem;
                x+= (int) cur.elem;
                NN.elem= x;
                cur= nextNode;

 public static void pairJoin(DNode dh1, DNode dh2) {

        DNode temp2 = dh2.next;
        int count = 0;
        while (temp1 !=null) {
            if (count%2== 0) {
                DNode curr= temp1.next;
                temp1.next= temp2;
                temp2.prev= temp1;
                temp1= curr;
            }
            else {
                DNode curr= temp2.next;
                temp2.next= temp1;
                temp1.prev= temp2;
                temp2= curr;}
            count++;
        }
        temp2.next= dh1;
        dh1.prev= temp2;

    }

 public static void rangeMove(DNode dh, int start, int end) {
        DNode prev= dh;
        DNode cur= dh.next;
        DNode accPrev= dh.prev;
        while (prev!= accPrev) {
            DNode nextNode= cur.next;
            int x=(int)cur.elem;
            if (start<=x&&x<=end) {
                DNode temp= cur;
                prev.next= nextNode;
                nextNode.prev= prev;
                cur= nextNode;
                temp.next= dh;
                temp.prev= dh.prev;
                dh.prev.next= temp;
                dh.prev= temp;}
            else {
                prev= cur;
                cur= nextNode;}
        }
    }
