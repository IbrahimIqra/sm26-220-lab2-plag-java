public static void sumOddAppend(Node dh) {
    Node pre = dh;
    Node current = dh.next;
    int total = 0;
    while(current != dh){
        if((Integer)current.element % 2!= 0){

            total += (Integer)current.element;
            pre.next = current.next;
            current = current.next;
        }
        else{
            pre = current;
            current = current.next;
        }
    }
    Node newNode = new Node(total,null);
    pre.next = newNode;
    newNode.next = dh;
    }
