            DNode next1 = temp1.next;
            DNode next2 = temp2.next;


         temp1.next = temp2;
         temp2.prev = temp1;


         temp2.next = next1;
 
         if(next1 != null){
             next1.prev = temp2;
         }

         temp1 = next1;
         temp2 = next2;


        }
        dh1.prev = last;
        last.next = dh1;


    }
