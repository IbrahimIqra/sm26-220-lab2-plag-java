public static void pairJoin(DNode dh1, DNode dh2) {
        // TO DO
        DNode temp1 = dh1.next;
        DNode temp2 = dh2.next;
        while(temp1!= null && temp2!= null){
            DNode next1 = temp1.next;
            DNode next2 = temp2.next;
            temp1.next = temp2;
            temp2.prev = temp1;
            temp2.next = next1;
            if(next1 != null){
                next1.prev = temp2;
            }
            temp1 = next1;
            temp2 = next2;
        }
        DNode temp = dh1;
        while(temp.next != null){
            temp = temp.next;
        }
        dh1.prev = temp;
        temp.next = dh1;


    }
