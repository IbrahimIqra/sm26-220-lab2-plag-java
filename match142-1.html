public static String checkSimilar( Node building1, Node building2 ){
        
        //You're not suppose to create any new Linked List for this task
        
        //TODO


        

    Node t1 = building1;
    Node t2 = building2;

    while (t1 != null && t2 != null) {

        if (!t1.elem.equals(t2.elem)) {
            return "Not Similar";
        }

        t1 = t1.next;
        t2 = t2.next;
    }

    if (t1 == null && t2 == null) {
        return "Similar";
    }

    return "Not Similar";
}

 public static void sumOddAppend(Node dh) {
        // TO DO


    Node previous= dh;
    Node current = dh.next;


    int s = 0;

   
    while (current != dh) {

        if ((int) current.elem % 2 != 0) {
            s += (int) current.elem;

          
            previous.next = current.next;
            current = previous.next;
        }
        else {
            previous = current;
            current = current.next;
        }
    }

   
    Node newnode = new Node(s);

    
    previous.next = newnode;
    newnode.next = dh;
}

        lastpart = lastpart.next;
    }

public static void rangeMove(DNode dh, int start, int end) {

    DNode reallast= dh.prev;
    DNode curr = dh.next;

    while (true) {

        DNode next = curr.next;

        if ((int) curr.elem >= start && (int) curr.elem <= end && curr != dh.prev) {

            
            curr.prev.next = curr.next;
            curr.next.prev = curr.prev;

            
            DNode last = dh.prev;
            last.next = curr;
            curr.prev = last;

            curr.next = dh;
            dh.prev = curr;
        }

        
        if (curr == reallast) {
            break;
        }

        curr = next;
    }
}
