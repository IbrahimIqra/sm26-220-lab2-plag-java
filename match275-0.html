public static String checkSimilar(Node building1, Node building2){    
        Node n1=building1, n2=building2;    


        while(n1!=null && n2!=null){
            if(!n1.elem.equals(n2.elem)){
                return "Not Similar";
            }        
            n1=n1.next;
            n2=n2.next;
        }
        if(n1==null && n2==null){
            return "Similar";
        }
        else{
            return "Not Similar";
        }
    }

public static Node organizeBooks(Node head, Integer popularity[]){   


    if(head==null || head.next==null){
        return head;
    }   
    int books=popularity.length;   
    for(int i=0;i<books-1;i++){   
        Node current=head;        
        for(int j=0;j<books-i-1;j++){          
            if(popularity[j]<popularity[j+1]){
                int temp=popularity[j];
                popularity[j]=popularity[j+1];
                popularity[j+1]=temp;


                Object book=current.elem;
                current.elem=current.next.elem;
                current.next.elem=book;
            }           
            current=current.next;
        }
    }   
    return head;
}

public static Node alternateMerge(Node head1, Node head2){
    if(head1==null){
        return head2;
    }
    if(head2==null){
        return head1;
    }
    Node current1=head1;
    Node current2=head2;


    while(current1!=null && current2!=null){
        Node next1=current1.next;  
        Node next2=current2.next;  
        current1.next=current2;       
        if(next1==null){
            break;
        }
        current2.next=next1;       
        current1=next1;            
        current2=next2;            
    }
    return head1;
}

public static void sumOddAppend(Node dh){
    if(dh==null || dh.next==dh){
        return;
    }
    Node previous=dh, current=dh.next;
    int sum=0;
    while(current!=dh){


        int value=(Integer)current.elem;
        if((value%2)!=0){
            sum+=value;
            previous.next=current.next;
            current=current.next;
        } 
        else{
            previous=current;
            current=current.next;
        }
    }
    Node sumNode=new Node(sum);


    previous.next=sumNode;
    sumNode.next=dh;
}

public static void pairJoin(DNode dh1, DNode dh2){
    DNode current1=dh1.next, current2=dh2.next, tail=dh1;
    while(current1!=null && current2!=null){
        DNode temp1=current1.next, temp2=current2.next;
        current1.next=current2;
        current2.prev=current1;
        if(temp1!=null){
            current2.next=temp1;
            temp1.prev=current2;
        }
        current1=temp1;
        current2=temp2;
    }
    while (tail.next!=null){


        tail=tail.next;
    }
    tail.next=dh1;
    dh1.prev=tail;
}

public static void rangeMove(DNode dh, int start, int end){
    if(dh.next==dh){
        return;
    }
    boolean finish=false;
    DNode original_tail=dh.prev, current=dh.next, current_tail=dh.prev;
    while(!finish){
        if(current==original_tail){
            finish=true;
        }


        DNode nextNode=current.next;
        int value=(int)current.elem;
        if(value>=start && value<=end){
            if(current!=current_tail){
                current.prev.next=current.next;
                current.next.prev=current.prev;
                current.next=dh;
                current.prev=current_tail;
                current_tail.next=current;
                dh.prev=current;
                current_tail=current;
            } 
            else{
                current_tail=current;
            }
        }
        current=nextNode;
    }
}
