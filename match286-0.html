public static String checkSimilar( Node building1, Node building2 ){
        Node t1 = building1;
        Node t2 = building2;


        while (t1 != null && t2 != null){
          if (!t1.elem.equals(t2.elem)){
            return "Not Similar";
          }
            t1 = t1.next;
            t2 = t2.next;
          }
        if (t1 == null && t2 == null){
        return "Similar";
        }
        return "Not Similar";
    }

 public static Node organizeBooks(Node head, Integer[] popularity){
    if (head == null || head.next == null)
        return head;


    Node cur = head;
    for (int i = 0; i < popularity.length - 1; i++){
        int maxIndex = i;
        Node maxNode = cur;
        Node tempNode = cur.next;
        for (int j = i + 1; j < popularity.length; j++){
            if (popularity[j] > popularity[maxIndex]){
                maxIndex = j;
                maxNode = tempNode;
            }
            tempNode = tempNode.next;
        }
        if (maxIndex != i){
            int temp = popularity[i];
            popularity[i] = popularity[maxIndex];
            popularity[maxIndex] = temp;
            Object tempBook = cur.elem;
            cur.elem = maxNode.elem;
            maxNode.elem = tempBook;
        }
        cur = cur.next;
    }
    return head;
}

 public static Node alternateMerge( Node head1, Node head2 ){
        Node t1 = head1;
        Node t2 = head2;
        while (t1 != null && t2 != null) {
            Node next1 = t1.next;
            Node next2 = t2.next;
            t1.next = t2;
            t2.next = next1;
            t1 = next1;
            t2 = next2;
        }
        return head1;
    }

 public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node newHead = null;
    Node temp = head1;
    while (temp != null) {


        newHead = new Node(temp.elem, newHead);
        temp = temp.next;
    }
    Node tail = newHead;
    while (tail.next != null) {
        tail = tail.next;
    }
    Node t2 = head2;
    Node t3 = head3;
    while (t2 != null && t3 != null) {
        int sum = (Integer)t2.elem + (Integer)t3.elem;
        if (sum >= 10) {
            sum = sum % 10;
        }
        tail.next = new Node(sum, null);
        tail = tail.next;
        t2 = t2.next;
        t3 = t3.next;
    }
    return newHead;
    }

 public static void sumOddAppend(Node dh){    
    Node prev = dh;
    Node cur = dh.next;
    int sum = 0;
    while (cur != dh) {
        if ((Integer) cur.elem % 2 != 0){


            sum += (Integer) cur.elem;
            prev.next = cur.next;
            cur = cur.next;
        }else{
            prev = cur;
            cur = cur.next;
        }
    }
    Node tail = dh;
    while (tail.next != dh){
        tail = tail.next;
    }
    Node newNode = new Node(sum);
    tail.next = newNode;
    newNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2){
    DNode p1 = dh1.next;
    DNode p2 = dh2.next;
    while (p1 != null && p2 != null){
        DNode next1 = p1.next;
        DNode next2 = p2.next;
        p1.next = p2;
        p2.prev = p1;
        if (next1 != null){
            p2.next = next1;
            next1.prev = p2;
        }else{
            p2.next = null;
        }
        p1 = next1;
        p2 = next2;
    }
    DNode tail = dh1;
    while (tail.next != null) {
        tail = tail.next;
    }
    tail.next = dh1;
    dh1.prev = tail;
}

 public static void rangeMove(DNode dh, int start, int end){
    DNode tail = dh.prev;
    DNode cur = dh.next;
    while (cur != dh){
        DNode next = cur.next;
        int value = (Integer) cur.elem;
        if (value >= start && value <= end){
            if (cur == tail){
                cur = next;
            }


            cur.prev.next = cur.next;
            cur.next.prev = cur.prev;
            tail.next = cur;
            cur.prev = tail;
            cur.next = dh;
            dh.prev = cur;
            tail = cur;
        }
        cur = next;
    }
}
