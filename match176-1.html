public static void rangeMove(DNode dh, int start, int end) 
{
        // TO DO
        if(dh.next == dh)
        {
          return;
        }

         DNode originalTail = dh.prev;
          DNode tail = dh.prev;
           DNode current = dh.next;

         while(true)
         {
           DNode next = current.next;

           int valuw = (Integer) current.elem;



           if(valuw >= start && valuw <= end)
           {
            
             current.prev.next = current.next;
              current.next.prev = current.prev;
               tail.next = current;
               current.prev = tail;
              current.next = dh;
             dh.prev = current;
             tail = current;
           }

           if(current == originalTail)
           {
              break;
           }

             current = next;
         }
}
