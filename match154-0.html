public static String checkSimilar( Node building1, Node building2 ) {
   Node temp1 = building1;
   Node temp2 = building2;
   int count1=0; int count2=0;


   while(temp1!=null && temp2!=null) {
       if (!temp1.elem.equals(temp2.elem)) {
           return "Not similar";
       }
       temp1 = temp1.next;
       temp2 = temp2.next;
   }
   while(temp1!=null){
       count1++;
       temp1=temp1.next;}
   while(temp2!=null){
       count2++;
       temp2=temp2.next;}
   if(count1!=count2){
       return " Not Similar";
   }
   return "Similar";


}
