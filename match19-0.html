        public static String checkSimilar(Node building1,Node building2){
        Node b1 = building1;
        Node b2 = building2;
        while (b1 != null && b2 != null) {
            if(b1.elem!=b2.elem) {
                return "Not Similar";
            }

            b2 = b2.next;

    public static Node organizeBooks(Node head, Integer[] popularity) {
        int n = popularity.length;

        for(int i=0; i<n-1; i++){
            Node current = head;
            Node nextNode = current.next;
            for(int j=0; j<n-i-1; j++){


                if(popularity[j] < popularity[j+1]){
                    int currentPopularity = popularity[j];
                    popularity[j] = popularity[j+1];
                    popularity[j+1] = currentPopularity;

                    Object temp = current.elem;
                    current.elem = nextNode.elem;
                    nextNode.elem = temp;
                }
                current = current.next;
                nextNode = current.next;
            }
        }
        return head;
    }

    public static Node alternateMerge( Node head1, Node head2 ){
        if (head1 == null) return head2;
        if (head2 == null) return head1;
    


        Node temp1 = head1;
        Node temp2 = head2;
        while(temp1!=null && temp2!=null){
            Node front = temp1.next;
            temp1.next = temp2;
            Node front2 = temp2.next;
            temp2.next = front;
            temp1 = front;
            temp2 = front2;
        }
        return head1;
    }

public void pairJoin(Node head1, Node head2) {
        Node current1 = head1.next;
        Node current2 = head2.next;
        while (current1 != null) {
            Node next1 = current1.next;
            Node next2 = current2.next;
          
            current1.next = current2;
            current2.prev = current1;
          
            current2.next = next1;
            next1.prev = current2;
          
            current1 = next1;
            current2 = next2;
        }
        current2.next = head1;
        head1.prev = current2;
    }

public void rangeMove(Node head, int start, int end) {
        int size = 0;
        Node counter = head.next;
        while (counter != head) {
            size++;
            counter = counter.next;
        }



        Node current = head.next;
        for (int i = 0; i < size; i++) {
            Node nextNodeToProcess = current.next;
            if (current.data >= start && current.data <= end) {


                current.prev.next = current.next;
                current.next.prev = current.prev;

                Node tail = head.prev;
                tail.next = current;
                current.prev = tail;
                current.next = head;
                head.prev = current;
            }
            current = nextNodeToProcess;
        }
}
