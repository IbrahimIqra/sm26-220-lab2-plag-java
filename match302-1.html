public static String checkSimilar( Node building1, Node building2 ){
    Node a=building1;
    Node b=building2;
    while(a!=null && b!=null){
        if(!a.elem.equals(b.elem)) return "Not Similar";
        a=a.next;
        b=b.next;
    }
    if(a!=null || b!=null) return "Not Similar";

    return "Similar";
}

public static Node organizeBooks(Node head, Integer[] popularity) {
    int n=0;
    Node t=head;
    while(t!=null){n++; t=t.next;}
    for(int i=0;i<n-1;i++){
        Node c=head;
        int k=0;
        for(int j=0;j<n-1-i;j++){
            Node x=c.next;
            if(popularity[k]<popularity[k+1]){
                Object e=c.elem; c.elem=x.elem; x.elem=e;
                Integer p=popularity[k]; popularity[k]=popularity[k+1]; popularity[k+1]=p;
            }
            c=c.next;
            k++;
        }
    }
    return head;
}

public static Node alternateMerge( Node head1, Node head2 ){
    Node a=head1;
    Node b=head2;
    while(a!=null && b!=null){
        Node x=a.next;
        Node y=b.next;
        a.next=b;


        if(x!=null) b.next=x;
        a=x;
        b=y;
    }
    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {
    Node p=null;
    Node c=head1;
    while(c!=null){
        Node x=c.next;
        c.next=p;
        p=c;
        c=x;
    }


    Node h=p;
    Node t=h;
    while(t.next!=null) t=t.next;
    Node a=head2;
    Node b=head3;
    while(a!=null && b!=null){
        int s=((Integer)a.elem+(Integer)b.elem)%10;
        t.next=new Node(s);
        t=t.next;


        a=a.next;
        b=b.next;
    }
    return h;
}

public static void sumOddAppend(Node dh) {
    int s=0;
    Node p=dh;
    Node c=dh.next;
    while(c!=dh){
        int v=(Integer)c.elem;
        if(v%2!=0){
            s+=v;
            p.next=c.next;
            c=c.next;
        }else{
            p=c;
            c=c.next;
        }
    }
    Node n=new Node(s);
    p.next=n;
    n.next=dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode a=dh1.next;
    DNode b=dh2.next;
    DNode t=dh1;
    while(a!=null && b!=null){
        DNode x=a.next;
        DNode y=b.next;


        t.next=a; a.prev=t; t=a;
        t.next=b; b.prev=t; t=b;
        a=x;
        b=y;
    }
    t.next=dh1;
    dh1.prev=t;
}

public static void rangeMove(DNode dh, int start, int end) {
    int n=0;
    DNode t=dh.next;
    while(t!=dh){n++; t=t.next;}
    DNode c=dh.next;
    for(int i=0;i<n;i++){
        DNode x=c.next;


        int v=(Integer)c.elem;
        if(v>=start && v<=end){
            c.prev.next=c.next;
            c.next.prev=c.prev;
            DNode l=dh.prev;
            l.next=c; c.prev=l;
            c.next=dh; dh.prev=c;
        }
        c=x;
    }
}
