public static String checkSimilar( Node building1, Node building2 ){
  Node a= building1;
  Node b = building2;
  int c = 1;
  while(a != null && b != null) {
    if(!a.elem.equals(b.elem)) {
         c = 0;
    break;
  }
    a = a.next;
    b = b.next;
  }
  if(a != null || b != null) {
        c = 0;
  }
    if(c == 1) {
        return "Similar";
  } else {
         return "Not Similar";
}

public static void sumOddAppend(Node dh) {
        int sum = 0;
        Node current= dh;
        
        while (current.next != dh) {
            int val = (Integer) current.next.elem;
            
            if (val % 2 !=0) {  sum = sum +val;
                current.next = current.next.next;
            } else {

                current=current.next;
            }
        }
        
        Node newNode = new Node(sum);
        current.next = newNode;
        newNode.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode c1=dh1.next;
        DNode c2=dh2.next;
        DNode t= dh1;
        
        while(c1 != null&& c2!= null) {
            DNode n1=c1.next;


            DNode n2=c2.next;
            
            t.next = c1;


            c1.prev=t;
            
            c1.next = c2;
            c2.prev = c1;
            
            t = c2;
            
            c1 = n1;
            c2 = n2;
        }
        
        t.next = dh1;
        dh1.prev =t;
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode c = dh.next;
        DNode stop = dh.prev;
        boolean flag = false;
        
        while (c != dh && !flag) {
            if (c == stop) {
                flag= true;
            }
            


            DNode n = c.next;
            int v = (Integer) c.elem;
            
            if (v >= start && v<= end) {
                c.prev.next = c.next;
                c.next.prev = c.prev;
                
                DNode t = dh.prev;
                t.next = c;
                c.prev = t;
                c.next = dh;
                dh.prev = c;
            }
            
            c = n;
        }
    }
