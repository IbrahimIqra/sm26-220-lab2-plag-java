public static String checkSimilar(Node building1, Node building2) {
        //You're not suppose to create any new Linked List for this task
        //TODO
        Node curr1 = building1;
        Node curr2 = building2;

        while (curr1 != null && curr2 != null) {
            if (curr1.elem != curr2.elem) {


                return "Not Similar";
            }
            curr1 = curr1.next;
            curr2 = curr2.next;
        }

    public static Node organizeBooks(Node head, Integer[] popularity) {

      
        int n = popularity.length;
        boolean isswapped = true;

        while (isswapped) {
            isswapped = false;
            Node curr = head;
            int i = 0;

            while (curr != null && curr.next != null && i < n - 1) {

                if (popularity[i] < popularity[i + 1]) {

                    int tempP = popularity[i];
                    popularity[i] = popularity[i + 1];
                    popularity[i + 1] = tempP;
                    Object tempE = curr.elem;
                    curr.elem = curr.next.elem;


                    curr.next.elem = tempE;
                    isswapped = true;
                }
                curr = curr.next;
                i++;
            }
        }

        return head;
        
    }

   public static Node alternateMerge(Node head1, Node head2) {

        if (head1 == null) {
            return head2;
        }
        if (head2 == null) {
            return head1;
        }
        Node curr1 = head1;
        Node curr2 = head2;


        while (curr1 != null && curr2 != null) {
            Node next1 = curr1.next;
            Node next2 = curr2.next;
            curr1.next = curr2;
            if (next1 != null) {
                curr2.next = next1;
            }
            curr1 = next1;
            curr2 = next2;
        }

        return head1;

   public static Node ReverseLL(Node head) {
        Node tail = head;
        Node prev = null;
        while (tail != null) {
            Node nxt = tail.next;
            tail.next = prev;
            prev = tail;
            tail = nxt;
        }
        return prev;
    }

    public static void rangeMove(DNode dh, int start, int end) {
      
        DNode curr = dh.next;
        DNode movedHead = null;
        DNode movedTail = null;

        while (curr != dh) {


            DNode nextNode = curr.next;
            int val = (int) curr.elem;
            if (val >= start && val <= end) {
                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;
                if (movedHead == null) {
                    movedHead = curr;
                    movedTail = curr;
                } else {
                    movedTail.next = curr;
                    curr.prev = movedTail;
                    movedTail = curr;
                }
            }

            curr = nextNode;
        }
        if (movedHead != null) {
            DNode tail = dh.prev;
            tail.next = movedHead;
            movedHead.prev = tail;
            movedTail.next = dh;
            dh.prev = movedTail;
        }
    }
