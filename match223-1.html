public static String checkSimilar(Node head1, Node head2){

    Node temp1 = head1;
    Node temp2 = head2;

    while(temp1 != null && temp2 != null){

        if(!temp1.elem.equals(temp2.elem)){
            return "Not Similar";
        }

        temp1 = temp1.next;
        temp2 = temp2.next;
    }

    if(temp1 == null && temp2 == null){
        return "Similar";   
 }

    return "Not Similar"; 
}

public static Node organizeBooks(Node head, Integer[] popularity) {

  
    int c = 0;
    Node temp = head;

    while (temp != null) {
        c++;


        temp = temp.next;
    }

    
    for (int i = 0; i < c - 1; i++) {

        Node current = head;

        for (int j = 0; j < c - i - 1; j++) {


            if (popularity[j] < popularity[j + 1]) {
                int p = popularity[j];
                popularity[j] = popularity[j + 1];
                popularity[j + 1] = p;

                Object book = current.elem;
                current.elem = current.next.elem;
                current.next.elem = book;
            }

            current = current.next;
        }
    }

    return head;
}

public static Node alternateMerge(Node head1, Node head2){

    Node newHead = head1;   


    while(head1 != null && head2 != null){

        Node temp1 = head1.next;
        Node temp2 = head2.next;

        head1.next = head2;

        if(temp1 == null){
            break;
        }

        head2.next = temp1;

        head1 = temp1;
        head2 = temp2;
    }

    return newHead;
}

public static void rangeMove(DNode dh, int start, int end) {

    DNode last = dh.prev;
    DNode tail = last;
    DNode curr = dh.next;

    while (true) {

        DNode next = curr.next;

        int value = (Integer) curr.elem;

        if (value >= start  && value <= end) {

                       curr.prev.next = curr.next;
            curr.next.prev = curr.prev;

            
            tail.next = curr;
            curr.prev = tail;

            curr.next = dh;
            dh.prev = curr;

            tail = curr;
        }

        if (curr == last) {
            break;
        }

        curr = next;
    }
}
