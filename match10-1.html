public static String checkSimilar( Node building1, Node building2 ){
        Node b1 = building1 , b2 = building2;


        while(b1!=null && b2!=null){
             if(b1.elem!=b2.elem)
                return "Not Similar";
                b1 = b1.next;
                b2 = b2.next;
        }
         if(b1==null && b2==null)
            return "Similar";
         else
            return "Not Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {
        int n = popularity.length;


        for (int i = 0; i < n; i++) {
            Node current = head;
            Node nextNode = current.next;
            for(int j=0;j<n-i-1;j++){
                if(popularity[j]<popularity[j+1]){
                   int currentPopularity = popularity[j];
                   popularity[j] = popularity[j+1];
                   popularity[j+1] = currentPopularity;
                   Object temp = current.elem;
                   current.elem = nextNode.elem;
                   nextNode.elem = temp;
                }
                current = current.next;
                nextNode = nextNode.next;
            }


        }
        return head;
    }

            Node front = curr.next;
            curr.next = prev;
            prev = curr;
            curr = front;
        }
        Node dummy = new Node(-1);
        Node tail = dummy;
        Node temp2 = head2;
        Node temp3 = head3;
        while(temp2 != null && temp3!= null){
            Node newNode = new Node ((((int)temp2.elem) + ((int)temp3.elem))%10);


            tail.next = newNode;
            tail = newNode;
            temp2 = temp2.next;
            temp3 = temp3.next;
        }
        Node current = prev;

            current = current.next;
        }
        current.next = dummy.next;
