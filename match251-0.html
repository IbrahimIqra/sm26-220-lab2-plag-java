public static String checkSimilar( Node case1, Node case2 ) {

        Node pointer2 = case2;

            pointer2 = pointer2.next;
        }

        }

public static Node alternateMerge( Node head1, Node head2 ) {
        if(head1 == null || head2 == null){


            if ( head1 == null ) {
              return head2;
          }
          if ( head2 == null ) {
              return head1;
          }
        }
        Node present1 = head1;
        Node present2 = head2;
        while ( present1 != null && present2 != null ) {
            Node upcoming1 = present1.next;
            Node upcoming2 = present2.next;
            present1.next = present2;
            if ( upcoming1 == null ) {
                break;
            }
            present2.next = upcoming1;
            present1 = upcoming1;
            present2 = upcoming2;
        }
        return head1; 
    }
