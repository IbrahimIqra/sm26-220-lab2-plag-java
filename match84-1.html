    public static String checkSimilar(Node building1, Node building2){
        int cnt1 = countElement(building1);
        int cnt2 = countElement(building2);

        if(cnt1 != cnt2){
            return "Not Similar";
        }

        for(int i=0; i<cnt1; i++){
            if( NodeAt(building1,i).elem != NodeAt(building2,i).elem ){
                return "Not Similar";
            }
        }
     return "Similar";
    }

 public static Node alternateMerge(Node head1,Node head2){
      Node tmp1 = head1;
      Node tmp2 = head2;

      while(tmp1!=null && tmp2!=null){
        Node nxt1 = tmp1.next;
        Node nxt2 = tmp2.next;

        tmp1.next = tmp2;
        if(nxt1==null){
            break;
        }

        tmp2.next = nxt1;
        tmp1 = nxt1;
        tmp2 = nxt2;
      }
      return head1;
    }

          temp = temp.next;

     temp.next = newNode;
      return head;

  public static void sumOddAppend(Node dh){
        Node prevNode = dh;
        Node currNode = dh.next;
        int summ = 0;

        while(currNode != dh){
          int val = (int)currNode.elem;
          
          if(val%2 != 0){  
             summ = summ+val;


             prevNode.next = currNode.next;
             currNode = currNode.next;
          }else{
             prevNode = currNode;
             currNode = currNode.next;
          }
        }

        Node notunNode = new Node(summ);


        prevNode.next = notunNode;
        notunNode.next = dh;
    }

    public static void rangeMove(DNode dh, int start, int end){
        


        DNode tail=dh.prev;
        int cnt =0;
        DNode tmp=dh.next;
        
        while (tmp!=dh){
           cnt++;
           tmp=tmp.next;
        }

        tmp=dh.next;

        while(cnt>0){
           DNode nxtN =tmp.next;
           int val=(int)tmp.elem;

           if(val<=end && val>=start){
               if(tmp!= tail){

                   tmp.prev.next = tmp.next;
                   tmp.next.prev = tmp.prev;
                   tmp.next=dh;
                
                   tmp.prev=tail;
                   tail.next=tmp;
                
                   dh.prev=tmp;
                   tail=tmp;
               }
            }
          
           tmp=nxtN ;
           cnt-- ;
        }
    }
