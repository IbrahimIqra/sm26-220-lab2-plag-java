 public static Node alternateMerge ( Node head1, Node head2 ){

    public static void sumOddAppend(Node dh) {
        // TO DO
        int sum=0;
        Node prev = dh;


        Node temp = dh.next;
        while (temp != dh ) { 
            if((int)temp.elem%2!=0){
                sum +=(int)temp.elem;
                prev.next = temp.next;
                temp = temp.next;
            }
            else{
                prev = temp;
                temp = temp.next;
            }
        }
        Node SumNode = new Node(sum);
        prev.next = SumNode;
        SumNode.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        // TO DO
        DNode temp1 = dh1.next;
        DNode temp2 = dh2.next;
        while (temp1.next != null) {
            DNode var1 = temp1.next;
            DNode var2 = temp2.next;
            temp1.next = temp2;


            temp2.prev = temp1;
            temp2.next = var1;
            var1.prev = temp2;
            temp1 = var1;
            temp2 = var2;
        }
        temp1.next = temp2;
        temp2.prev = temp1;
        temp2.next = dh1;
        dh1.prev = temp2;
    }

public static void rangeMove(DNode dh, int start, int end) {
        // TO DO
        DNode temp = dh.next;
        DNode last = dh.prev;
        while (temp != last) {
            DNode next = temp.next;
            if((int)temp.elem>= start && (int)temp.elem<=end){
                temp.prev.next = next;
                next.prev = temp.prev;            
                dh.prev.next = temp;
                temp.prev = dh.prev;
                temp.next = dh;
                dh.prev = temp;
            }
            temp = next;
        }
    }
