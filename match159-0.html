 public static String checkSimilar( Node building1, Node building2 ){
      Node t1 = building1;
      Node t2 = building2;
      while(t1!=null && t2!=null){
        if(!t1.equals(t2)){
          return "Not similar";
        }
        t1= t1.next;
        t2= t2.next;
      }
      if(t1 == null && t2 == null){
          return "Similar";
        }
      return "Not similar";
      }

 public static Node organizeBooks(Node head, Integer[] popularity) {
      if(head == null || head.next == null){
        return head;
      }


      int len = popularity.length;
      for(int i =0; i < len-1; i++){
        Node present = head;
        for(int j = 0; j<len-i-1; j++){ //condition for bubble sort
          if(popularity[j] < popularity[j+1]){
            int temp = popularity[j];
            popularity[j] = popularity[j+1]; //swapping the popularity ratings
            popularity[j+1] = temp;
            Object temp2 = present.elem;
            present.elem = present.next.elem; //swapping the book names 
            present.next.elem = temp2;
          }
          present = present.next;
        }
      }
      return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){
      if(head1 == null){ //empty list 1 will return list 2's head
        return head2;
      }
      if(head2 == null){ //empty list 2 will return list 1's head
        return head1;
      }
      Node temp1 = head1;
      Node temp2 = head2;


      while(temp1 != null && temp2 != null){
        Node n1 = temp1.next;
        Node n2 = temp2.next;
        temp1.next = temp2;
        if(n1==null){
          break; // breaking the loop because if the list 1 ends 
        }
        temp2.next = n1; //connecting list 2 to list 1
        temp1 = n1;
        temp2 = n2;
      }       
        return head1;
    }

 public static Node idGenerator(Node head1, Node head2, Node head3) {
       Node prev = null;
       Node current = head1;
       while(current != null){
         Node next = current.next;
         current.next = prev;
         prev = current;
         current = next;
       }
       Node newHead = prev;


       Node tail = newHead;
       while( tail.next != null){
         tail = tail.next;
       }
       Node temp2 = head2;
       Node temp3 = head3;

       }

    }

 public static void sumOddAppend(Node dh) {
        Node prev = dh;
        Node current = dh.next;
        int sum = 0;
        while(current != dh){
          if((int)current.elem % 2 != 0){ // condition for all odd element
            sum = sum + (int)current.elem;


            prev.next = current.next;
            current = current.next;
          }
          else{
            prev = current;
            current = current.next;
          }  
        }
        Node newNode = new Node (sum);
        Node tail = dh;
        while(tail.next != dh){
          tail = tail.next;
        }
        tail.next = newNode; //appending 
        newNode.next = dh;
    }

 public static void pairJoin(DNode dh1, DNode dh2) {
        DNode temp1 = dh1.next;
        DNode temp2 = dh2.next;
        while( temp1 != null){
          DNode next1 = temp1.next;
          DNode next2 = temp2.next;
          temp1.next = temp2;
          temp2.prev = temp1;
          temp2.next = next1;
          if(next1 != null){


            next1.prev = temp2;
          }
          temp1 = next1;
          temp2 = next2;
        }
        DNode tail = dh1;
        while(tail.next != null){
          tail = tail.next;
        }
        tail.next = dh1; // making it circular linked list 
        dh1.prev = tail;
        dh1.next.prev = dh1;
    }

 public static void rangeMove(DNode dh, int start, int end) {
        DNode mainTail = dh.prev;
        DNode tail = dh.prev;
        DNode current = dh.next;
        while(current != dh){
          DNode next = current.next;


          int value = (int)current.elem;
          if(value >= start && value <= end){
            current.prev.next = current.next;
            current.next.prev = current.prev;
            current.prev = tail;
            current.next = dh;
            tail.next = current;
            dh.prev = current;
            tail = current;
          }
          if(current == mainTail){
            break;
          }
          current = next;
        }
    }
