public static String checkSimilar( Node building1, Node building2 ){
        
        Node build1 = building1;
        Node build2 = building2;

        while(build1!=null && build2!=null){
            if(!build1.elem.equals(build2.elem)){
                return "Not Similar";
            }
            build1 = build1.next;
            build2 = build2.next;
        }
        if(build1!=null || build2!=null){
                return "Not Similar";
        }


        return "Similar";
 }

 public static Node organizeBooks(Node head, Integer[] popularity) {
        
       int l = popularity.length;
       for(int i=0; i<l; i++){
        Node curr = head;
        Node after = curr.next; 

        for(int j=0; j<l-i-1; j++){
           if(popularity[j] < popularity[j+1]){
            int temp = popularity[j];
            popularity[j] = popularity[j+1];
            popularity[j+1] = temp;
            Object val = curr.elem;
            curr.elem = after.elem;
            after.elem = val;
           } 
           curr = curr.next;
           after = after.next;
        }
       }
        return head; 
    }

 public static Node alternateMerge( Node head1, Node head2 ){
        
        Node curr1 = head1; 
        Node curr2 = head2;
        
        while(curr1!=null && curr2!=null){
            Node temp1 = curr1.next;
            Node temp2 = curr2.next;
            curr1.next = curr2;


            curr2.next = temp1;
            curr1 = temp1;
            curr2 = temp2;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        
        Node dummy = null;
        Node curr = head1;
        while(curr != null){
            Node temp = curr.next;
            curr.next = dummy;
            dummy = curr;
            curr = temp;
        }

        Node temp1 = dummy;
        while(temp1.next != null){
           temp1 = temp1.next;
        }

        while(head2!=null && head3!=null){
          Node nNode = new Node(((int)head2.elem+(int)head3.elem)%10);


          temp1.next = nNode;
          temp1 = nNode;
          head2 = head2.next;
          head3 = head3.next;
        }

        return dummy;
    }

 public static void sumOddAppend(Node dh) {
        Node temp = dh;
        Node curr = temp.next;
        int sum = 0;

        while(curr != dh){
            
            if((Integer)curr.elem%2 != 0){
                sum += (Integer)curr.elem;
                temp.next = curr.next;   
                curr = curr.next;
            }
            else{
                temp = curr;
                curr = curr.next;             
            }
        }
        Node nSum = new Node(sum);
        temp.next = nSum;
        nSum.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
         DNode curr1 = dh1.next;
         DNode curr2 = dh2.next;

        while(curr1!=null && curr2!=null){ 
            DNode x1 = curr1.next;


            DNode x2 = curr2.next; 
            curr1.next = curr2;
            curr2.prev = curr1;
            curr2.next = x1;
            
            if(x1 != null){
             x1.prev = curr2;   
            }
            else{
                curr2.next = dh1;
                dh1.prev = curr2;
            }
            curr1 = x1;
            curr2 = x2;
        }
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode curr = dh.next;
        int count = 0;

        while(curr != dh){
            count++;
            curr = curr.next;
        }
        curr = dh.next; 
        for(int i=0; i<count; i++){
            DNode temp = curr.next; 

            if(start<=(Integer)curr.elem && (Integer)curr.elem<=end){
                curr.prev.next = temp;
                temp.prev = curr.prev;
                DNode last = dh.prev;
                last.next = curr;
                curr.prev = last;
                dh.prev = curr;
                curr.next = dh;
            }
            curr = temp;
        }
    }
