public static Node alternateMerge( Node head1, Node head2 ){
        
       Node temp1 = head1;
       Node temp2 = head2;

    while(temp1 != null && temp2 != null){

        Node next1 = temp1.next;
        Node next2 = temp2.next;

          temp1.next = temp2;


          temp2.next = next1;

           temp1 = next1;
           temp2 = next2;
       }

       return head1;
    }

    public static Node idGenerator(Node head1, Node head2, Node head3) {
         
         Node curr = head1;

         while(curr != null){

             Node next = curr.next;
            
             if(curr == head1){
                curr.next = null;
             }
             else{
                curr.next = head1;
                head1 = curr;
             }
             curr = next;
         }

        Node tail = head1;

        while(tail.next != null){


               tail = tail.next;
        }

        while (head2 != null && head3 != null) {

          int sum = (Integer)head2.elem+(Integer)head3.elem;

           if (sum >= 10){
            sum = sum % 10;
        }

         tail.next = new Node(sum);
         tail = tail.next;



          head2 = head2.next;
          head3 = head3.next;
     }
    return head1;
}

    public static void sumOddAppend(Node dh) {

     Node prev = dh;
     Node curr = dh.next;

     int sum = 0;

     while(curr != dh){

        if((int)curr.elem %2 != 0){
            
            sum += (int)curr.elem;

            prev.next = curr.next;
            curr = curr.next;
        }
        else{
            prev = curr;
            curr = curr.next;
        }
     }
       Node sumNode = new Node(sum);

       sumNode.next = dh;
       prev.next = sumNode;      
 }

    public static void rangeMove(DNode dh, int start, int end) {
        
            DNode curr = dh.next;
            DNode tail = dh.prev;

            while(curr != tail.next){



                DNode next = curr.next;
                int val = (int)curr.elem;

                if( val >= start && val <= end){

                    DNode prev = curr.prev;
                    DNode next2 = curr.next;

                        prev.next = next2;
                        next2.prev = prev;
                        curr.next = dh;
                        curr.prev = tail;
                        tail.next = curr;
                        dh.prev = curr;

                        tail = curr;
                }
                    curr = next;
            }
    }
