public static String checkSimilar( Node building1, Node building2 ){
        Node a=building1;
        Node b=building2;
        while(a!=null && b!=null){
            if(a.elem!=b.elem){


                return "Not similar";
            }
            a=a.next;
            b=b.next;
        }
        if(a==null && b==null){
            return "similar";
        }
        else{
            return "Not similar";
        }
    }

// AssignmentTask2: Organize Books

 public static Node organizeBooks(Node head, Integer[] popularity) {
        if (head == null) {
            return null;
        }
        Node NN = head;
        int a= 0;
        while (NN != null) {
            Node temp = NN.next;
            int b = a + 1;
            while (temp != null) {
                if (popularity[b] > popularity[a]) {
                    Object book = NN.elem;
                    NN.elem = temp.elem;


                    temp.elem = book;

                    Integer c = popularity[a];
                    popularity[a] = popularity[b];
                    popularity[b] = c;

                }
                temp = temp.next;
                b++;
            }
            NN = NN.next;
            a++;
        }


        return head;
    }

// AssignmentTask3: Alternate Merge

 public static Node alternateMerge( Node head1, Node head2 ){
        if (head1 == null) {
            return head2;
        }
        if (head2 == null) {
            return head1;
        }
        Node Current1 = head1;
        Node Current2 = head2;
        while (Current1 != null && Current2 != null) {
            Node NN1 = Current1.next;
            Node NN2 = Current2.next;
            Current1.next = Current2;
            if (NN1 == null) {
                break;
            }
            Current2.next = NN1;
            Current1 = NN1;
            Current2 = NN2;
        }
        return head1;
    }

// AssignmentTask4: ID Generator

public static Node idGenerator(Node head1, Node head2, Node head3) {

        if (head1 == null || head2 == null || head3 == null) {
            return null;
        }


        Node n1 = head1;
        Node prev = null;
        while (n1 != null) {
            Node NN = n1.next;
            n1.next = prev;
            prev = n1;
            n1 = NN;
        }
        Node Head = prev;

        Node tail = Head;
        while (tail.next != null) {
            tail = tail.next;
        }
        Node n2 = head2;
        Node n3 = head3;
        while (n2 != null && n3 != null) {
            int sum = (int) n2.elem + (int) n3.elem;
            if (sum >= 10) {
                sum = sum % 10;
            }
            n2.elem = sum;
            tail.next = n2;
            tail = tail.next;

            n2 = n2.next;
            n3 = n3.next;
        }

        tail.next = null;


        return Head;
    }

public static void sumOddAppend(Node dh) {
        Node prev = dh;
        Node curr = dh.next;
        int sum = 0;
        while (curr != dh) {
            int value = (int) curr.elem;
            if (value % 2 != 0) {
                sum += value;


                prev.next = curr.next;
            }
            else {
                prev = curr;
            }
            curr = curr.next;
        }
        Node Result = new Node(sum);
        prev.next = Result;
        Result.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode tail = dh1;


        DNode d1 = dh1.next;
        DNode d2 = dh2.next;
        while (d1 != null && d2 != null) {
            DNode tmp1 = d1.next;
            DNode tmp2 = d2.next;
            d1.next = d2;
            d2.prev = d1;
            if (tmp1 != null) {
                d2.next = tmp1;
                tmp1.prev = d2;
            }
            d1 = tmp1;


            d2 = tmp2;
        }
        while (tail.next != null) {
            tail = tail.next;
        }
        tail.next = dh1;
        dh1.prev = tail;
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode Head = dh.next;
        DNode tail = dh.prev;
        while (Head != tail.next) {
            DNode tmp = Head.next;
            int val = (int) Head.elem;
            if (val >= start && val <= end) {
                Head.prev.next = Head.next;
                Head.next.prev = Head.prev;
                DNode tmp2 = dh.prev;
                tmp2.next = Head;
                Head.prev = tmp2;
                Head.next = dh;
                dh.prev = Head;
            }

            Head = tmp;
        }
    }
