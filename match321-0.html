public static String checkSimilar( Node building1, Node building2 ){


        String result="Similar"; 
        Node temp1=building1;
        Node temp2=building2;


        while(temp1!=null && temp2!=null){
            if(!temp1.elem.equals(temp2.elem)){
                result= "Not Similar";
            }
            temp1=temp1.next;
            temp2=temp2.next;
        }


        if (temp1!=null ||temp2!=null) {
            result= "Not Similar";
        }
        return result;
    }

   public static Node alternateMerge( Node head1, Node head2 ){
        
        Node first= head1;
        Node second= head2;


        while(first!=null && second!=null){
            Node nextOfFirst= first.next;
            Node nextOfSecond=second.next;


            first.next=second;


            second.next=nextOfFirst;


            first=nextOfFirst;
            second=nextOfSecond;
        }
        return head1;
    }

public static void sumOddAppend(Node dh) {
        Node prev=dh;
        Node current=dh.next;
        int sum=0;


        while(current!=dh){
            if((int)current.elem%2==1){
                sum+=(int)current.elem;


                prev.next=current.next;
                current=prev.next;
            }
            else{
                prev=current;
                current=current.next;
            }
        }


        Node endNode=dh;
        while(endNode.next!=dh){
            endNode=endNode.next;
        }




        Node newNode= new Node(sum);
        newNode.next=dh;
        endNode.next=newNode;


    }

public static void pairJoin(DNode dh1, DNode dh2) {


        DNode curr1=dh1.next;
        DNode curr2=dh2.next;


        while(curr1!=null && curr2!=null){   //Connecting the 2 diff lists
            DNode next1=curr1.next;
            DNode next2=curr2.next;


            curr1.next=curr2;
            curr2.prev=curr1;


            curr2.next=next1;
            if(next1!=null){
            next1.prev=curr2;
            }
            curr1=next1;
            curr2=next2;
        }


        dh2.next=null;
        dh2.prev=null;


        DNode previous=dh1;
        DNode curr=dh1.next;


        while(curr!=null){
            curr.prev=previous;
            previous=curr;
            curr=curr.next;
        }


        previous.next=dh1;


        dh1.prev=previous;
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode tail=dh.prev;
        DNode checkTail=tail;


        DNode current=dh.next;


        while(current!=checkTail.next){
            DNode next=current.next;




            int value=(int)current.elem;


            if(value>=start && value<=end){
                current.prev.next=current.next;
                current.next.prev=current.prev;


                tail.next=current;
                current.prev=tail;
                current.next=dh;
                dh.prev=current;
                tail=current;
            }
            current=next;
        }
    }
