public static String checkSimilar( Node building1, Node building2 ){
        Node temp1 = building1;
        Node temp2 = building2;
        while (temp1 != null && temp2 != null) {
            if (temp1.elem!=temp2.elem){


                return "Not Similar";
            }
            temp1 = temp1.next;
            temp2 = temp2.next;
        }
        if (temp1 == null && temp2 == null) {
            return "Similar";
        } 
        return "Not Similar";

    }

public static Node organizeBooks(Node head, Integer[] popularity) {
if (head == null||head.next == null) {
            return head;
        }
        boolean swapped = true;
        while (true) {
            swapped = false;
            Node curr1 = head;
            Node prev = null;
            int index = 0;
            while (curr1 != null && curr1.next != null) {


                if (popularity[index] < popularity[index + 1]) {
                    int tempScore = popularity[index];
                    popularity[index] = popularity[index + 1];
                    popularity[index + 1] = tempScore;


                    Node nextN = curr1.next;
                    curr1.next = nextN.next;
                    nextN.next = curr1;
                    if (prev == null) {
                        head = nextN; 
                    } else {
                        prev.next = nextN; 
                    }
                    prev = nextN;
                    swapped = true;
                } else {
                    prev = curr1;
                    curr1 = curr1.next;
                }
                index++;
            }
            if(swapped==false)break;
        } 


        return head; 
       }

public static Node alternateMerge( Node head1, Node head2 ){
            Node curr1 = head1;
        Node curr2 = head2;
        while (curr1 != null && curr2 != null) {
            Node front1 = curr1.next;
            Node front2 = curr2.next;
            curr1.next = curr2;
            if (front1 == null) {
                break;
            }
            curr2.next = front1;
            curr1 = front1;
            curr2 = front2;
        }
        return head1;
    }

public static void sumOddAppend(Node dh) {
    int sum = 0;
    Node prev = dh;
    Node curr = dh.next;
    while (curr != dh) {
        int val = (Integer) curr.elem; 
        if (val % 2 != 0) {
            sum += val;


            prev.next = curr.next;
            curr = curr.next;
        } else {
            prev = curr;
            curr = curr.next;
        }
    }
    Node sumN = new Node(sum);


    sumN.next=dh;
    prev.next = sumN;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode front1 = dh1.next;
        DNode front2 = dh2.next;
        DNode back = dh1;
        while (front1 != null) {


            DNode next1 = front1.next;
            DNode next2 = front2.next;
            back.next = front1;
            front1.prev = back;
            back = front1;
            back.next = front2;
            front2.prev = back;
            back = front2;
            front1 = next1;
            front2 = next2;
        }
        back.next = dh1;
        dh1.prev = back;
        }
