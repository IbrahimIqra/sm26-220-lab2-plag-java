 public static String checkSimilar( Node building1, Node building2 ){

        Node cur2 = building2;
        while (cur1!=null && cur2!=null) {
            if (!cur1.elem.equals(cur2.elem)){
                return "Not Similar";
            }
            cur1= cur1.next;
            cur2=cur2.next;
