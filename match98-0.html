public class AssignmentTask1 {



    public static String checkSimilar( Node building1, Node building2 ){
        
        Node b1= building1;
        Node b2=building2;
        while(b1!=null && b2!=null){
            if(b1.elem!=b2.elem){


                return "Not Similar";
            }
            b1=b1.next;
            b2=b2.next;
        }
        if(b1==null && b2==null){
            return "Similar";
        }
        return "Not Similar";
    }

public class AssignmentTask2{
    
    public static Node organizeBooks(Node head, Integer[] popularity) {
        
        int n=popularity.length;
        for(int i=0; i<n; i++){
           Node currentNode=head;
           Node nextNode= currentNode.next; 
           for(int j=0; j<n-i-1; j++){
            if(popularity[j]<popularity[j+1]){
                int temp=popularity[j];
                popularity[j]=popularity[j+1];
                popularity[j+1]=temp;

                Object change=currentNode.elem;
                currentNode.elem=nextNode.elem;
                nextNode.elem=change;
            }
            currentNode=currentNode.next;
            nextNode=nextNode.next;
           }
        }
        return head;
    }

public class AssignmentTask5 {
    
    public static void sumOddAppend(Node dh) {
        Node temp=dh;
        Node curr=dh.next;
        int odd_sum=0;
        while(curr!=dh){
            if((int)curr.elem%2!=0){
                odd_sum+=(int)curr.elem;
                curr=curr.next;
                temp.next=curr;
            }
            else{
                curr=curr.next;
                temp=temp.next;
            }
        }
        Node newNode=new Node(odd_sum);
        temp.next=newNode;
        newNode.next=dh;
    }

public class AssignmentTask6 {

    public static void pairJoin(DNode dh1, DNode dh2) {
        DNode firstNode=dh1.next;
        DNode secondNode= dh2.next;
        while(firstNode!=null){
            DNode firstRest=firstNode.next;
            DNode secondRest=secondNode.next;

            firstNode.next=secondNode;
            secondNode.prev=firstNode;

            secondNode.next=firstRest;

            if(firstRest!=null){
                firstRest.prev=secondNode;
            }

            if(firstRest == null){
             secondNode.next = dh1;
             dh1.prev = secondNode;
            }
            
            firstNode=firstRest;
            secondNode=secondRest;

        }

    }

public class AssignmentTask7 {



    public static void rangeMove(DNode dh, int start, int end) {
        int size =0;
        DNode temp=dh.next;
        while(temp!=dh){
            size++;
            temp=temp.next;
        }
        DNode curr=dh.next;
        for(int i=0; i<size; i++){
            DNode nextNode=curr.next;
            if((int)curr.elem>=start && (int)curr.elem<=end){


                curr.prev.next=curr.next;
                curr.next.prev=curr.prev;

                DNode tail=dh.prev;
                tail.next=curr;
                curr.prev=tail;
                curr.next=dh;
                dh.prev=curr;
            }
            curr=nextNode;
        }
    }
