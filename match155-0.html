public static String buildingBlocks(Node buildingOne, Node buildingTwo) {
    Node blockOne = buildingOne;
    Node blockTwo = buildingTwo;
    
    while (blockOne != null && blockTwo != null) {
        if (!blockOne.elem.equals(blockTwo.elem)) {


            return "Not Similar";
        }
        blockOne = blockOne.next;
        blockTwo = blockTwo.next;
    }
    
    if (blockOne == null && blockTwo == null) {
        return "Similar";
    }
    
    return "Not Similar";
}

public static Node organizeBooks(Node head, int[] popularity) {
    if (head == null || head.next == null) {
        return head;
    }
    
    boolean isSorted = false;
    int length = popularity.length;
    
    while (!isSorted) {
        isSorted = true;
        Node currentBook = head;
        
        for (int i = 0; i < length - 1; i++) {


            if (popularity[i] < popularity[i + 1]) {
                int tempPopularity = popularity[i];
                popularity[i] = popularity[i + 1];
                popularity[i + 1] = tempPopularity;
                
                Object tempTitle = currentBook.elem;
                currentBook.elem = currentBook.next.elem;
                currentBook.next.elem = tempTitle;
                
                isSorted = false;
            }
            currentBook = currentBook.next;
        }
        length--;
    }
    


    return head;
}

public static Node alternateMerge(Node headOne, Node headTwo) {
    if (headOne == null) return headTwo;
    if (headTwo == null) return headOne;
    
    Node listOneNode = headOne;
    Node listTwoNode = headTwo;
    
    while (listOneNode != null && listTwoNode != null) {
        Node nextInListOne = listOneNode.next;
        Node nextInListTwo = listTwoNode.next;
        
        listOneNode.next = listTwoNode;
        
        if (nextInListOne == null) {
            break;
        }
        


        listTwoNode.next = nextInListOne;
        
        listOneNode = nextInListOne;
        listTwoNode = nextInListTwo;
    }
    
    return headOne;
}

public static Node idGenerator(Node reversedPart, Node sumPartOne, Node sumPartTwo) {
    Node finalIdHead = null;
    Node currentReversedNode = reversedPart;
    
    while (currentReversedNode != null) {


        Node newDigitNode = new Node(currentReversedNode.elem);
        newDigitNode.next = finalIdHead;
        finalIdHead = newDigitNode;
        currentReversedNode = currentReversedNode.next;
    }
    
    Node tailNode = finalIdHead;
    while (tailNode != null && tailNode.next != null) {
        tailNode = tailNode.next;
    }
    
    Node currentSumOneNode = sumPartOne;
    Node currentSumTwoNode = sumPartTwo;
    
    while (currentSumOneNode != null && currentSumTwoNode != null) {
        int valOne = (Integer) currentSumOneNode.elem;
        int valTwo = (Integer) currentSumTwoNode.elem;
        
        int combinedSum = (valOne + valTwo) % 10;
        Node sumNode = new Node(combinedSum);
        
        if (finalIdHead == null) {
            finalIdHead = sumNode;
            tailNode = sumNode;
        } else {


            tailNode.next = sumNode;
            tailNode = sumNode;
        }
        
        currentSumOneNode = currentSumOneNode.next;
        currentSumTwoNode = currentSumTwoNode.next;
    }
    
    return finalIdHead;
}

public static void sumOddAppend(Node dh) {
    Node previousNode = dh;
    Node currentNode = dh.next;
    int oddSum = 0;

    while (currentNode != dh) {
        int nodeValue = (Integer) currentNode.elem;
        if (nodeValue % 2 != 0) {


            oddSum += nodeValue;
            previousNode.next = currentNode.next;
        } else {
            previousNode = currentNode;
        }
        currentNode = currentNode.next;
    }

    Node newSumNode = new Node(oddSum);
    previousNode.next = newSumNode;
    newSumNode.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode mergedTail = dh1;
    DNode nodeFromList1 = dh1.next;
    DNode nodeFromList2 = dh2.next;

    while (nodeFromList1 != null && nodeFromList2 != null) {
        mergedTail.next = nodeFromList1;
        nodeFromList1.prev = mergedTail;
        mergedTail = nodeFromList1;
        nodeFromList1 = nodeFromList1.next;

        mergedTail.next = nodeFromList2;
        nodeFromList2.prev = mergedTail;
        mergedTail = nodeFromList2;
        nodeFromList2 = nodeFromList2.next;
    }

    while (nodeFromList1 != null) {
        mergedTail.next = nodeFromList1;
        nodeFromList1.prev = mergedTail;
        mergedTail = nodeFromList1;
        nodeFromList1 = nodeFromList1.next;
    }

    while (nodeFromList2 != null) {
        mergedTail.next = nodeFromList2;
        nodeFromList2.prev = mergedTail;
        mergedTail = nodeFromList2;


        nodeFromList2 = nodeFromList2.next;
    }

    mergedTail.next = dh1;
    dh1.prev = mergedTail;
}

public static void rangeMove(DNode dh, int start, int end) {
    DNode extractedListHead = null;
    DNode extractedListTail = null;
    DNode currentNode = dh.next;

    while (currentNode != dh) {
        DNode nextToProcess = currentNode.next;


        int nodeValue = (Integer) currentNode.elem;

        if (nodeValue >= start && nodeValue <= end) {
            currentNode.prev.next = currentNode.next;
            currentNode.next.prev = currentNode.prev;

            if (extractedListHead == null) {
                extractedListHead = currentNode;
                extractedListTail = currentNode;
            } else {
                extractedListTail.next = currentNode;
                currentNode.prev = extractedListTail;
                extractedListTail = currentNode;
            }
        }
        currentNode = nextToProcess;
    }

    if (extractedListHead != null) {
        DNode originalTail = dh.prev;
        
        originalTail.next = extractedListHead;
        extractedListHead.prev = originalTail;
        
        extractedListTail.next = dh;
        dh.prev = extractedListTail;
    }
}
