public static Node alternateMerge( Node head1, Node head2 ){

                Node temp2 = getNode(head2, 0);

        public static Node getNode(Node head, int i) {
            Node temp = head;
            int currInd = 0;
            while (temp != null) {
                if (currInd == i) {
                    return temp;
                }
                temp = temp.next;
                currInd++;
            }
            return null;
        }


        public static Node insertAt(Node head, Node newNode, int i) {
            if (i == 0) {
                newNode.next = head;
                return newNode;
            }
            Node temp = head;
            int currInd = 0;
            while (temp != null && currInd < i - 1) {
                temp = temp.next;
                currInd++;
            }
            if (temp != null) {
                newNode.next = temp.next;


                temp.next = newNode;
            }
            return head;
        }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node rh = null;
        Node temp1 = head1;
        while (temp1 != null) {
            Node newNode = new Node(temp1.elem);
            newNode.next = rh;
            rh = newNode;
            temp1 = temp1.next;
        }
        Node last = rh;
        while (last != null && last.next != null) {
            last = last.next;
        }
        Node temp2 = head2;
        Node temp3 = head3;
        while (temp2 != null && temp3 != null) {
            int sum = (int) temp2.elem + (int) temp3.elem;


            int finalDigit = sum % 10;
            Node newNode = new Node(finalDigit);
            if (last == null) {
                rh = newNode;
                last = newNode;
            } else {
                last.next = newNode;
                last = last.next;
            }


            temp2 = temp2.next;
            temp3 = temp3.next;
        }
        return rh;
    }

public static void pairJoin(DNode dh1, DNode dh2){
       
        DNode temp1 = dh1.next;
        while(temp1!=null&& dh2.next!=null){
            DNode temp2= getNode(dh2,0);
            if(temp2==null){
                break;
            }
            dh2.next = temp2.next;
            if(temp2.next!=null){
                temp2.next.prev = dh2;
            }
            DNode next1 = temp1.next;
            temp1.next = temp2;
            temp2.prev = temp1;
            temp2.next = next1;
            if(next1!=null){
                next1.prev = temp2;
            }
            temp1 = next1;
         }
         DNode last = dh1;
         while(last.next!=null){
            last = last.next;
         }
         last.next = dh1;
         dh1.prev = last;
        }
