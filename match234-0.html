public static String checkSimilar( Node building1, Node building2 ){
        Node t=building1;
        Node t1=building2;
        while(t!=null && t1!=null){
            if(!t.elem.equals(t1.elem)) return "Not Similar";
            t=t.next;
            t1=t1.next;
        }
        if(t==null && t1==null) return "Similar";
        return "Not Similar";
    }

 public static Node organizeBooks(Node head, Integer[] popularity) {
   Node temp = head;

    for (int i = 0; i < popularity.length - 1; i++) {

        int max = i;

        for (int j = i + 1; j < popularity.length; j++) {
            if (popularity[j] > popularity[max]) {
                max = j;
            }
        }
        
        Integer x = popularity[i];
        popularity[i] = popularity[max];
        popularity[max] = x;

        Node t = temp;
        for (int j = i; j < max; j++) {
            t = t.next;
        }

        Object book = temp.elem;
        temp.elem = t.elem;


        t.elem = book;
        temp = temp.next;
    }
    return head;
}

  public static Node alternateMerge( Node head1, Node head2 ){
         if (head1 == null) return head2;
    if (head2 == null) return head1;

    Node t1 = head1;
    Node t2 = head2;

    while (t1 != null && t2 != null) {
        Node next1 = t1.next;
        Node next2 = t2.next;
        t1.next = t2;

        if (next1 == null)  break;
        
        t2.next = next1;
        t1 = next1;
        t2 = next2;
    }

    return head1;
}

public static Node idGenerator(Node head1, Node head2, Node head3) {


    Node prev = null;
    Node curr = head1;
    while (curr != null) {
        Node next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }
    if (prev == null) {
        while (head2 != null && head3 != null) {
            Node n = new Node((int) head2.elem + (int) head3.elem);

            if (prev == null) {
                prev = n;
                head1 = n;
            } else {
                Node tail = prev;
                while (tail.next != null) {
                    tail = tail.next;
                }
                tail.next = n;
            }

            head2 = head2.next;
            head3 = head3.next;
        }
        return head1;
    }

    Node tail = prev;
    while (tail.next != null) {
        tail = tail.next;
    }

    while (head2 != null && head3 != null) {
        int sum = ((int) head2.elem + (int) head3.elem)%10;


        tail.next = new Node(sum);
        tail = tail.next;
        head2 = head2.next;
        head3 = head3.next;
    }

    return prev;
}

public static void sumOddAppend(Node dh) {
    Node prev = dh;
    Node curr = dh.next;
    int sum = 0;

    while (curr != dh) {

        if ((int) curr.elem % 2 != 0) {
            sum += (int) curr.elem;


            prev.next = curr.next;
            curr = prev.next;
        } else {
            prev = curr;
            curr = curr.next;
        }
    }

    Node n = new Node(sum);


    prev.next = n;
    n.next = dh;
}

public static void pairJoin(DNode dh1, DNode dh2) {
    DNode t1 = dh1.next;
    DNode t2 = dh2.next;

    while (t1 != null && t2 != null) {

        DNode next1 = t1.next;
        DNode next2 = t2.next;

        t1.next = t2;
        t2.prev = t1;

        if (next1 != null) {
            t2.next = next1;
            next1.prev = t2;
        } else {
            t2.next = null;
        }

        t1 = next1;
        t2 = next2;
    }

    DNode tail = dh1;
    while (tail.next != null) {
        tail = tail.next;
    }
    tail.next = dh1;
    dh1.prev = tail;
}

public static void rangeMove(DNode dh, int start, int end) {
     DNode tail = dh.prev;
    DNode last = tail;
    DNode curr = dh.next;

    while (curr != dh) {
        DNode next = curr.next;
        if ((int) curr.elem >= start && (int) curr.elem <= end) {
            if (curr == tail) {
                tail = tail.prev;
            }



            curr.prev.next = curr.next;
            curr.next.prev = curr.prev;

            tail.next = curr;
            curr.prev = tail;
            curr.next = dh;
            dh.prev = curr;

            tail = curr;
        }

        if (curr == last)
            break;

        curr = next;
    }
}
