public static String checkSimilar( Node building1, Node building2 ){
        Node block1=building1;
        Node block2=building2;
        while (block1!=null && block2!=null){
            if(block1.elem!=block2.elem){
                return "Not Similar";
            }
            block1=block1.next;
            block2=block2.next;
        }
        if(block1==null && block2==null){
            return "Similar";
        }
        else{
            return "Not Similar";
        }
    }

public static Node organizeBooks(Node head, Integer[] popularity) {
        for(int i=0;i<popularity.length-1;i++){
            Node curr=head;
            for(int j=0;j<popularity.length-1-i;j++){
                Node newnode=curr.next;


                if(popularity[j]<popularity[j+1]){
                    int a=popularity[j];
                    popularity[j]=popularity[j+1];
                    popularity[j+1]=a;
                    Object b=curr.elem;
                    curr.elem=newnode.elem;
                    newnode.elem=b;
                }
                curr=curr.next;
            }
        }
        return head; 
    }

    public static Node alternateMerge( Node head1, Node head2 ){


        Node a=head1;
        Node b=head2;
        while(a!=null && b!=null){
            Node prev1=a.next;
            a.next=b;
            Node prev2=b.next;
            b.next=prev1;
            a=prev1;
            b=prev2;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
      Node curr1 = head1;
      Node prev = null;
      while(curr1!=null){
        Node ahead = curr1.next;
        curr1.next = prev;
        prev = curr1;
        curr1 = ahead;
      }
      Node var = new Node(null);
      Node temp = var;
      Node curr2 = head2;
      Node curr3 = head3;
      while(curr2!=null && curr3!=null){
        Node newNode = new Node(((Integer)curr2.elem+(Integer)curr3.elem)%10);
        temp.next = newNode;
        temp = newNode;
        curr2 = curr2.next;
        curr3 = curr3.next;
      }
      Node current = prev;
      while(current.next!=null){
        current = current.next;
      }
      current.next = var.next;
      return prev;
    }

public static void sumOddAppend(Node dh) {
      int sum = 0;
      Node curr = dh.next;
      Node create = dh;
      while(curr!=dh){
        if((Integer)curr.elem%2==1){
          sum+=(Integer)curr.elem;
          create.next = curr.next;
          curr = curr.next;
        }
        else{
          create = curr;
          curr = curr.next;
        }
      }
      Node newNode = new Node(sum);
      create.next = newNode;
      newNode.next = dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode curr1 = dh1.next;
        DNode curr2 = dh2.next;
        while(curr1!=null){
          DNode temp1 = curr1.next;
          DNode temp2 = curr2.next;
          curr1.next = curr2;
          curr2.prev = curr1;
          cur2.next = temp1;
          temp1.prev = curr2;
          curr1 = temp1;
          curr2 = temp2;
        }
        curr2.next = dh1;
        dh1.prev = curr2;
    }

public static void rangeMove(DNode dh, int start, int end) {
        int count = 0;
        DNode curr = dh.next;
        while(curr!=dh){
          count++;
          curr = curr.next;
        }
        curr = dh.next;
        for(int i=0; i<count; i++){
          DNode temp = curr.next;
          if((Integer)curr.elem>=start && (Integer)curr.elem<=end){
            curr.prev.next = curr.next;
            curr.next.prev = curr.prev;
            DNode tail = dh.prev;
            tail.next = curr;
            curr.prev = tail;
            curr.next = dh;
            dh.prev = curr;
          }
          curr = temp;
        }
    }
