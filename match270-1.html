 public static void pairJoin(DNode dh1, DNode dh2) {

            DNode next1 = pr1.next;

            tail.next = pr1;
            pr1.prev = tail;
            tail = pr1;

            DNode next2 = pr2.next;

            tail.next = pr2;
            pr2.prev = tail;
            tail = pr2;

            pr2 = next2;
        }
    }

    tail.next = dh1;


    dh1.prev = tail;
}
