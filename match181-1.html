public static String checkSimilar( Node building1, Node building2 ){


       Node temp1=building1;
       Node temp2=building2;
       while(temp1!=null&&temp2!=null){
           if(!temp1.elem.equals(temp2.elem)){
               return "Not Similar";
           }
           temp1=temp1.next;
           temp2=temp2.next;
       }
       if(temp1!=null||temp2!=null){
           return "Not Similar";
       }
       return "Similar";
   }

       Node temp1=head;
       int i=0;
       while(temp1!=null){
           Node temp2=temp1.next;
           int j=i+1;
           while(temp2!=null){
               if(popularity[i]<popularity[j]){
                   Object x=temp1.elem;
                   temp1.elem=temp2.elem;
                   temp2.elem=x;


                   int p=popularity[i];
                   popularity[i]=popularity[j];
                   popularity[j]=p;


               }
               temp2=temp2.next;
               j++;
           }
           temp1=temp1.next;


           i++;
       }
       return head;


       }

public static Node alternateMerge(Node head1, Node head2) {


   Node temp1 = head1;
   Node temp2 = head2;


   while (temp1 != null && temp2 != null) {


       Node nn1 = temp1.next;
       Node nn2 = temp2.next;


       temp1.next = temp2;
       temp2.next = nn1;


       temp1 = nn1;
       temp2 = nn2;
   }


   return head1;
}

public static void sumOddAppend(Node dh) {
     
   int sum = 0;


   Node temp = dh;
   Node curr = dh.next;




   while (curr != dh) {
       if ((Integer) curr.elem % 2 != 0) {
           sum += (Integer) curr.elem;
           temp.next = curr.next;
           curr = temp.next;
       } else {
           temp = curr;


           curr = curr.next;
       }
   }


   Node sumNode = new Node(sum);
   sumNode.next = dh;
   temp.next = sumNode;
}

public static void pairJoin(DNode dh1, DNode dh2) {
   DNode curr1 = dh1.next;
   DNode curr2 = dh2.next;


   while (curr1 != null && curr2 != null) {


       DNode temp1 = curr1.next;
       DNode temp2 = curr2.next;


       curr1.next = curr2;
       curr2.prev = curr1;
       curr2.next = temp1;


       if (temp1 != null) {


       temp1.prev = curr2;
       }


       curr1 = temp1;
       curr2 = temp2;
   }


  
   DNode tail = dh1;
   while (tail.next != null) {
   tail = tail.next;
   }
   tail.next = dh1;
   dh1.prev = tail;
}

public static void rangeMove(DNode dh, int start, int end) {


   DNode curr = dh.next;
   int size = 0;
   DNode temp = dh.next;


   while (temp != dh) {
       size++;
       temp = temp.next;
   }


   for (int i = 0; i < size; i++) {
       DNode NN = curr.next;


       if ((Integer) curr.elem >= start && (Integer) curr.elem <= end) {
           curr.prev.next = curr.next;
           curr.next.prev = curr.prev;
           DNode tail = dh.prev;
           tail.next = curr;
           curr.prev = tail;
           curr.next = dh;
           dh.prev = curr;
       }
       curr = NN;
   }
}
