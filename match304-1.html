public static String checkSimilar( Node building1, Node building2 ){
        Node h1=building1;
        Node h2=building2;
        // if(h1==null && h2==null){
        //     return "Similar";
        // }
        while(h1!=null && h2!=null){
            if(!h1.elem.equals(h2.elem)){


                return "Not Similar";
            }
            h1=h1.next;
            h2=h2.next;
      }
      if(h1!=null || h2!=null){
        return "Not Similar";
      }
      return "Similar";
    }

    public static Node organizeBooks(Node head, Integer[] popularity) {
        Node curr=head;
        if(head==null){
            return null;
        }
        int count=0;
        while(curr!=null){
            curr=curr.next;
            count++;
        }
        for(int j=0;j<count-1;j++){
            int i=0;
            curr=head;
            while(curr.next!=null){
                if(popularity[i]<popularity[i+1]){
                    Object temp=curr.elem;
                    curr.elem=curr.next.elem;
                    curr.next.elem=temp;

                    int xchng=popularity[i];
                    popularity[i]=popularity[i+1];


                    popularity[i+1]=xchng;
                }
                curr=curr.next;
                i++;
            }
        }
        return head;
    }

public static Node alternateMerge( Node head1, Node head2 ){
        Node h1=head1;
        Node h2=head2;
        if(head1==null){
            return head1;
        }
        if(head2==null){
            return head2;
        }
        while(h1!=null  && h2!=null){


            Node n1=h1.next;
            Node n2=h2.next;

            h1.next=h2;
            h2.next=n1;

            h1=n1;
            h2=n2;
        }
        if(h2!=null){


            h1.next=h2;
        }
        return head1;
    }

public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node dh= null;
        Node curr=head1;
        while(curr!=null){
            Node temp=new Node(curr.elem);
            temp.next=dh;
            dh=temp;
            curr=curr.next;    
        }
        Node newNode=dh;
        while(newNode.next!=null){
            newNode=newNode.next;
        }
        Node h2=head2;
        Node h3=head3;
        //int sum=0;
        while(h2!=null && h3!=null){
            int sum=(Integer)h2.elem+(Integer)h3.elem;
            if(sum>=10){
                sum=sum%10;
            }
            newNode.next=new Node (sum);


            newNode=newNode.next;

            h2=h2.next;
            h3=h3.next;
        }
        return dh;
    }

public static void sumOddAppend(Node dh) {
        Node curr=dh;
        int sum=0;
        while(curr.next!=dh){
            if((int)curr.next.elem%2!=0){
                sum+=(int)curr.next.elem;
                curr.next=curr.next.next;
            }else{
            curr=curr.next;
            }
        }


        Node res=new Node(sum);
        curr.next=res;
        res.next=dh;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode curr=dh1.next;
        DNode n2=dh2.next;
        DNode lenju=dh1;
        while (curr!=null && n2!=null){
            DNode t1=curr.next;
            DNode t2=n2.next;
            
            lenju.next=curr;


            curr.prev=lenju;
            curr.next=n2;
            n2.prev=curr;
            lenju=n2;
            curr=t1;
            n2=t2;
        }
        lenju.next=dh1;
        dh1.prev=lenju;
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode curr=dh.next;
        DNode lenju=dh.prev;
        DNode lastN=lenju;
        while(true){


            DNode temp=curr.next;
            if((int)curr.elem>=start && (int)curr.elem<=end){
                curr.prev.next=curr.next;
                curr.next.prev=curr.prev;

                lenju.next=curr;
                curr.prev=lenju;
                curr.next=dh;
                dh.prev=curr;
                lenju=curr;
            }
            if(curr==lastN){
                break;
            }
            curr=temp;
    }
