public static String checkSimilar( Node building1, Node building2 ){
      
       Node curr1 = building1;
       Node curr2 = building2;
      


       while (curr1 != null && curr2 != null) {
           if (!curr1.elem.equals(curr2.elem)) {
               return "Not Similar";
           }
           curr1 = curr1.next;
           curr2 = curr2.next;
       }
      
       if (curr1 != null || curr2 != null) {
           return "Not Similar";
       }
      
       return "Similar";
   }

public static Node organizeBooks(Node head, Integer[] popularity) {


       if (head == null || head.next == null) {
           return head;
       }
      


       int n = popularity.length;
       for (int i = 0; i < n - 1; i++) {
           Node curr = head;
           for (int j=0; j < n-i-1; j++) {
               if (popularity[j] < popularity[j + 1]) {
                   int tempp = popularity[j];
                   popularity[j] = popularity[j + 1];
                   popularity[j +1] = tempp;
                  
                   Object tempbook = curr.elem;
                   curr.elem = curr.next.elem;
                   curr.next.elem = tempbook;
               }
               curr = curr.next;
           }
       }
       return head;
   }

public static Node alternateMerge( Node head1, Node head2 ){
      
       if (head1 == null) {
          return head2;
       }
       if (head2 == null) {
          return head1;
       }
       Node curr1 = head1;
       Node curr2 = head2;
       while (curr1 != null && curr2 != null) {
           Node next1 = curr1.next;
           Node next2 = curr2.next;
           curr1.next = curr2;
           if (next1 == null) {
               break;
           }
           curr2.next = next1;
           curr1 = next1;
           curr2 = next2;
       }


       return head1;
   }

public static void sumOddAppend(Node dh) {
       int sum = 0;
       Node prev = dh;
       Node curr = dh.next;


       while (curr != dh && curr != null) {


           int val = (Integer) curr.elem;
          
           if (val % 2 != 0) {
               sum += val;
               prev.next = curr.next;
           } else {
               prev = curr;
           }
           curr = curr.next;
       }


       Node sumn = new Node(sum);
      
       sumn.next = dh;
       prev.next = sumn;
   }

public static void pairJoin(DNode dh1, DNode dh2) {
       DNode curr1 = dh1.next;
       DNode curr2 = dh2.next;
       DNode tail = dh1;


       while (curr1 != null && curr1 != dh1 && curr2 != null && curr2 != dh2) {
           DNode next1 = curr1.next;
           DNode next2 = curr2.next;


           tail.next = curr1;


           curr1.prev = tail;
           curr1.next = curr2;


           curr2.prev = curr1;
           tail = curr2;
           curr1 = next1;
           curr2 = next2;
       }


       tail.next = dh1;
       dh1.prev = tail;
   }

public static void rangeMove(DNode dh, int start, int end) {
       DNode curr = dh.next;
       DNode rangehead = null;
       DNode rangetail = null;
       DNode tailmain = dh;


       while (curr != dh && curr != null) {
           DNode nextNode = curr.next;
           int val = (Integer) curr.elem;


           if (val >= start && val <= end) {
               if (rangehead == null) {
                   rangehead = curr;
                   rangetail = curr;
               } else {
                   rangetail.next = curr;
                   curr.prev = rangetail;
                   rangetail = curr;
               }
           } else {
               tailmain.next = curr;
               curr.prev = tailmain;
               tailmain = curr;
           }
           curr = nextNode;
       }


       if (rangehead != null) {
           tailmain.next = rangehead;
           rangehead.prev = tailmain;
          
           rangetail.next = dh;
           dh.prev = rangetail;
       } else {
           tailmain.next = dh;
           dh.prev = tailmain;
       }
   }
