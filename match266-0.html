public static void pairJoin(DNode dh1, DNode dh2) {
        DNode cur1 = dh1.next;
        DNode cur2 = dh2.next;


        while (cur1 != null && cur2 != null) {
            DNode tmp1 = cur1.next;
            DNode tmp2 = cur2.next;
            if(tmp1 != null && tmp2 != null) {
                cur1.next = cur2;
                cur2.prev = cur1;
                cur2.next = tmp1;
                tmp1.prev=cur2;
            }
            else {
                cur1.next = cur2;
                cur2.prev = cur1;
                cur2.next = dh1;
                dh1.prev = cur2;
            }
            cur1 = tmp1;
            cur2 = tmp2;
        }
    }
