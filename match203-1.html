public static String checkSimilar( Node building1, Node building2 ){
        Node temp1 = building1;
        Node temp2 = building2;
        
        while(temp1 != null && temp2 != null){
            if (!temp1.elem.equals(temp2.elem)) {
                return " Not Similar";
            }
            temp1 = temp1.next;
            temp2 = temp2.next;
        }
        if(temp1 !=null || temp2 != null){
            return "Not Similar";
        }
        return "Similar";
    }

public static Node organizeBooks(Node head, Integer[] popularity) {
        if (head == null || head.next == null){
            return head;
        }
        
        int n = popularity.length;

        for(int i =0;i<n;i++){
            Node temp = head;
            for(int j=0;j<n-1;j++){
                if(popularity[j] < popularity[j+1]){
                    int p = popularity[j];


                    popularity[j] = popularity[j+1];
                    popularity[j+1] = p;

                    Object k = temp.elem;
                    temp.elem = temp.next.elem;
                    temp.next.elem =k;
                }
                temp = temp.next;
            } 
        }

        return head; 
    }

public static Node alternateMerge( Node head1, Node head2 ){
        if (head1 == null){
            return head2;
        }
        if (head2 ==  null){
            return head1;
        }
        Node temp1 = head1;
        Node temp2 = head2;

        while(temp1 != null && temp2 != null){
            Node next1 = temp1.next;
            Node next2 = temp2.next;

            temp1.next = temp2;
    
            if (next1 == null) {
                break;
            }
            temp2.next = next1;
            temp1 = next1;
            temp2 = next2;
        }

        return head1;
    }

    public static Node reverseCopy(Node head){
        if(head ==null){
            return null; 
        }
        Node temp = reverseCopy(head.next);
        Node newNode = new Node(head.elem);
        if(temp == null){
            return newNode;
        }
        Node t = temp;
        while (t.next!=null) {
            t =t.next;
        }
        t.next=newNode;
        return temp;
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode a1 = dh1.next;
        DNode a2 = dh2.next;
        while (a1 != null && a2 != null) {
            DNode next1 = a1.next;
            DNode next2 = a2.next;
            a1.next =a2;
            a2.prev =a1;
            a2.next =next1;
            if (next1 != null) {
                next1.prev = a2;
            }
            a1 = next1;
            a2 = next2;
        }
        DNode tail = dh1;
        while (tail.next !=null) {
            tail = tail.next;   
        }
        tail.next = dh1;
        dh1.prev = tail;
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode tail = dh.prev;
        DNode curr = dh.next;
        while (curr!= dh) {


            DNode next = curr.next;
            int value =(Integer)curr.elem;
            if(value >= start && value <= end && curr !=tail){
                curr.prev.next = curr.next;
                curr.next.prev = curr.prev;
                tail.next=curr;
                curr.prev=tail;
                curr.next=dh;
                dh.prev=curr;
                tail = curr;
            }
            curr = next;
        }
    }
