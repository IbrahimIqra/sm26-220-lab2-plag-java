  public static String checkSimilar( Node building1, Node building2 ){
        
       Node b1 = building1; 
       Node b2 = building2; 

       while(b1!= null && b2!= null){
        String first = (String)b1.elem;
         String second = (String)b2.elem;


         if(!first.equals(second)){
            return "Not Similiar" ;
        
         }
         b1 = b1.next;
         b2=b2.next;
        }
         
        if(b1 != null || b2!=null){
            return "Not similar" ; 
        }

       

        return "Similar";
    }

 public static Node alternateMerge( Node head1, Node head2 ){
        Node first_temp = head1;
        Node sec_temp = head2;

        while(first_temp!= null && sec_temp != null){

            Node first_next = first_temp.next;
            Node sec_next = sec_temp.next;

            first_temp.next = sec_temp;


            sec_temp.next= first_next;

            first_temp = first_next;
        sec_temp = sec_next;

        }


      return head1; 
    }

    public static void sumOddAppend(Node dh) {
        
        Node prev = dh;
        Node now = dh.next;

    int sum =0;
    
    

    while(now != dh){


        int current = (int)now.elem; 


        if(current % 2!=0){
            sum+=current;
            prev.next= now.next;
            now= prev.next; 

        }
        else{
            prev = now;


            now = now.next;
        }
    }
    Node node1 = new Node(sum);
    prev.next = node1;
    node1.next= dh;

    }
