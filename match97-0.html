 public static String checkSimilar( Node building1, Node building2 ){
        Node cur1=building1;
        Node cur2=building2;
        
        while( cur1!=null && cur2!=null){
            String val1=String.valueOf(cur1.elem);
            String val2=String.valueOf(cur2.elem);



            if(!val1.equals(val2)){
                return "Not Similar";
            }
            cur1=cur1.next;
            cur2=cur2.next;
            
        }
        if(cur1!=null || cur2!=null){
            return "Not Similar";
        }
        
        
        return "Similar";
    }
