public static String checkSimilar(Node building1,Node building2){
Node temp1=building1;
Node temp2=building2;
while(temp1!=null&&temp2!=null){
if(!temp1.element.equals(temp2.element)){
return "Not Similar";
}
temp1=temp1.next;
temp2=temp2.next;
}
if(temp1==null&&temp2==null){
return "Similar";
}
return "Not Similar";
}

public static void sumOddAppend(Node head){
Node curr1=head.next;
Node prev1=head;
int oddSum=0;
while(curr1!=head){
if((Integer)curr1.elem%2!=0){
oddSum+=(Integer)curr1.elem;


prev1.next=curr1.next;
curr1=curr1.next;
}
else{
prev1=curr1;
curr1=curr1.next;
}
}

prev1.next=newNode;
newNode.next=head;
}
