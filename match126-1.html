   public static String checkSimilar( Node building1, Node building2 ){

            return "Not Similar";

public static Node organizeBooks(Node head, Integer[] popularity) {

    Node curr= head;
    int i= 0;

    while(curr!= null){

        Node maxNode = curr;
        int maxIndx = i;

        Node temp = curr.next;
        int j= i+1;


        while(temp!=null){
            if (popularity[j] > popularity[maxIndx]){
                maxNode= temp;
                maxIndx= j;
             }

            temp= temp.next;
            j++;
            }

        if(maxIndx!= i){

            String book= curr.elem;


            curr.elem= maxNode.elem;
            maxNode.elem= book;

            int score= popularity[i];
            popularity[i]= popularity[maxIndx];
            popularity[maxIndx]= score;
        }

        curr= curr.next;
        i++;
    }

    return head;
}

  public static Node alternateMerge( Node head1, Node head2 ){
        
    Node t1 = head1;
    Node t2 = head2;

    while(t1!=null && t2!=null){

        Node nxt1= t1.next;
        Node nxt2= t2.next;

        t1.next= t2;


        t2.next= nxt1;
        t1= nxt1;
        t2= nxt2;
    }

    return head1;
}
