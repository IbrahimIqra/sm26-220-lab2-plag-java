    public static String checkSimilar( Node building1, Node building2 ){
        Node temp1=building1;
        Node temp2=building2;


        while(temp1!=null&& temp2!=null){
            if(!temp1.elem.equals(temp2.elem)){
                return "Not Similar";
            }
            temp1=temp1.next;
            temp2=temp2.next;
        }
            if(temp1==null && temp2==null){


                return "Similar";
            }
        return "Not Similar";
    }

    public static Node alternateMerge( Node head1, Node head2 ){
        Node temp1=head1;
        Node temp2=head2;
        while(temp1!=null&&temp2!=null){
           Node n1=temp1.next;
           Node n2=temp2.next;
           temp1.next=temp2;
           temp2.next=n1;
           temp1=n1;
           temp2=n2;
        }
        return head1;
     }

   public static Node idGenerator(Node head1, Node head2, Node head3) {
        Node temp1=head1;
        Node temp2=head2;
        Node temp3=head3;
        if(head1 == null){
          return null;
        }
        Node revhead=new Node(temp1.elem);
        Node curr=temp1.next;
        //reversing 1st LL
        while(curr!=null){
            Node newNode=new Node(curr.elem);
            newNode.next=revhead;
            revhead=newNode;


            curr=curr.next;
        }
        while(temp2!=null&&temp3!=null){  //merging sums
            int sum=(Integer)temp2.elem+(Integer)temp3.elem;
            if(sum>=10){
                sum=sum%10;
            }
            Node tail=revhead;
            while(tail.next!=null){
                tail=tail.next;
            }
            tail.next=new Node(sum); 


            temp2=temp2.next;
            temp3=temp3.next;

            }
        return revhead ; 
        }

  public static void sumOddAppend(Node dh) {
        Node temp=dh;
        Node current=dh.next;
        int sum=0;
        while(current!=dh){
                Node nn=current.next; //a new node
                if(((Integer)current.elem)%2!=0){  //As elem is an object
                    sum+=(Integer)current.elem;
                    temp.next=nn;
                  }
                else{
                    temp=current;
                }
                current=nn;
             }


                Node sum_node=new Node(sum);
                temp.next=sum_node;
                sum_node.next=dh;
            }

 public static void pairJoin(DNode dh1, DNode dh2) {
           DNode temp1=dh1.next;
           DNode temp2=dh2.next;
           while(temp1!=null && temp2!=null ){
            DNode n1=temp1.next;
            DNode n2=temp2.next;

           temp1.next=temp2;
           temp2.prev=temp1;
           temp2.next=n1;
           
           if(n1!=null){
            n1.prev=temp2;
           }


           temp1=n1;
           temp2=n2;
          }
           DNode end=dh1;
           while(end.next!=null){
            end=end.next;
           }
           end.next=dh1;
           dh1.prev=end;
    }

public static void rangeMove(DNode dh, int start, int end) {
    DNode tail=dh.prev;
    DNode real_tail=tail;
    DNode curr=dh.next;
    while(curr!=real_tail){
       DNode  nn=curr.next;
        int elem=(Integer)curr.elem;
        DNode pred=curr.prev;
        DNode  succ=curr.next;
        //removing
        if(elem>=start&&elem<=end){ 


         pred.next=succ;       
         succ.prev=pred;

         tail.next=curr; //joining at tail
         curr.prev=tail;
         curr.next=dh;
         dh.prev=curr;
         tail=curr;
       }
    if(curr==real_tail){
        break;
      }
        curr=nn;
    }
