            return head;
        }

        Node max=head.next;
        int index=1;
        while(max!=null){
            String book=(String)max.elem;
            int bookPop=popularity[index];

            Node min=head;
            int  minIdx=0;
            while(min.next!=max){
                min=min.next;
                minIdx++;
            }
            while(minIdx>=0 && popularity[minIdx]<bookPop){
                Node temp=head;
                for(int k=0;k<minIdx;k++){
                    temp=temp.next;
                }
                temp.next.elem=temp.elem;

                popularity[minIdx+1]=popularity[minIdx];
                minIdx--;
            }
            Node temp=head;
            for(int k=0;k<minIdx+1;k++){
                temp=temp.next;
            }
            temp.elem=book;
            popularity[minIdx+1]=bookPop;

            max=max.next;
            index++;


        }
