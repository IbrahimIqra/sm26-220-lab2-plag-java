public static String checkSimilar( Node building1, Node building2 ){
        
        Node h1= building1;
        Node h2= building2;

        int count1= 0;
        int count2= 0;
        while(h1!=null){
            count1++;
            h1=h1.next;
        }
        while(h2!=null){
            count2++;
            h2=h2.next;
        }

        if(count1!=count2){
            return "Not Similar" ;
        }
        else{
            h1= building1;
            h2= building2;



            while(h1!=null){
                if(!h1.elem.equals(h2.elem)){
                    return "Not Similar" ;
                }
                h1= h1.next;
                h2= h2.next;
                
            }
            return "Similar" ;
            }
        }

public static Node organizeBooks(Node head, Integer[] popularity) {
            int n = popularity.length;

    for (int i=0;i<n-1;i++) {
        Node cur= head;
        for (int j=0;j<n-i-1;j++) {
            if (popularity[j] < popularity[j + 1]) {
                int temp = popularity[j];
                popularity[j] = popularity[j + 1];
                popularity[j + 1] = temp;

                Object b = cur.elem;
                cur.elem = cur.next.elem;
                cur.next.elem = b;
            }



            cur = cur.next;
        }
    }
    return head;

    
    }

public static Node alternateMerge( Node head1, Node head2 ){
        Node n1= head1;
        Node n2= head2;
    while (n1!=null && n2!=null) {
        Node next1= n1.next;
        Node next2= n2.next;
        n1.next= n2;
        if (next1==null) {
            break;
        }
        n2.next= next1;
        n1= next1;
        n2= next2;
    }
    return head1;


    }

public static void sumOddAppend(Node dh) {
        Node dH= dh;
        Node n= dH.next;
        int sum= 0;
        while(n!=dh){
            if((int)n.elem %2!=0){
                sum+= (int)n.elem;
                dH.next= n.next;
                n=n.next; }
            else{                         //if the value is even
                dH= n;
                n= n.next;}   
        }
        Node newNode= new Node(sum);      //Creating a new node for the sum
        dH=dh;
        n= dH.next;
        while(n.next!=dH){


            n= n.next;}
        n.next=newNode;               //adding the new node at the last
        newNode.next= dH;


        
    }

public static void pairJoin(DNode dh1, DNode dh2) {
        DNode n1= dh1.next;
        DNode n2= dh2.next;
        
        while(n1!=dh1 && n2!=dh2){       //Since both are equal
            DNode next1= n1.next;
            DNode next2= n2.next;

            n1.next=n2;
            n2.prev= n1;
            if(next1==null){
                dh1.prev = n2;
                n2.next = dh1;
                
                break;
            }


            n2.next= next1;
            next1.prev= n2;
            n1= next1;
            n2= next2;
        }
    }

public static void rangeMove(DNode dh, int start, int end) {
        DNode dH= dh;
        DNode s= dH.next;
        DNode e= dH.prev;   //will use it in the loop
        DNode last= dH.prev;    //will use it to stop the loop
        
        int r1= start;
        int r2= end;

        while(s!=dH){
            DNode nextS= s.next;                         //storing 
           if(r1<= (int)s.elem && r2>=(int)s.elem){


            if(s!=e){                                  //if it is not already at last
            s.prev.next= s.next;
            s.next.prev= s.prev;

            e.next= s;
            s.prev= e;
            s.next= dH;
            dH.prev= s;
            e= s;                      //updating prev of the dummy head

            }
              
           }
           if(s==last){        //Stopping the loop 
            break;
           }
           s= nextS;            //Entering the stored value
        } 

    }
